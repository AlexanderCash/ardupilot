129 uint16_t AP_InertialSensor_LSM9DS0::_init_sensor( Sample_rate sample_rate)
130 {
131  if (_initialised) return _lsm9ds0_product_id;
132  _initialised = true;
133 
134  _spi = hal.spi->device(AP_HAL::SPIDevice_LSM9DS0_AM);
135  _spi_sem = _spi->get_semaphore();
136 
137  _drdy_pin_a = hal.gpio->channel(BBB_P8_8);
138  _drdy_pin_m = hal.gpio->channel(BBB_P8_10);
139  _drdy_pin_g = hal.gpio->channel(BBB_P8_34);
140 
141  // For some reason configuring the pins as an inputs make the driver fail
142  // _drdy_pin_a->mode(GPIO_IN);
143  // _drdy_pin_m->mode(GPIO_IN);
144  // _drdy_pin_g->mode(GPIO_IN);
145 
146  hal.scheduler->suspend_timer_procs();
147 
148  uint8_t tries = 0;
149  do {
150  bool success = _hardware_init(sample_rate);
151  if (success) {
152  hal.scheduler->delay(5+2);
153  if (!_spi_sem->take(100)) {
154  hal.scheduler->panic(PSTR("LSM9DS0: Unable to get semaphore"));
155  }
156  if (_data_ready()) {
157  _spi_sem->give();
158  break;
159  } else {
160  hal.console->println_P(
161  PSTR("LSM9DS0 startup failed: no data ready"));
162  }
163  _spi_sem->give();
164  }
165  if (tries++ > 5) {
166  hal.scheduler->panic(PSTR("PANIC: failed to boot LSM9DS0 5 times"));
167  }
168  } while (1);
169 
170  hal.scheduler->resume_timer_procs();
171 
172  /* read the first lot of data.
173  * _read_data_transaction requires the spi semaphore to be taken by
174  * its caller. */
175  _last_sample_time_micros = hal.scheduler->micros();
176  hal.scheduler->delay(10);
177  if (_spi_sem->take(100)) {
178  _read_data_transaction_g();
179  _read_data_transaction_xm();
180  _spi_sem->give();
181  }
182 
183  // start the timer process to read samples
184  hal.scheduler->register_timer_process(AP_HAL_MEMBERPROC(&AP_InertialSensor_LSM9DS0::_poll_data));
185 
186 #if LSM9DS0_DEBUG
187  _dump_registers();
188 #endif
189  return _lsm9ds0_product_id;
190 }

203  if (_sample_available()) {
204  return true;
205  }
206  }
207  return false;

383 void AP_InertialSensor_LSM9DS0::_poll_data(void)
384 {
385  if (hal.scheduler->in_timerprocess()) {
386  if (!_spi_sem->take_nonblocking()) {
387  /*
388  the semaphore being busy is an expected condition when the
389  mainline code is calling wait_for_sample() which will
390  grab the semaphore. We return now and rely on the mainline
391  code grabbing the latest sample.
392  */
393  return;
394  }
395  if (_data_ready() == 1) {
396  _last_sample_time_micros = hal.scheduler->micros();
397  _read_data_transaction_g();
398  } else if (_data_ready() == 2){
399  _last_sample_time_micros = hal.scheduler->micros();
400  _read_data_transaction_xm();
401  } else if (_data_ready() == 3){
402  _last_sample_time_micros = hal.scheduler->micros();
403  _read_data_transaction_g();
404  _read_data_transaction_xm();
405  }
406  _spi_sem->give();
407  } else {
408  /* Synchronous read - take semaphore */
409  if (_spi_sem->take(10)) {
410  if (_data_ready() == 1) {
411  _last_sample_time_micros = hal.scheduler->micros();
412  _read_data_transaction_g();
413  } else if (_data_ready() == 2){
414  _last_sample_time_micros = hal.scheduler->micros();
415  _read_data_transaction_xm();
416  } else if (_data_ready() == 3){
417  _last_sample_time_micros = hal.scheduler->micros();
418  _read_data_transaction_g();
419  _read_data_transaction_xm();
420  }
421  _spi_sem->give();
422  } else {
423  hal.scheduler->panic(
424  PSTR("PANIC: AP_InertialSensor_LSM9DS0::_poll_data "
425  "failed to take SPI semaphore synchronously"));
426  }
427  }

431 void AP_InertialSensor_LSM9DS0::_read_data_transaction_g() {
432 
433  // read gyro values
434  uint8_t temp[6]; // We'll read six bytes from the gyro into temp
435  for (uint8_t i=0;i<6;i++){
436  temp[i] = _register_read_g(OUT_X_L_G + i);
437  }
438 
439  uint16_t gx = (temp[1] << 8) | temp[0]; // Store x-axis values into gx
440  uint16_t gy = (temp[3] << 8) | temp[2]; // Store y-axis values into gy
441  uint16_t gz = (temp[5] << 8) | temp[4]; // Store z-axis values into gz
442 
443  _gyro_sum.x += gx;
444  _gyro_sum.y += gy;
445  _gyro_sum.z -= gz;
446  _sum_count_g++;
447 
448  if (_sum_count_g == 0) {
449  _gyro_sum.zero();
450  }
451 }

578 void AP_InertialSensor_LSM9DS0::_initGyro()
579 {
580  /* CTRL_REG1_G sets output data rate, bandwidth, power-down and enables
581  Bits[7:0]: DR1 DR0 BW1 BW0 PD Zen Xen Yen
582  DR[1:0] - Output data rate selection
583  00=95Hz, 01=190Hz, 10=380Hz, 11=760Hz
584  BW[1:0] - Bandwidth selection (sets cutoff frequency)
585  Value depends on ODR. See datasheet table 21.
586  PD - Power down enable (0=power down mode, 1=normal or sleep mode)
587  Zen, Xen, Yen - Axis enable (o=disabled, 1=enabled) */
588  _register_write_g(CTRL_REG1_G, 0x0F); // Normal mode, enable all axes
589  hal.scheduler->delay(1);
590 
591  /* CTRL_REG2_G sets up the HPF
592  Bits[7:0]: 0 0 HPM1 HPM0 HPCF3 HPCF2 HPCF1 HPCF0
593  HPM[1:0] - High pass filter mode selection
594  00=normal (reset reading HP_RESET_FILTER, 01=ref signal for filtering,
595  10=normal, 11=autoreset on interrupt
596  HPCF[3:0] - High pass filter cutoff frequency
597  Value depends on data rate. See datasheet table 26.
598  */
599  _register_write_g(CTRL_REG2_G, 0x00); // Normal mode, high cutoff frequency
600  hal.scheduler->delay(1);
601 
602  /* CTRL_REG3_G sets up interrupt and DRDY_G pins
603  Bits[7:0]: I1_IINT1 I1_BOOT H_LACTIVE PP_OD I2_DRDY I2_WTM I2_ORUN I2_EMPTY
604  I1_INT1 - Interrupt enable on INT_G pin (0=disable, 1=enable)
605  I1_BOOT - Boot status available on INT_G (0=disable, 1=enable)
606  H_LACTIVE - Interrupt active configuration on INT_G (0:high, 1:low)

771  // algorithm to calculate g/(ADC tick) based on that 3-bit value:
772  _aRes = aScl == A_SCALE_16G ? 16.0f / 32768.0f :
