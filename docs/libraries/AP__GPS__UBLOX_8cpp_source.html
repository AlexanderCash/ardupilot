80 /*
81  send the next step of rate updates to the GPS. This reconfigures the
82  GPS on the fly to have the right message rates. It needs to be
83  careful to only send a message if there is sufficient buffer space
84  available on the serial port to avoid it blocking the CPU
85  */
86 void
87 AP_GPS_UBLOX::send_next_rate_update(void)
88 {
89  if (port->txspace() < (int16_t)(sizeof(struct ubx_header)+sizeof(struct ubx_cfg_nav_rate)+2)) {
90  // not enough space - do it next time
91  return;
92  }
93 
94  //hal.console->printf_P(PSTR("next_rate: %u\n"), (unsigned)rate_update_step);
95 
96  switch (rate_update_step++) {
97  case 0:
98  _configure_navigation_rate(200);
99  break;
100  case 1:
101  _configure_message_rate(CLASS_NAV, MSG_POSLLH, 1); // 28+8 bytes
102  break;
103  case 2:
104  _configure_message_rate(CLASS_NAV, MSG_STATUS, 1); // 16+8 bytes
105  break;
106  case 3:
107  _configure_message_rate(CLASS_NAV, MSG_SOL, 1); // 52+8 bytes
108  break;
109  case 4:
110  _configure_message_rate(CLASS_NAV, MSG_VELNED, 1); // 36+8 bytes
111  break;
112 #if UBLOX_HW_LOGGING
113  case 5:
114  // gather MON_HW at 0.5Hz
115  _configure_message_rate(CLASS_MON, MSG_MON_HW, 2); // 64+8 bytes
116  break;
117  case 6:
118  // gather MON_HW2 at 0.5Hz
119  _configure_message_rate(CLASS_MON, MSG_MON_HW2, 2); // 24+8 bytes
120  break;
121 #endif
122 #if UBLOX_VERSION_AUTODETECTION
123  case 7:
124  _request_version();
125  break;
126 #endif
127 #if UBLOX_RXM_RAW_LOGGING
128  case 8:
129  _configure_message_rate(CLASS_RXM, MSG_RXM_RAW, gps._raw_data); // 24*16+8 bytes
130  break;
131 #endif
132  default:
133  need_rate_update = false;
134  rate_update_step = 0;
135  break;
136  }
137 }

140 // Process bytes available from the stream
141 //
142 // The stream is assumed to contain only messages we recognise. If it
143 // contains other messages, and those messages contain the preamble
144 // bytes, it is possible for this code to fail to synchronise to the
145 // stream immediately. Without buffering the entire message and
146 // re-processing it from the top, this is unavoidable. The parser
147 // attempts to avoid this when possible.
148 //
149 bool
150 AP_GPS_UBLOX::read(void)
151 {
152  uint8_t data;
153  int16_t numc;
154  bool parsed = false;
155 
156  if (need_rate_update) {
157  send_next_rate_update();
158  }
159 
160  numc = port->available();
161  for (int16_t i = 0; i < numc; i++) { // Process bytes received
162 
163  // read the next byte
164  data = port->read();
165 
166  reset:
167  switch(_step) {
168 
169  // Message preamble detection
170  //
171  // If we fail to match any of the expected bytes, we reset
172  // the state machine and re-consider the failed byte as
173  // the first byte of the preamble. This improves our
174  // chances of recovering from a mismatch and makes it less
175  // likely that we will be fooled by the preamble appearing
176  // as data in some other message.
177  //
178  case 1:
179  if (PREAMBLE2 == data) {
180  _step++;
181  break;
182  }
183  _step = 0;
184  Debug("reset %u", __LINE__);
185  // FALLTHROUGH
186  case 0:
187  if(PREAMBLE1 == data)
188  _step++;
189  break;
190 
191  // Message header processing
192  //
193  // We sniff the class and message ID to decide whether we
194  // are going to gather the message bytes or just discard
195  // them.
196  //
197  // We always collect the length so that we can avoid being
198  // fooled by preamble bytes in messages.
199  //
200  case 2:
201  _step++;
202  _class = data;
203  _ck_b = _ck_a = data; // reset the checksum accumulators
204  break;
205  case 3:
206  _step++;
207  _ck_b += (_ck_a += data); // checksum byte
208  _msg_id = data;
209  break;
210  case 4:
211  _step++;
212  _ck_b += (_ck_a += data); // checksum byte
213  _payload_length = data; // payload length low byte
214  break;
215  case 5:
216  _step++;
217  _ck_b += (_ck_a += data); // checksum byte
218 
219  _payload_length += (uint16_t)(data<<8);
220  if (_payload_length > 512) {
221  Debug("large payload %u", (unsigned)_payload_length);
222  // assume very large payloads are line noise
223  _payload_length = 0;
224  _step = 0;
225  goto reset;
226  }
227  _payload_counter = 0; // prepare to receive payload
228  break;
229 
230  // Receive message data
231  //
232  case 6:
233  _ck_b += (_ck_a += data); // checksum byte
234  if (_payload_counter < sizeof(_buffer)) {
235  _buffer.bytes[_payload_counter] = data;
236  }
237  if (++_payload_counter == _payload_length)
238  _step++;
239  break;
240 
241  // Checksum and message processing
242  //
243  case 7:
244  _step++;
245  if (_ck_a != data) {
246  Debug("bad cka %x should be %x", data, _ck_a);
247  _step = 0;
248  goto reset;
249  }
250  break;
251  case 8:
252  _step = 0;
253  if (_ck_b != data) {
254  Debug("bad ckb %x should be %x", data, _ck_b);
255  break; // bad checksum
256  }
257 
258  if (_parse_gps()) {
259  parsed = true;
260  }
261  }
262  }
263  return parsed;
264 }

366 bool
367 AP_GPS_UBLOX::_parse_gps(void)
368 {
369  if (_class == CLASS_ACK) {
370  Debug("ACK %u", (unsigned)_msg_id);
371  return false;
372  }
373 
374  if (_class == CLASS_CFG && _msg_id == MSG_CFG_NAV_SETTINGS) {
375  Debug("Got settings %u min_elev %d drLimit %u\n",
376  (unsigned)_buffer.nav_settings.dynModel,
377  (int)_buffer.nav_settings.minElev,
378  (unsigned)_buffer.nav_settings.drLimit);
379  _buffer.nav_settings.mask = 0;
380  if (gps._navfilter != AP_GPS::GPS_ENGINE_NONE &&
381  _buffer.nav_settings.dynModel != gps._navfilter) {
382  // we've received the current nav settings, change the engine
383  // settings and send them back
384  Debug("Changing engine setting from %u to %u\n",
385  (unsigned)_buffer.nav_settings.dynModel, (unsigned)gps._navfilter);
386  _buffer.nav_settings.dynModel = gps._navfilter;
387  _buffer.nav_settings.mask |= 1;
388  }
389  if (gps._min_elevation != -100 &&
390  _buffer.nav_settings.minElev != gps._min_elevation) {
391  Debug("Changing min elevation to %d\n", (int)gps._min_elevation);
392  _buffer.nav_settings.minElev = gps._min_elevation;
393  _buffer.nav_settings.mask |= 2;
394  }
395  if (_buffer.nav_settings.mask != 0) {
396  _send_message(CLASS_CFG, MSG_CFG_NAV_SETTINGS,
397  &_buffer.nav_settings,
398  sizeof(_buffer.nav_settings));
399  }
400  return false;
401  }
402 
403  if (_class == CLASS_CFG && _msg_id == MSG_CFG_SBAS && gps._sbas_mode != 2) {
404  Debug("Got SBAS settings %u %u %u 0x%x 0x%x\n",
405  (unsigned)_buffer.sbas.mode,
406  (unsigned)_buffer.sbas.usage,
407  (unsigned)_buffer.sbas.maxSBAS,
408  (unsigned)_buffer.sbas.scanmode2,
409  (unsigned)_buffer.sbas.scanmode1);
410  if (_buffer.sbas.mode != gps._sbas_mode) {
411  _buffer.sbas.mode = gps._sbas_mode;
412  _send_message(CLASS_CFG, MSG_CFG_SBAS,
413  &_buffer.sbas,
414  sizeof(_buffer.sbas));
415  }
416  }
417 
418 #if UBLOX_HW_LOGGING
419  if (_class == CLASS_MON) {
420  if (_msg_id == MSG_MON_HW) {
421  if (_payload_length == 60 || _payload_length == 68) {
422  log_mon_hw();
423  }
424  } else if (_msg_id == MSG_MON_HW2) {
425  if (_payload_length == 28) {
426  log_mon_hw2();
427  }
428  } else {
429  unexpected_message();
430  }
431  return false;
432  }
433 #endif // UBLOX_HW_LOGGING
434 
435 #if UBLOX_RXM_RAW_LOGGING
436  if (_class == CLASS_RXM && _msg_id == MSG_RXM_RAW && gps._raw_data != 0) {
437  log_rxm_raw(_buffer.rxm_raw);
438  return false;
439  }
440 #endif // UBLOX_RXM_RAW_LOGGING
441 
442  if (_class != CLASS_NAV) {
443  unexpected_message();
444  return false;
445  }
446 
447  switch (_msg_id) {
448  case MSG_POSLLH:
449  Debug("MSG_POSLLH next_fix=%u", next_fix);
450  _last_pos_time = _buffer.posllh.time;
451  state.location.lng = _buffer.posllh.longitude;
452  state.location.lat = _buffer.posllh.latitude;
453  state.location.alt = _buffer.posllh.altitude_msl / 10;
454  state.status = next_fix;
455  _new_position = true;
456 #if UBLOX_FAKE_3DLOCK
457  state.location.lng = 1491652300L;
458  state.location.lat = -353632610L;
459  state.location.alt = 58400;
460 #endif
461  state.horizontal_accuracy = _buffer.posllh.horizontal_accuracy*1.0e-3f;
462  state.vertical_accuracy = _buffer.posllh.vertical_accuracy*1.0e-3f;
463  state.have_horizontal_accuracy = true;
464  state.have_vertical_accuracy = true;
465  break;
466  case MSG_STATUS:
467  Debug("MSG_STATUS fix_status=%u fix_type=%u",
468  _buffer.status.fix_status,
469  _buffer.status.fix_type);
470  if (_buffer.status.fix_status & NAV_STATUS_FIX_VALID) {
471  if( _buffer.status.fix_type == AP_GPS_UBLOX::FIX_3D) {
472  next_fix = AP_GPS::GPS_OK_FIX_3D;
473  }else if (_buffer.status.fix_type == AP_GPS_UBLOX::FIX_2D) {
474  next_fix = AP_GPS::GPS_OK_FIX_2D;
475  }else{
476  next_fix = AP_GPS::NO_FIX;
477  state.status = AP_GPS::NO_FIX;
478  }
479  }else{
480  next_fix = AP_GPS::NO_FIX;
481  state.status = AP_GPS::NO_FIX;
482  }
483 #if UBLOX_FAKE_3DLOCK
484  state.status = AP_GPS::GPS_OK_FIX_3D;
485  next_fix = state.status;
486 #endif
487  break;
488  case MSG_SOL:
489  Debug("MSG_SOL fix_status=%u fix_type=%u",
490  _buffer.solution.fix_status,
491  _buffer.solution.fix_type);
492  if (_buffer.solution.fix_status & NAV_STATUS_FIX_VALID) {
493  if( _buffer.solution.fix_type == AP_GPS_UBLOX::FIX_3D) {
494  next_fix = AP_GPS::GPS_OK_FIX_3D;
495  }else if (_buffer.solution.fix_type == AP_GPS_UBLOX::FIX_2D) {
496  next_fix = AP_GPS::GPS_OK_FIX_2D;
497  }else{
498  next_fix = AP_GPS::NO_FIX;
499  state.status = AP_GPS::NO_FIX;
500  }
501  }else{
502  next_fix = AP_GPS::NO_FIX;
503  state.status = AP_GPS::NO_FIX;
504  }
505  state.num_sats = _buffer.solution.satellites;
506  state.hdop = _buffer.solution.position_DOP;
507  if (next_fix >= AP_GPS::GPS_OK_FIX_2D) {
508  state.last_gps_time_ms = hal.scheduler->millis();
509  if (state.time_week == _buffer.solution.week &&
510  state.time_week_ms + 200 == _buffer.solution.time) {
511  // we got a 5Hz update. This relies on the way
512  // that uBlox gives timestamps that are always
513  // multiples of 200 for 5Hz
514  _last_5hz_time = state.last_gps_time_ms;
515  }
516  state.time_week_ms = _buffer.solution.time;
517  state.time_week = _buffer.solution.week;
518  }
519 #if UBLOX_FAKE_3DLOCK
520  next_fix = state.status;
521  state.num_sats = 10;
522  state.hdop = 200;
523  state.time_week = 1721;
524  state.time_week_ms = hal.scheduler->millis() + 3*60*60*1000 + 37000;
525  state.last_gps_time_ms = hal.scheduler->millis();
526 #endif
527  break;
528  case MSG_VELNED:
529  Debug("MSG_VELNED");
530  _last_vel_time = _buffer.velned.time;
531  state.ground_speed = _buffer.velned.speed_2d*0.01f; // m/s
532  state.ground_course_cd = _buffer.velned.heading_2d / 1000; // Heading 2D deg * 100000 rescaled to deg * 100
533  state.have_vertical_velocity = true;
534  state.velocity.x = _buffer.velned.ned_north * 0.01f;
535  state.velocity.y = _buffer.velned.ned_east * 0.01f;
536  state.velocity.z = _buffer.velned.ned_down * 0.01f;
537  state.have_speed_accuracy = true;
538  state.speed_accuracy = _buffer.velned.speed_accuracy*0.01f;
539  _new_speed = true;
540  break;
541 #if UBLOX_VERSION_AUTODETECTION
542  case MSG_NAV_SVINFO:
543  {
544  Debug("MSG_NAV_SVINFO\n");
545  static const uint8_t HardwareGenerationMask = 0x07;
546  uint8_t hardware_generation = _buffer.svinfo_header.globalFlags & HardwareGenerationMask;
547  switch (hardware_generation) {
548  case UBLOX_5:
549  case UBLOX_6:
550  /*speed already configured */;
551  break;
552  case UBLOX_7:
553  case UBLOX_M8:
554  port->begin(4000000U);
555  Debug("Changed speed to 5Mhzfor SPI-driven UBlox\n");
556  break;
557  default:
558  hal.console->printf("Wrong Ublox' Hardware Version%u\n", hardware_generation);
559  break;
560  };
561  /* We don't need that anymore */
562  _configure_message_rate(CLASS_NAV, MSG_NAV_SVINFO, 0);
563  break;
564  }
565 #endif
566  default:
567  Debug("Unexpected NAV message 0x%02x", (unsigned)_msg_id);
568  if (++_disable_counter == 0) {
569  Debug("Disabling NAV message 0x%02x", (unsigned)_msg_id);
570  _configure_message_rate(CLASS_NAV, _msg_id, 0);
571  }
572  return false;
573  }
574 
575  // we only return true when we get new position and speed data
576  // this ensures we don't use stale data
577  if (_new_position && _new_speed && _last_vel_time == _last_pos_time) {
578  _new_speed = _new_position = false;
579  _fix_count++;
580  if ((hal.scheduler->millis() - _last_5hz_time) > 15000U && !need_rate_update) {
581  // the GPS is running slow. It possibly browned out and
582  // restarted with incorrect parameters. We will slowly
583  // send out new parameters to fix it
584  need_rate_update = true;
585  rate_update_step = 0;
586  _last_5hz_time = hal.scheduler->millis();
587  }
588 
589  if (_fix_count == 50 && gps._sbas_mode != 2) {
590  // ask for SBAS settings every 20 seconds
591  Debug("Asking for SBAS setting\n");
592  _send_message(CLASS_CFG, MSG_CFG_SBAS, NULL, 0);
593  }
594  if (_fix_count == 100) {
595  // ask for nav settings every 20 seconds
596  Debug("Asking for engine setting\n");
597  _send_message(CLASS_CFG, MSG_CFG_NAV_SETTINGS, NULL, 0);
598  _fix_count = 0;
599  }
600 
601 #if UBLOX_HW_LOGGING
602  log_accuracy();
603 #endif //UBLOX_HW_LOGGING
604 
605  return true;
606  }
607  return false;
608 }

614  * update checksum for a set of bytes

616 void
617 AP_GPS_UBLOX::_update_checksum(uint8_t *data, uint16_t len, uint8_t &ck_a, uint8_t &ck_b)

697 /*
698  detect a Ublox GPS. Adds one byte, and returns true if the stream
699  matches a UBlox
700  */
701 bool
702 AP_GPS_UBLOX::_detect(struct UBLOX_detect_state &state, uint8_t data)
703 {
704 reset:
705  switch (state.step) {
706  case 1:
707  if (PREAMBLE2 == data) {
708  state.step++;
709  break;
710  }
711  state.step = 0;
712  case 0:
713  if (PREAMBLE1 == data)
714  state.step++;
715  break;
716  case 2:
717  state.step++;
718  state.ck_b = state.ck_a = data;
719  break;
720  case 3:
721  state.step++;
722  state.ck_b += (state.ck_a += data);
723  break;
724  case 4:
725  state.step++;
726  state.ck_b += (state.ck_a += data);
727  state.payload_length = data;
728  break;
729  case 5:
730  state.step++;
731  state.ck_b += (state.ck_a += data);
732  state.payload_counter = 0;
733  break;
734  case 6:
735  state.ck_b += (state.ck_a += data);
736  if (++state.payload_counter == state.payload_length)
737  state.step++;
738  break;
739  case 7:
740  state.step++;
741  if (state.ck_a != data) {
742  state.step = 0;
743  goto reset;
744  }
745  break;
746  case 8:
747  state.step = 0;
748  if (state.ck_b == data) {
749  // a valid UBlox packet
750  return true;
751  } else {
752  goto reset;
753  }
754  }
755  return false;
756 }
