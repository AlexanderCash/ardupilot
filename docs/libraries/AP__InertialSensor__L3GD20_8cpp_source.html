297 void AP_InertialSensor_L3GD20::_poll_data(void)
298 {
299  if (hal.scheduler->in_timerprocess()) {
300  if (!_spi_sem->take_nonblocking()) {
301  /*
302  the semaphore being busy is an expected condition when the
303  mainline code is calling wait_for_sample() which will
304  grab the semaphore. We return now and rely on the mainline
305  code grabbing the latest sample.
306  */
307  return;
308  }
309  if (_data_ready()) {
310  _last_sample_time_micros = hal.scheduler->micros();
311  _read_data_transaction();
312  }
313  _spi_sem->give();
314  } else {
315  /* Synchronous read - take semaphore */
316  if (_spi_sem->take(10)) {
317  if (_data_ready()) {
318  _last_sample_time_micros = hal.scheduler->micros();
319  _read_data_transaction();
320  }
321  _spi_sem->give();
322  } else {
323  hal.scheduler->panic(
324  PSTR("PANIC: AP_InertialSensor_L3GD20::_poll_data "
325  "failed to take SPI semaphore synchronously"));
326  }
327  }

330 void AP_InertialSensor_L3GD20::_read_data_transaction() {
331 
332  struct {
333  uint8_t cmd;
334  uint8_t temp;
335  uint8_t status;
336  int16_t x;
337  int16_t y;
338  int16_t z;
339  } raw_report;
340 
341  /* fetch data from the sensor */
342  memset(&raw_report, 0, sizeof(raw_report));
343  raw_report.cmd = ADDR_OUT_TEMP | DIR_READ | ADDR_INCREMENT;
344  _spi->transaction((uint8_t *)&raw_report, (uint8_t *)&raw_report, sizeof(raw_report));
345 
346 #if L3GD20_USE_DRDY
347  if ((raw_report.status & 0xF) != 0xF) {
348  /*
349  we waited for DRDY, but did not see DRDY on all axes
350  when we captured. That means a transfer error of some sort
351  */
352  hal.console->println("L3GD20: DRDY is not on in all axes, transfer error");
353  return;
354  }
355 #endif
356  _gyro_sum.x += raw_report.x;
357  _gyro_sum.y += raw_report.y;
358  _gyro_sum.z -= raw_report.z;
359  _sum_count++;
360 
361  if (_sum_count == 0) {
362  // rollover - v unlikely
363  _gyro_sum.zero();
364  }
365 }
