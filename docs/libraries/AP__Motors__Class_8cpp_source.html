236 // update_max_throttle - updates the limits on _max_throttle if necessary taking into account slow_start_throttle flag
237 void AP_Motors::update_max_throttle()
238 {
239  // ramp up minimum spin speed if necessary
240  if (_flags.slow_start_low_end) {
241  _spin_when_armed_ramped += AP_MOTOR_SLOW_START_LOW_END_INCREMENT;
242  if (_spin_when_armed_ramped >= _spin_when_armed) {
243  _spin_when_armed_ramped = _spin_when_armed;
244  _flags.slow_start_low_end = false;
245  }
246  }
247 
248  // implement slow start
249  if (_flags.slow_start) {
250  // increase slow start throttle
251  _max_throttle += AP_MOTOR_SLOW_START_INCREMENT;
252 
253  // turn off slow start if we've reached max throttle
254  if (_max_throttle >= _rc_throttle.servo_out) {
255  _max_throttle = AP_MOTORS_DEFAULT_MAX_THROTTLE;
256  _flags.slow_start = false;
257  }
258  return;
259  }
260 
261  // current limit throttle
262  current_limit_max_throttle();
263 }

265 // current_limit_max_throttle - limits maximum throttle based on current
266 void AP_Motors::current_limit_max_throttle()
267 {
268  // return maximum if current limiting is disabled
269  if (_batt_current_max <= 0) {
270  _throttle_limit = 1.0f;
271  _max_throttle = AP_MOTORS_DEFAULT_MAX_THROTTLE;
272  return;
273  }
274 
275  // remove throttle limit if throttle is at zero or disarmed
276  if(_rc_throttle.servo_out <= 0 || !_flags.armed) {
277  _throttle_limit = 1.0f;
278  }
279 
280  // limit throttle if over current
281  if (_batt_current > _batt_current_max*1.25f) {
282  // Fast drop for extreme over current (1 second)
283  _throttle_limit -= 1.0f/_loop_rate;
284  } else if(_batt_current > _batt_current_max) {
285  // Slow drop for extreme over current (5 second)
286  _throttle_limit -= 0.2f/_loop_rate;
287  } else {
288  // Increase throttle limit (2 second)
289  _throttle_limit += 0.5f/_loop_rate;
290  }
291 
292  // throttle limit drops to 20% between hover and full throttle
293  _throttle_limit = constrain_float(_throttle_limit, 0.2f, 1.0f);
294 
295  // limit max throttle
296  _max_throttle = _hover_out + ((1000-_hover_out)*_throttle_limit);
297 }

299 // apply_thrust_curve_and_volt_scaling - returns throttle curve adjusted pwm value (i.e. 1000 ~ 2000)
300 int16_t AP_Motors::apply_thrust_curve_and_volt_scaling(int16_t pwm_out, int16_t pwm_min, int16_t pwm_max) const
301 {
302  // convert to 0.0 to 1.0 ratio
303  float throttle_ratio = ((float)(pwm_out-pwm_min))/((float)(pwm_max-pwm_min));
304 
305  // apply thrust curve - domain 0.0 to 1.0, range 0.0 to 1.0
306  if (_thrust_curve_expo > 0.0f){
307  throttle_ratio = ((_thrust_curve_expo-1.0f) + safe_sqrt((1.0f-_thrust_curve_expo)*(1.0f-_thrust_curve_expo) + 4.0f*_thrust_curve_expo*_lift_max*throttle_ratio))/(2.0f*_thrust_curve_expo*_batt_voltage_filt.get());
308  }
309 
310  // scale to maximum thrust point
311  throttle_ratio *= _thrust_curve_max;
312 
313  // convert back to pwm range, constrain and return
314  return (int16_t)constrain_float(throttle_ratio*(pwm_max-pwm_min)+pwm_min, pwm_min, (pwm_max-pwm_min)*_thrust_curve_max+pwm_min);
315 }

317 // update_lift_max from battery voltage - used for voltage compensation
318 void AP_Motors::update_lift_max_from_batt_voltage()
319 {
320  // sanity check battery_voltage_min is not too small
321  // if disabled or misconfigured exit immediately
322  if((_batt_voltage_max <= 0) || (_batt_voltage_min >= _batt_voltage_max) || (_batt_voltage < 0.25f*_batt_voltage_min)) {
323  _batt_voltage_filt.reset(1.0f);
324  _lift_max = 1.0f;
325  return;
326  }
327 
328  _batt_voltage_min = max(_batt_voltage_min, _batt_voltage_max * 0.6f);
329 
330  // add current based voltage sag to battery voltage
331  float batt_voltage = _batt_voltage + _batt_current * _batt_resistance;
332  batt_voltage = constrain_float(batt_voltage, _batt_voltage_min, _batt_voltage_max);
333 
334  // filter at 0.5 Hz
335  float bvf = _batt_voltage_filt.apply(batt_voltage/_batt_voltage_max, 1.0f/_loop_rate);
336 
337  // calculate lift max
338  _lift_max = bvf*(1-_thrust_curve_expo) + _thrust_curve_expo*bvf*bvf;
339 }

341 // update_battery_resistance - calculate battery resistance when throttle is above hover_out
342 void AP_Motors::update_battery_resistance()
343 {
344  // if motors are stopped, reset resting voltage and current
345  if (_rc_throttle.servo_out <= 0 || !_flags.armed) {
346  _batt_voltage_resting = _batt_voltage;
347  _batt_current_resting = _batt_current;
348  _batt_timer = 0;
349  } else {
350  // update battery resistance when throttle is over hover throttle
351  if ((_batt_timer < 400) && ((_batt_current_resting*2.0f) < _batt_current)) {
352  if (_rc_throttle.servo_out >= _hover_out) {
353  // filter reaches 90% in 1/4 the test time
354  _batt_resistance += 0.05f*(( (_batt_voltage_resting-_batt_voltage)/(_batt_current-_batt_current_resting) ) - _batt_resistance);
355  _batt_timer += 1;
356  } else {
357  // initialize battery resistance to prevent change in resting voltage estimate
358  _batt_resistance = ((_batt_voltage_resting-_batt_voltage)/(_batt_current-_batt_current_resting));
359  }
360  }
361  }
362 }

365 void AP_Motors::update_throttle_thr_mix()
366 {
367  // slew _throttle_thr_mix to _throttle_thr_mix_desired
368  if (_throttle_thr_mix < _throttle_thr_mix_desired) {
369  // increase quickly (i.e. from 0.1 to 0.9 in 0.4 seconds)

374  }
