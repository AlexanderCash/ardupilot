267 void *LinuxScheduler::_timer_thread(void* arg)
268 {
269  LinuxScheduler* sched = (LinuxScheduler *)arg;
270 
271  while (sched->system_initializing()) {
272  poll(NULL, 0, 1);
273  }
274  /*
275  this aims to run at an average of 1kHz, so that it can be used
276  to drive 1kHz processes without drift
277  */
278  uint64_t next_run_usec = sched->micros64() + 1000;
279  while (true) {
280  uint64_t dt = next_run_usec - sched->micros64();
281  if (dt > 2000) {
282  // we've lost sync - restart
283  next_run_usec = sched->micros64();
284  } else {
285  sched->_microsleep(dt);
286  }
287  next_run_usec += 1000;
288  // run registered timers
289  sched->_run_timers(true);
290  }
291  return NULL;
292 }
