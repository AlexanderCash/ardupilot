29 // DigitalLPF implements the filter math
30 class DigitalLPF
31 {
32 public:
33  // constructor
34  DigitalLPF() :
35  _output(0.0f) {}
36 
37  struct lpf_params {
38  float cutoff_freq;
39  float sample_freq;
40  float alpha;
41  };
42 
43  // add a new raw value to the filter, retrieve the filtered result
44  float apply(float sample, float cutoff_freq, float dt) {
45  if (cutoff_freq <= 0.0f || dt <= 0.0f) {
46  _output = sample;
47  return _output;
48  }
49 
50  float rc = 1.0f/(M_2PI_F*cutoff_freq);
51  float alpha = constrain_float(dt/(dt+rc), 0.0f, 1.0f);
52  _output += (sample - _output) * alpha;
53  return _output;
54  }
55 
56  // get latest filtered value from filter (equal to the value returned by latest call to apply method)
57  float get() const {
58  return _output;
59  }
60 
61  void reset(float value) { _output = value; }
62 
63 private:
64  float _output;
65 };
