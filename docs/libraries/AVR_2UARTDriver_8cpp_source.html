59 void AVRUARTDriver::begin(uint32_t baud, uint16_t rxSpace, uint16_t txSpace) {
60  uint16_t ubrr;
61  bool use_u2x = true;
62  bool need_allocate = true;
63 
64  // if we are currently open...
65  if (_open) {
66  // If the caller wants to preserve the buffer sizing, work out what
67  // it currently is...
68  if (0 == rxSpace)
69  rxSpace = _rxBuffer->mask + 1;
70  if (0 == txSpace)
71  txSpace = _txBuffer->mask + 1;
72 
73  if (rxSpace == (_rxBuffer->mask + 1U) &&
74  txSpace == (_txBuffer->mask + 1U)) {
75  // avoid re-allocating the buffers if possible
76  need_allocate = false;
77  *_ucsrb &= ~(_portEnableBits | _portTxBits);
78  } else {
79  // close the port in its current configuration, clears _open
80  end();
81  }
82  }
83 
84  if (need_allocate) {
85  // allocate buffers
86  if (!_allocBuffer(_rxBuffer, rxSpace ? : _default_rx_buffer_size)
87  || !_allocBuffer(_txBuffer, txSpace ? : _default_tx_buffer_size)) {
88  end();
89  return; // couldn't allocate buffers - fatal
90  }
91  }
92 
93  // reset buffer pointers
94  _txBuffer->head = _txBuffer->tail = 0;
95  _rxBuffer->head = _rxBuffer->tail = 0;
96 
97  // mark the port as open
98  _open = true;
99 
100  // If the user has supplied a new baud rate, compute the new UBRR value.
101  if (baud > 0) {
102 #if F_CPU == 16000000UL
103  // hardcoded exception for compatibility with the bootloader shipped
104  // with the Duemilanove and previous boards and the firmware on the 8U2
105  // on the Uno and Mega 2560.
106  if (baud == 57600)
107  use_u2x = false;
108 #endif
109 
110  if (use_u2x) {
111  *_ucsra = 1 << _u2x;
112  ubrr = (F_CPU / 4 / baud - 1) / 2;
113  } else {
114  *_ucsra = 0;
115  ubrr = (F_CPU / 8 / baud - 1) / 2;
116  }
117 
118  *_ubrrh = ubrr >> 8;
119  *_ubrrl = ubrr;
120  }
121 
122  *_ucsrb |= _portEnableBits;
123 
124 }

162 void AVRUARTDriver::flush(void) {
163  // don't reverse this or there may be problems if the RX interrupt
164  // occurs after reading the value of _rxBuffer->head but before writing
165  // the value to _rxBuffer->tail; the previous value of head
166  // may be written to tail, making it appear as if the buffer
167  // don't reverse this or there may be problems if the RX interrupt
168  // occurs after reading the value of head but before writing
169  // the value to tail; the previous value of rx_buffer_head
170  // may be written to tail, making it appear as if the buffer
171  // were full, not empty.
172  _rxBuffer->head = _rxBuffer->tail;
173 
174  // don't reverse this or there may be problems if the TX interrupt
175  // occurs after reading the value of _txBuffer->tail but before writing
176  // the value to _txBuffer->head.
177  _txBuffer->tail = _txBuffer->head;
178 }

180 size_t AVRUARTDriver::write(uint8_t c) {
181  uint8_t i;
182 
183  if (!_open) // drop bytes if not open
184  return 0;
185 
186  // wait for room in the tx buffer
187  i = (_txBuffer->head + 1) & _txBuffer->mask;
188 
189  // if the port is set into non-blocking mode, then drop the byte
190  // if there isn't enough room for it in the transmit buffer
191  if (_nonblocking_writes && i == _txBuffer->tail) {
192  return 0;
193  }
194 
195  while (i == _txBuffer->tail)
196  ;
197 
198  // add byte to the buffer
199  _txBuffer->bytes[_txBuffer->head] = c;
200  _txBuffer->head = i;
201 
202  // enable the data-ready interrupt, as it may be off if the buffer is empty
203  *_ucsrb |= _portTxBits;
204 
205  // return number of bytes written (always 1)
206  return 1;
207 }

212 size_t AVRUARTDriver::write(const uint8_t *buffer, size_t size)
213 {
214  if (!_open) {
215  return 0;
216  }
217 
218  if (!_nonblocking_writes) {
219  /*
220  use the per-byte delay loop in write() above for blocking writes
221  */
222  size_t ret = 0;
223  while (size--) {
224  if (write(*buffer++) != 1) break;
225  ret++;
226  }
227  return ret;
228  }
229 
230  int16_t space = txspace();
231  if (space <= 0) {
232  return 0;
233  }
234  if (size > (size_t)space) {
235  // throw away remainder if too much data
236  size = space;
237  }
238  if (_txBuffer->tail > _txBuffer->head) {
239  // perform as single memcpy
240  memcpy(&_txBuffer->bytes[_txBuffer->head], buffer, size);
241  _txBuffer->head = (_txBuffer->head + size) & _txBuffer->mask;
242  // enable the data-ready interrupt, as it may be off if the buffer is empty
243  *_ucsrb |= _portTxBits;
244  return size;
245  }
246 
247  // perform as two memcpy calls
248  uint16_t n = (_txBuffer->mask+1) - _txBuffer->head;
249  if (n > size) n = size;
250  memcpy(&_txBuffer->bytes[_txBuffer->head], buffer, n);
251  _txBuffer->head = (_txBuffer->head + n) & _txBuffer->mask;
252  buffer += n;
253  n = size - n;
254  if (n > 0) {
255  memcpy(&_txBuffer->bytes[0], buffer, n);
256  _txBuffer->head = (_txBuffer->head + n) & _txBuffer->mask;
257  }
258 
259  // enable the data-ready interrupt, as it may be off if the buffer is empty
260  *_ucsrb |= _portTxBits;
261  return size;
262 }

267 bool AVRUARTDriver::_allocBuffer(Buffer *buffer, uint16_t size)
268 {
269  uint8_t mask;
270  uint8_t shift;
271 
272  // init buffer state
273  buffer->head = buffer->tail = 0;
274 
275  // Compute the power of 2 greater or equal to the requested buffer size
276  // and then a mask to simplify wrapping operations. Using __builtin_clz
277  // would seem to make sense, but it uses a 256(!) byte table.
278  // Note that we ignore requests for more than BUFFER_MAX space.
279  for (shift = 1; (1U << shift) < min(_max_buffer_size, size); shift++)
280  ;
281  mask = (1U << shift) - 1;
282 
283  // If the descriptor already has a buffer allocated we need to take
284  // care of it.
285  if (buffer->bytes) {
286 
287  // If the allocated buffer is already the correct size then
288  // we have nothing to do
289  if (buffer->mask == mask)
290  return true;
291 
292  // Dispose of the old buffer.
293  free(buffer->bytes);
294  }
295  buffer->mask = mask;
296 
297  // allocate memory for the buffer - if this fails, we fail.
298  buffer->bytes = (uint8_t *) malloc(buffer->mask + (size_t)1);
299 
300  return (buffer->bytes != NULL);
301 }
