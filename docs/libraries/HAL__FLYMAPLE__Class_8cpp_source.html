65 void HAL_FLYMAPLE::init(int argc,char* const argv[]) const {
66  /* initialize all drivers and private members here.
67  * up to the programmer to do this in the correct order.
68  * Scheduler should likely come first. */
69  scheduler->init(NULL);
70 
71  /* uartA is the serial port used for the console, so lets make sure
72  * it is initialized at boot */

77  rcin->init(NULL);
