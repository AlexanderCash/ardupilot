275 // output_test - spin a motor at the pwm value specified
276 // motor_seq is the motor's sequence number from 1 to the number of motors on the frame
277 // pwm value is an actual pwm value that will be output, normally in the range of 1000 ~ 2000
278 void AP_MotorsHeli::output_test(uint8_t motor_seq, int16_t pwm)
279 {
280  // exit immediately if not armed
281  if (!armed()) {
282  return;
283  }
284 
285  // output to motors and servos
286  switch (motor_seq) {
287  case 1:
288  // swash servo 1
289  hal.rcout->write(pgm_read_byte(&_motor_to_channel_map[AP_MOTORS_MOT_1]), pwm);
290  break;
291  case 2:
292  // swash servo 2
293  hal.rcout->write(pgm_read_byte(&_motor_to_channel_map[AP_MOTORS_MOT_2]), pwm);
294  break;
295  case 3:
296  // swash servo 3
297  hal.rcout->write(pgm_read_byte(&_motor_to_channel_map[AP_MOTORS_MOT_3]), pwm);
298  break;
299  case 4:
300  // external gyro & tail servo
301  if (_tail_type == AP_MOTORS_HELI_TAILTYPE_SERVO_EXTGYRO) {
302  write_aux(_ext_gyro_gain);
303  }
304  hal.rcout->write(pgm_read_byte(&_motor_to_channel_map[AP_MOTORS_MOT_4]), pwm);
305  break;
306  case 5:
307  // main rotor
308  hal.rcout->write(pgm_read_byte(&_motor_to_channel_map[AP_MOTORS_HELI_RSC]), pwm);
309  break;
310  default:
311  // do nothing
312  break;
313  }
314 }

345 // recalc_scalers - recalculates various scalers used. Should be called at about 1hz to allow users to see effect of changing parameters
346 void AP_MotorsHeli::recalc_scalers()
347 {
348  // recalculate rotor ramp up increment
349  if (_rsc_ramp_time <= 0) {
350  _rsc_ramp_time = 1;
351  }
352  _rsc_ramp_increment = 1000.0f / (_rsc_ramp_time / _dt);
353 
354  // recalculate rotor runup increment
355  if (_rsc_runup_time <= 0 ) {
356  _rsc_runup_time = 1;
357  }
358  if (_rsc_runup_time < _rsc_ramp_time) {
359  _rsc_runup_time = _rsc_ramp_time;
360  }
361  _rsc_runup_increment = 1000.0f / (_rsc_runup_time * 100.0f);
362 }

447 // init_swash - initialise the swash plate
448 void AP_MotorsHeli::init_swash()
449 {
450 
451  // swash servo initialisation
452  _servo_1.set_range(0,1000);
453  _servo_2.set_range(0,1000);
454  _servo_3.set_range(0,1000);
455  _servo_4.set_angle(4500);
456 
457  // range check collective min, max and mid
458  if( _collective_min >= _collective_max ) {
459  _collective_min = 1000;
460  _collective_max = 2000;
461  }
462  _collective_mid = constrain_int16(_collective_mid, _collective_min, _collective_max);
463 
464  // calculate collective mid point as a number from 0 to 1000
465  _collective_mid_pwm = ((float)(_collective_mid-_collective_min))/((float)(_collective_max-_collective_min))*1000.0f;
466 
467  // determine roll, pitch and collective input scaling
468  _roll_scaler = (float)_roll_max/4500.0f;
469  _pitch_scaler = (float)_pitch_max/4500.0f;
470  _collective_scalar = ((float)(_collective_max-_collective_min))/1000.0f;
471 
472  // calculate factors based on swash type and servo position
473  calculate_roll_pitch_collective_factors();
474 
475  // servo min/max values
476  _servo_1.radio_min = 1000;
477  _servo_1.radio_max = 2000;
478  _servo_2.radio_min = 1000;
479  _servo_2.radio_max = 2000;

487 // calculate_roll_pitch_collective_factors - calculate factors based on swash type and servo position
488 void AP_MotorsHeli::calculate_roll_pitch_collective_factors()
489 {
490  if (_swash_type == AP_MOTORS_HELI_SWASH_CCPM) { //CCPM Swashplate, perform control mixing
491 
492  // roll factors
493  _rollFactor[CH_1] = cosf(radians(_servo1_pos + 90 - (_phase_angle + _delta_phase_angle)));
494  _rollFactor[CH_2] = cosf(radians(_servo2_pos + 90 - (_phase_angle + _delta_phase_angle)));
495  _rollFactor[CH_3] = cosf(radians(_servo3_pos + 90 - (_phase_angle + _delta_phase_angle)));
496 
497  // pitch factors
498  _pitchFactor[CH_1] = cosf(radians(_servo1_pos - (_phase_angle + _delta_phase_angle)));
499  _pitchFactor[CH_2] = cosf(radians(_servo2_pos - (_phase_angle + _delta_phase_angle)));
500  _pitchFactor[CH_3] = cosf(radians(_servo3_pos - (_phase_angle + _delta_phase_angle)));
501 
502  // collective factors
503  _collectiveFactor[CH_1] = 1;
504  _collectiveFactor[CH_2] = 1;
505  _collectiveFactor[CH_3] = 1;
506 
507  }else{ //H1 Swashplate, keep servo outputs seperated
508 
509  // roll factors
510  _rollFactor[CH_1] = 1;
511  _rollFactor[CH_2] = 0;
512  _rollFactor[CH_3] = 0;
513 
514  // pitch factors
515  _pitchFactor[CH_1] = 0;
516  _pitchFactor[CH_2] = 1;
517  _pitchFactor[CH_3] = 0;
518 
519  // collective factors
520  _collectiveFactor[CH_1] = 0;
521  _collectiveFactor[CH_2] = 0;
522  _collectiveFactor[CH_3] = 1;
523  }
524 }

527 // heli_move_swash - moves swash plate to attitude of parameters passed in

531 // collective: 0 ~ 1000

535 {

538

542  limit.throttle_lower = false;
543  limit.throttle_upper = false;
544 
545  if (_servo_manual == 1) { // are we in manual servo mode? (i.e. swash set-up mode)?

553  }else{ // regular flight mode
554

557  init_swash();
558  }
559 
560  // rescale roll_out and pitch-out into the min and max ranges to provide linear motion
561  // across the input range instead of stopping when the input hits the constrain value
562  // these calculations are based on an assumption of the user specified roll_max and pitch_max
563  // coming into this equation at 4500 or less, and based on the original assumption of the
564  // total _servo_x.servo_out range being -4500 to 4500.
565  roll_out = roll_out * _roll_scaler;
566  if (roll_out < -_roll_max) {
567  roll_out = -_roll_max;
568  limit.roll_pitch = true;
569  }
570  if (roll_out > _roll_max) {
571  roll_out = _roll_max;
572  limit.roll_pitch = true;
573  }
574 
575  // scale pitch and update limits
576  pitch_out = pitch_out * _pitch_scaler;
577  if (pitch_out < -_pitch_max) {
578  pitch_out = -_pitch_max;
579  limit.roll_pitch = true;
580  }
581  if (pitch_out > _pitch_max) {
582  pitch_out = _pitch_max;
583  limit.roll_pitch = true;
584  }
585 
586  // constrain collective input
587  _collective_out = coll_in;
588  if (_collective_out <= 0) {
589  _collective_out = 0;
590  limit.throttle_lower = true;
591  }
592  if (_collective_out >= 1000) {
593  _collective_out = 1000;
594  limit.throttle_upper = true;
595  }
596 
597  // ensure not below landed/landing collective
598  if (_heliflags.landing_collective && _collective_out < _land_collective_min) {
599  _collective_out = _land_collective_min;
600  limit.throttle_lower = true;
601  }
602 
603  // scale collective pitch
604  coll_out_scaled = _collective_out * _collective_scalar + _collective_min - 1000;
605 
606  // rudder feed forward based on collective
607  // the feed-forward is not required when the motor is shut down and not creating torque
608  // also not required if we are using external gyro
609  if ((_rotor_desired > 0) && _tail_type != AP_MOTORS_HELI_TAILTYPE_SERVO_EXTGYRO) {
610  // sanity check collective_yaw_effect
611  _collective_yaw_effect = constrain_float(_collective_yaw_effect, -AP_MOTOR_HELI_COLYAW_RANGE, AP_MOTOR_HELI_COLYAW_RANGE);
612  yaw_offset = _collective_yaw_effect * abs(_collective_out - _collective_mid_pwm);
613  }
614  }
615 
616  // swashplate servos
617  _servo_1.servo_out = (_rollFactor[CH_1] * roll_out + _pitchFactor[CH_1] * pitch_out)/10 + _collectiveFactor[CH_1] * coll_out_scaled + (_servo_1.radio_trim-1500);
618  _servo_2.servo_out = (_rollFactor[CH_2] * roll_out + _pitchFactor[CH_2] * pitch_out)/10 + _collectiveFactor[CH_2] * coll_out_scaled + (_servo_2.radio_trim-1500);
619  if (_swash_type == AP_MOTORS_HELI_SWASH_H1) {
620  _servo_1.servo_out += 500;
621  _servo_2.servo_out += 500;
622  }
623  _servo_3.servo_out = (_rollFactor[CH_3] * roll_out + _pitchFactor[CH_3] * pitch_out)/10 + _collectiveFactor[CH_3] * coll_out_scaled + (_servo_3.radio_trim-1500);
624  _servo_4.servo_out = yaw_out + yaw_offset;
625 
626  // constrain yaw and update limits
627  if (_servo_4.servo_out < -4500) {
628  _servo_4.servo_out = -4500;
629  limit.yaw = true;
630  }
631  if (_servo_4.servo_out > 4500) {
632  _servo_4.servo_out = 4500;
633  limit.yaw = true;
634  }
635 
636  // use servo_out to calculate pwm_out and radio_out
637  _servo_1.calc_pwm();
638  _servo_2.calc_pwm();
639  _servo_3.calc_pwm();
640  _servo_4.calc_pwm();
641 
642  // actually move the servos
643  hal.rcout->write(pgm_read_byte(&_motor_to_channel_map[AP_MOTORS_MOT_1]), _servo_1.radio_out);
644  hal.rcout->write(pgm_read_byte(&_motor_to_channel_map[AP_MOTORS_MOT_2]), _servo_2.radio_out);
645  hal.rcout->write(pgm_read_byte(&_motor_to_channel_map[AP_MOTORS_MOT_3]), _servo_3.radio_out);
646  hal.rcout->write(pgm_read_byte(&_motor_to_channel_map[AP_MOTORS_MOT_4]), _servo_4.radio_out);
647 
648  // output gain to exernal gyro
649  if (_tail_type == AP_MOTORS_HELI_TAILTYPE_SERVO_EXTGYRO) {
650  write_aux(_ext_gyro_gain);
651  }
652 }

654 // rsc_control - update value to send to tail and main rotor's ESC
655 // desired_rotor_speed is a desired speed from 0 to 1000
656 void AP_MotorsHeli::rsc_control()
657 {
658  // if disarmed output minimums
659  if (!armed()) {
660  // shut down tail rotor
661  if (_tail_type == AP_MOTORS_HELI_TAILTYPE_DIRECTDRIVE_VARPITCH || _tail_type == AP_MOTORS_HELI_TAILTYPE_DIRECTDRIVE_FIXEDPITCH) {
662  _tail_direct_drive_out = 0;
663  write_aux(_tail_direct_drive_out);
664  }
665  // shut down main rotor
666  if (_rsc_mode != AP_MOTORS_HELI_RSC_MODE_NONE) {
667  _rotor_out = 0;
668  _rotor_speed_estimate = 0;
669  write_rsc(_rotor_out);
670  }
671  return;
672  }
673 
674  // ramp up or down main rotor and tail
675  if (_rotor_desired > 0) {
676  // ramp up tail rotor (this does nothing if not using direct drive variable pitch tail)
677  tail_ramp(_direct_drive_tailspeed);
678  // note: this always returns true if not using direct drive variable pitch tail
679  if (tail_rotor_runup_complete()) {
680  rotor_ramp(_rotor_desired);
681  }
682  }else{
683  // shutting down main rotor
684  rotor_ramp(0);
685  // shut-down tail rotor. Note: this does nothing if not using direct drive vairable pitch tail
686  tail_ramp(0);
687  }
688 
689  // direct drive fixed pitch tail servo gets copy of yaw servo out (ch4) while main rotor is running
690  if (_tail_type == AP_MOTORS_HELI_TAILTYPE_DIRECTDRIVE_FIXEDPITCH) {
691  // output fixed-pitch speed control if Ch8 is high
692  if (_rotor_desired > 0 || _rotor_speed_estimate > 0) {
693  // copy yaw output to tail esc
694  write_aux(_servo_4.servo_out);
695  }else{
696  write_aux(0);
697  }
698  }
699 }

701 // rotor_ramp - ramps rotor towards target
702 // result put in _rotor_out and sent to ESC
703 void AP_MotorsHeli::rotor_ramp(int16_t rotor_target)
704 {
705  // return immediately if not ramping required
706  if (_rsc_mode == AP_MOTORS_HELI_RSC_MODE_NONE) {
707  _rotor_out = rotor_target;
708  return;
709  }
710 
711  // range check rotor_target
712  rotor_target = constrain_int16(rotor_target,0,1000);
713 
714  // ramp rotor esc output towards target
715  if (_rotor_out < rotor_target) {
716  // allow rotor out to jump to rotor's current speed
717  if (_rotor_out < _rotor_speed_estimate) {
718  _rotor_out = _rotor_speed_estimate;
719  }
720  // ramp up slowly to target
721  _rotor_out += _rsc_ramp_increment;
722  if (_rotor_out > rotor_target) {
723  _rotor_out = rotor_target;
724  }
725  }else{
726  // ramping down happens instantly
727  _rotor_out = rotor_target;
728  }
729 
730  // ramp rotor speed estimate towards rotor out
731  if (_rotor_speed_estimate < _rotor_out) {
732  _rotor_speed_estimate += _rsc_runup_increment;
733  if (_rotor_speed_estimate > _rotor_out) {
734  _rotor_speed_estimate = _rotor_out;
735  }
736  }else{
737  _rotor_speed_estimate -= _rsc_runup_increment;
738  if (_rotor_speed_estimate < _rotor_out) {
739  _rotor_speed_estimate = _rotor_out;
740  }
741  }
742 
743  // set runup complete flag
744  if (!_heliflags.motor_runup_complete && rotor_target > 0 && _rotor_speed_estimate >= rotor_target) {
745  _heliflags.motor_runup_complete = true;
746  }
747  if (_heliflags.motor_runup_complete && rotor_target == 0 && _rotor_speed_estimate <= 0) {
748  _heliflags.motor_runup_complete = false;
749  }
750 
751  // output to rsc servo
752  write_rsc(_rotor_out);
753 }
