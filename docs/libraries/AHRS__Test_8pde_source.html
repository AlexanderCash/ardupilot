91 void loop(void)
92 {
93  static uint16_t counter;
94  static uint32_t last_t, last_print, last_compass;
95  uint32_t now = hal.scheduler->micros();
96  float heading = 0;
97 
98  if (last_t == 0) {
99  last_t = now;

102  last_t = now;
103 
104  if (now - last_compass > 100*1000UL &&
105  compass.read()) {
106  heading = compass.calculate_heading(ahrs.get_dcm_matrix());
107  // read compass at 10Hz
108  last_compass = now;
109 #if WITH_GPS
110  g_gps->update();
111 #endif
112  }
113 
114  ahrs.update();
115  counter++;
116 
117  if (now - last_print >= 100000 /* 100ms : 10hz */) {
118  Vector3f drift = ahrs.get_gyro_drift();
119  hal.console->printf_P(
120  PSTR("r:%4.1f p:%4.1f y:%4.1f "
121  "drift=(%5.1f %5.1f %5.1f) hdg=%.1f rate=%.1f\n"),
122  ToDeg(ahrs.roll),
123  ToDeg(ahrs.pitch),
124  ToDeg(ahrs.yaw),
125  ToDeg(drift.x),
126  ToDeg(drift.y),
127  ToDeg(drift.z),
128  compass.use_for_yaw() ? ToDeg(heading) : 0.0f,
129  (1.0e6f*counter)/(now-last_print));
130  last_print = now;
131  counter = 0;
132  }
