137 /*
138  convert a set of roll rates from earth frame to body frame
139  output values are in radians/second
140 */
141 void SITL::convert_body_frame(double rollDeg, double pitchDeg,
142  double rollRate, double pitchRate, double yawRate,
143  double *p, double *q, double *r)
144 {
145  double phi, theta, phiDot, thetaDot, psiDot;
146 
147  phi = ToRad(rollDeg);
148  theta = ToRad(pitchDeg);
149  phiDot = ToRad(rollRate);
150  thetaDot = ToRad(pitchRate);
151  psiDot = ToRad(yawRate);
152 
153  *p = phiDot - psiDot*sinf(theta);
154  *q = cosf(phi)*thetaDot + sinf(phi)*psiDot*cosf(theta);
155  *r = cosf(phi)*psiDot*cosf(theta) - sinf(phi)*thetaDot;
156 }

159 /*
160  convert angular velocities from body frame to
161  earth frame.
162 
163  all inputs and outputs are in radians/s
164 */
165 Vector3f SITL::convert_earth_frame(const Matrix3f &dcm, const Vector3f &gyro)
166 {
167  float p = gyro.x;
168  float q = gyro.y;
169  float r = gyro.z;
170 
171  float phi, theta, psi;
172  dcm.to_euler(&phi, &theta, &psi);
173 
174  float phiDot = p + tanf(theta)*(q*sinf(phi) + r*cosf(phi));
175  float thetaDot = q*cosf(phi) - r*sinf(phi);
176  if (fabsf(cosf(theta)) < 1.0e-20f) {
177  theta += 1.0e-10f;
178  }
179  float psiDot = (q*sinf(phi) + r*cosf(phi))/cosf(theta);
180  return Vector3f(phiDot, thetaDot, psiDot);
181 }
