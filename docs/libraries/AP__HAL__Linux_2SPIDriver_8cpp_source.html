148 void LinuxSPIDeviceManager::cs_assert(enum AP_HAL::SPIDevice type)
149 {
150  uint16_t bus = 0, i;
151  for (i=0; i<LINUX_SPI_DEVICE_NUM_DEVICES; i++) {
152  if (_device[i]._type == type) {
153  bus = _device[i]._bus;
154  break;
155  }
156  }
157  if (i == LINUX_SPI_DEVICE_NUM_DEVICES) {
158  hal.scheduler->panic("Bad device type");
159  }
160 
161  // Kernel-mode CS handling
162  if(_device[i]._cs_pin == SPI_CS_KERNEL)
163  return;
164 
165  for (i=0; i<LINUX_SPI_DEVICE_NUM_DEVICES; i++) {
166  if (_device[i]._bus != bus) {
167  // not the same bus
168  continue;
169  }
170  if (_device[i]._type != type) {
171  if (_device[i]._cs->read() != 1) {
172  hal.console->printf("two CS enabled at once i=%u %u and %u\n",
173  (unsigned)i, (unsigned)type, (unsigned)_device[i]._type);
174  }
175  }
176  }
177  for (i=0; i<LINUX_SPI_DEVICE_NUM_DEVICES; i++) {
178  if (_device[i]._type == type) {
179  _device[i]._cs->write(0);
180  }
181  }

210 void LinuxSPIDeviceManager::transaction(LinuxSPIDeviceDriver &driver, const uint8_t *tx, uint8_t *rx, uint16_t len)
211 {
212  // we set the mode before we assert the CS line so that the bus is
213  // in the correct idle state before the chip is selected
214  ioctl(driver._fd, SPI_IOC_WR_MODE, &driver._mode);
215 
216  cs_assert(driver._type);
217  struct spi_ioc_transfer spi[1];
218  memset(spi, 0, sizeof(spi));
219  spi[0].tx_buf = (uint64_t)tx;
220  spi[0].rx_buf = (uint64_t)rx;
221  spi[0].len = len;
222  spi[0].delay_usecs = 0;
223  spi[0].speed_hz = driver._speed;
224  spi[0].bits_per_word = driver._bitsPerWord;
225  spi[0].cs_change = 0;
226 
227  if (rx != NULL) {
228  // keep valgrind happy
229  memset(rx, 0, len);
230  }
231 
232  ioctl(driver._fd, SPI_IOC_MESSAGE(1), &spi);
233  cs_release(driver._type);
234 }
