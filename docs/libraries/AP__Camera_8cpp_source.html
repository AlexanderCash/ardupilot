231 /* update location, for triggering by GPS distance moved
232  This function returns true if a picture should be taken
233  The caller is responsible for taking the picture based on the return value of this function.
234  The caller is also responsible for logging the details about the photo
235 */
236 bool AP_Camera::update_location(const struct Location &loc)
237 {
238  if (is_zero(_trigg_dist)) {
239  return false;
240  }
241  if (_last_location.lat == 0 && _last_location.lng == 0) {
242  _last_location = loc;
243  return false;
244  }
245  if (_last_location.lat == loc.lat && _last_location.lng == loc.lng) {
246  // we haven't moved - this can happen as update_location() may
247  // be called without a new GPS fix
248  return false;
249  }
250  if (get_distance(loc, _last_location) < _trigg_dist) {
251  return false;
252  }
253  _last_location = loc;
254  return true;
255 }
