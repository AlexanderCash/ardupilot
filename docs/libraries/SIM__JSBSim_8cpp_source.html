158 /*
159  start JSBSim child
160  */
161 bool JSBSim::start_JSBSim(void)
162 {
163  if (started_jsbsim) {
164  return true;
165  }
166  if (!open_fdm_socket()) {
167  return false;
168  }
169 
170  int p[2];
171  int devnull = open("/dev/null", O_RDWR);
172  if (pipe(p) != 0) {
173  hal.scheduler->panic("Unable to create pipe");
174  }
175  pid_t child_pid = fork();
176  if (child_pid == 0) {
177  // in child
178  dup2(devnull, 0);
179  dup2(p[1], 1);
180  close(p[0]);
181  for (uint8_t i=3; i<100; i++) {
182  close(i);
183  }
184  char *logdirective;
185  char *script;
186 
187  asprintf(&logdirective, "--logdirectivefile=%s", jsbsim_fgout);
188  asprintf(&script, "--script=%s", jsbsim_script);
189 
190  if (chdir(autotest_dir) != 0) {
191  perror(autotest_dir);
192  exit(1);
193  }
194 
195  int ret = execlp("JSBSim",
196  "JSBSim",
197  "--realtime",
198  "--suspend",
199  "--nice",
200  "--simulation-rate=1000",
201  logdirective,
202  script,
203  NULL);
204  if (ret != 0) {
205  perror("JSBSim");
206  }
207  exit(1);
208  }
209  close(p[1]);
210  jsbsim_stdout = p[0];
211 
212  // read startup to be sure it is running
213  char c;
214  if (read(jsbsim_stdout, &c, 1) != 1) {
215  hal.scheduler->panic("Unable to start JSBSim");
216  }
217 
218  if (!expect("JSBSim Execution beginning")) {
219  hal.scheduler->panic("Failed to start JSBSim");
220  }
221  if (!open_control_socket()) {
222  hal.scheduler->panic("Failed to open JSBSim control socket");
223  }
224 
225  fcntl(jsbsim_stdout, F_SETFL, fcntl(jsbsim_stdout, F_GETFL, 0) | O_NONBLOCK);
226 
227  started_jsbsim = true;
228  check_stdout();
229  return true;
230 }

308 /*
309  decode and send servos
310 */
311 void JSBSim::send_servos(const struct sitl_input &input)
312 {
313  char *buf = NULL;
314  float aileron = (input.servos[0]-1500)/500.0f;
315  float elevator = (input.servos[1]-1500)/500.0f;
316  float throttle = (input.servos[2]-1000)/1000.0;
317  float rudder = (input.servos[3]-1500)/500.0f;
318  if (frame == FRAME_ELEVON) {
319  // fake an elevon plane
320  float ch1 = aileron;
321  float ch2 = elevator;
322  aileron = (ch2-ch1)/2.0f;
323  // the minus does away with the need for RC2_REV=-1
324  elevator = -(ch2+ch1)/2.0f;
325  } else if (frame == FRAME_VTAIL) {
326  // fake a vtail plane
327  float ch1 = elevator;
328  float ch2 = rudder;
329  // this matches VTAIL_OUTPUT==2
330  elevator = (ch2-ch1)/2.0f;
331  rudder = (ch2+ch1)/2.0f;
332  }
333  asprintf(&buf,
334  "set fcs/aileron-cmd-norm %f\n"
335  "set fcs/elevator-cmd-norm %f\n"
336  "set fcs/rudder-cmd-norm %f\n"
337  "set fcs/throttle-cmd-norm %f\n"
338  "step\n",
339  aileron, elevator, rudder, throttle);
340  sock_control.send(buf, strlen(buf));
341  free(buf);
342 }

366 /*
367  receive an update from the FDM
368  This is a blocking function
369  */
370 void JSBSim::recv_fdm(const struct sitl_input &input)
371 {
372  FGNetFDM fdm;
373  check_stdout();
374  while (sock_fgfdm.recv(&fdm, sizeof(fdm), 100) != sizeof(fdm)) {
375  send_servos(input);
376  check_stdout();
377  }
378  fdm.ByteSwap();
379 
380  accel_body = Vector3f(fdm.A_X_pilot, fdm.A_Y_pilot, fdm.A_Z_pilot) * FEET_TO_METERS;
381 
382  double p, q, r;
383  SITL::convert_body_frame(degrees(fdm.phi), degrees(fdm.theta),
384  degrees(fdm.phidot), degrees(fdm.thetadot), degrees(fdm.psidot),
385  &p, &q, &r);
386  gyro = Vector3f(p, q, r);
387 
388  velocity_ef = Vector3f(fdm.v_north, fdm.v_east, fdm.v_down) * FEET_TO_METERS;
389  location.lat = degrees(fdm.latitude) * 1.0e7;
390  location.lng = degrees(fdm.longitude) * 1.0e7;
391  location.alt = fdm.agl*100 + home.alt;
392  dcm.from_euler(fdm.phi, fdm.theta, fdm.psi);
393 
394  // assume 1kHz for now
395  time_now_us += 1000;
396 }
