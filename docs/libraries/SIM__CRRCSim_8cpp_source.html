49 /*
50  decode and send servos for heli
51 */
52 void CRRCSim::send_servos_heli(const struct sitl_input &input)
53 {
54  float swash1 = (input.servos[0]-1000) / 1000.0f;
55  float swash2 = (input.servos[1]-1000) / 1000.0f;
56  float swash3 = (input.servos[2]-1000) / 1000.0f;
57  float tail_rotor = (input.servos[3]-1000) / 1000.0f;
58  float rsc = (input.servos[7]-1000) / 1000.0f;
59 
60  float col_pitch = (swash1+swash2+swash3)/3.0 - 0.5f;
61  float roll_rate = (swash1 - swash2)/2;
62  float pitch_rate = -((swash1 + swash2)/2.0 - swash3)/2;
63  float yaw_rate = -(tail_rotor - 0.5);
64 
65  servo_packet pkt;
66  pkt.roll_rate = constrain_float(roll_rate, -0.5, 0.5);
67  pkt.pitch_rate = constrain_float(pitch_rate, -0.5, 0.5);
68  pkt.throttle = constrain_float(rsc, 0, 1);
69  pkt.yaw_rate = constrain_float(yaw_rate, -0.5, 0.5);
70  pkt.col_pitch = constrain_float(col_pitch, -0.5, 0.5);
71 
72  sock.send(&pkt, sizeof(pkt));
73 }

107 /*
108  receive an update from the FDM
109  This is a blocking function
110  */
111 void CRRCSim::recv_fdm(const struct sitl_input &input)
112 {
113  fdm_packet pkt;
114 
115  /*
116  we re-send the servo packet every 0.1 seconds until we get a
117  reply. This allows us to cope with some packet loss to the FDM
118  */
119  while (!sock.recv(&pkt, sizeof(pkt), 100)) {
120  send_servos(input);
121  }
122 
123  accel_body = Vector3f(pkt.xAccel, pkt.yAccel, pkt.zAccel);
124  gyro = Vector3f(pkt.rollRate, pkt.pitchRate, pkt.yawRate);
125  velocity_ef = Vector3f(pkt.speedN, pkt.speedE, pkt.speedD);
126 
127  Location loc1, loc2;
128  loc2.lat = pkt.latitude * 1.0e7;
129  loc2.lng = pkt.longitude * 1.0e7;
130  memset(&loc1, 0, sizeof(loc1));
131  Vector2f posdelta = location_diff(loc1, loc2);
132  position.x = posdelta.x;
133  position.y = posdelta.y;
134  position.z = -pkt.altitude;
135 
136  dcm.from_euler(pkt.roll, pkt.pitch, pkt.yaw);
137 
138  // auto-adjust to crrcsim frame rate
139  double deltat = pkt.timestamp - last_timestamp;
140  time_now_us += deltat * 1.0e6;
141 
142  if (deltat < 0.01 && deltat > 0) {
143  adjust_frame_time(1.0/deltat);
144  }
145  last_timestamp = pkt.timestamp;
146 }
