128  // initGyro() -- Sets up the gyroscope to begin reading.
129  // This function steps through all five gyroscope control registers.
130  // Upon exit, the following parameters will be set:
131  // - CTRL_REG1_G = 0x0F: Normal operation mode, all axes enabled.
132  // 95 Hz ODR, 12.5 Hz cutoff frequency.
133  // - CTRL_REG2_G = 0x00: HPF set to normal mode, cutoff frequency
134  // set to 7.2 Hz (depends on ODR).
135  // - CTRL_REG3_G = 0x88: Interrupt enabled on INT_G (set to push-pull and
136  // active high). Data-ready output enabled on DRDY_G.
137  // - CTRL_REG4_G = 0x00: Continuous update mode. Data LSB stored in lower
138  // address. Scale set to 245 DPS. SPI mode set to 4-wire.
139  // - CTRL_REG5_G = 0x00: FIFO disabled. HPF disabled.
140  void _initGyro();

142  // initAccel() -- Sets up the accelerometer to begin reading.
143  // This function steps through all accelerometer related control registers.
144  // Upon exit these registers will be set as:
145  // - CTRL_REG0_XM = 0x00: FIFO disabled. HPF bypassed. Normal mode.
146  // - CTRL_REG1_XM = 0x57: 100 Hz data rate. Continuous update.
147  // all axes enabled.
148  // - CTRL_REG2_XM = 0x00: 2g scale. 773 Hz anti-alias filter BW.
149  // - CTRL_REG3_XM = 0x04: Accel data ready signal on INT1_XM pin.
150  void _initAccel();

152  // initMag() -- Sets up the magnetometer to begin reading.

154  // Upon exit these registers will be set as:
155  // - CTRL_REG4_XM = 0x04: Mag data ready signal on INT2_XM pin.
156  // - CTRL_REG5_XM = 0x14: 100 Hz update rate. Low resolution. Interrupt
157  // requests don't latch. Temperature sensor disabled.
158  // - CTRL_REG6_XM = 0x00: 2 Gs scale.
159  // - CTRL_REG7_XM = 0x00: Continuous conversion mode. Normal HPF mode.
