55 /*
56  update the helicopter simulation by one time step
57  */
58 void Helicopter::update(const struct sitl_input &input)
59 {
60  float swash1 = (input.servos[0]-1000) / 1000.0f;
61  float swash2 = (input.servos[1]-1000) / 1000.0f;
62  float swash3 = (input.servos[2]-1000) / 1000.0f;
63  float tail_rotor = (input.servos[3]-1000) / 1000.0f;
64  float rsc = (input.servos[7]-1000) / 1000.0f;
65 
66  // how much time has passed?
67  float delta_time = frame_time_us * 1.0e-6f;
68 
69  float thrust = (rsc/rsc_setpoint)*(swash1+swash2+swash3)/3.0f;
70 
71  // very simplistic mapping to body euler rates
72  float roll_rate = swash1 - swash2;
73  float pitch_rate = (swash1 + swash2)/2.0f - swash3;
74  float yaw_rate = tail_rotor - 0.5f;
75 
76  float rsc_scale = rsc/rsc_setpoint;
77 
78  roll_rate *= rsc_scale;
79  pitch_rate *= rsc_scale;
80  yaw_rate *= rsc_scale;
81 
82  // rotational acceleration, in rad/s/s, in body frame
83  Vector3f rot_accel;
84  rot_accel.x = roll_rate * roll_rate_max;
85  rot_accel.y = pitch_rate * pitch_rate_max;
86  rot_accel.z = yaw_rate * yaw_rate_max;
87 
88  // rotational air resistance
89  rot_accel.x -= gyro.x * radians(5000.0) / terminal_rotation_rate;
90  rot_accel.y -= gyro.y * radians(5000.0) / terminal_rotation_rate;
91  rot_accel.z -= gyro.z * radians(400.0) / terminal_rotation_rate;
92 
93  // torque effect on tail
94  rot_accel.z += (rsc_scale+thrust) * rotor_rot_accel;
95 
96  // update rotational rates in body frame
97  gyro += rot_accel * delta_time;
98 
99  // update attitude
100  dcm.rotate(gyro * delta_time);
101  dcm.normalize();
102 
103  // air resistance
104  Vector3f air_resistance = -velocity_ef * (GRAVITY_MSS/terminal_velocity);
105 
106  // scale thrust to newtons
107  thrust *= thrust_scale;
108 
109  accel_body = Vector3f(0, yaw_rate * rsc_scale * tail_thrust_scale, -thrust / mass);
110  Vector3f accel_earth = dcm * accel_body;
111  accel_earth += Vector3f(0, 0, GRAVITY_MSS);
112  accel_earth += air_resistance;
113 
114  // if we're on the ground, then our vertical acceleration is limited
115  // to zero. This effectively adds the force of the ground on the aircraft
116  if (on_ground(position) && accel_earth.z > 0) {
117  accel_earth.z = 0;
118  }
119 
120  // work out acceleration as seen by the accelerometers. It sees the kinematic
121  // acceleration (ie. real movement), plus gravity
122  accel_body = dcm.transposed() * (accel_earth + Vector3f(0, 0, -GRAVITY_MSS));
123 
124  // add some noise
125  add_noise(thrust / thrust_scale);
126 
127  // new velocity vector
128  velocity_ef += accel_earth * delta_time;
129 
130  // new position vector
131  Vector3f old_position = position;
132  position += velocity_ef * delta_time;
133 
134  // constrain height to the ground
135  if (on_ground(position)) {
136  if (!on_ground(old_position)) {
137  printf("Hit ground at %f m/s\n", velocity_ef.z);
138 
139  velocity_ef.zero();
140 
141  // zero roll/pitch, but keep yaw
142  float r, p, y;
143  dcm.to_euler(&r, &p, &y);
144  dcm.from_euler(0, 0, y);
145 
146  position.z = -(ground_level + frame_height - home.alt*0.01f);
147  }
148  }
149 
150  // update lat/lon/altitude
151  update_position();
152 }
