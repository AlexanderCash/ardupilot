90 void DataFlash_APM2::Init(const struct LogStructure *structure, uint8_t num_types)
91 {
92  DataFlash_Class::Init(structure, num_types);
93  // init to zero
94  df_NumPages = 0;
95 
96  hal.gpio->pinMode(DF_RESET, HAL_GPIO_OUTPUT);
97  hal.gpio->pinMode(DF_CARDDETECT, HAL_GPIO_INPUT);
98 
99  // Reset the chip
100  hal.gpio->write(DF_RESET,0);
101  hal.scheduler->delay(1);
102  hal.gpio->write(DF_RESET,1);
103 
104  _spi = hal.spi->device(AP_HAL::SPIDevice_Dataflash);
105  if (_spi == NULL) {
106  hal.scheduler->panic(
107  PSTR("PANIC: DataFlash SPIDeviceDriver not found"));
108  return;
109  }
110  _spi_sem = _spi->get_semaphore();
111  if (_spi_sem == NULL) {
112  hal.scheduler->panic(
113  PSTR("PANIC: DataFlash SPIDeviceDriver semaphore is null"));
114  return; /* never reached */
115  }
116 
117  if (!_sem_take(5))
118  return;
119  // get page size: 512 or 528 (by default: 528)
120  df_PageSize=PageSize();
121 
122  ReadManufacturerID();
123 
124  _spi_sem->give();
125 
126  // see page 22 of the spec for the density code
127  uint8_t density_code = (df_device >> 8) & 0x1F;
128 
129  // note that we set df_NumPages to one lower than the highest, as
130  // the last page is reserved for a config page
131  if (density_code == 0x7) {
132  // 32 Mbit
133  df_NumPages = 8191;
134  } else if (density_code == 0x6) {
135  // 16 Mbit
136  df_NumPages = 4095;
137  }
138 
139  //serialDebug("density_code %d pages %d, size %d\n", density_code, df_NumPages, df_PageSize);
140 }

143 void DataFlash_APM2::ReadManufacturerID()
144 {
145  // activate dataflash command decoder
146  _spi->cs_assert();
147 
148  // Read manufacturer and ID command...
149  _spi->transfer(DF_READ_MANUFACTURER_AND_DEVICE_ID);
150 
151  df_manufacturer = _spi->transfer(0xff);
152  df_device = _spi->transfer(0xff);
153  df_device = (df_device<<8) | _spi->transfer(0xff);
154  _spi->transfer(0xff);
155 
156  // release SPI bus for use by other sensors
157  _spi->cs_release();
158 }

205 void DataFlash_APM2::PageToBuffer(uint8_t BufferNum, uint16_t PageAdr)
206 {
207  if (!_sem_take(1))
208  return;
209 
210  // activate dataflash command decoder
211  _spi->cs_assert();
212 
213  uint8_t cmd[4];
214  cmd[0] = BufferNum?DF_TRANSFER_PAGE_TO_BUFFER_2:DF_TRANSFER_PAGE_TO_BUFFER_1;
215  if(df_PageSize==512) {
216  cmd[1] = (uint8_t)(PageAdr >> 7);
217  cmd[2] = (uint8_t)(PageAdr << 1);
218  }else{
219  cmd[1] = (uint8_t)(PageAdr >> 6);
220  cmd[2] = (uint8_t)(PageAdr << 2);
221  }
222  cmd[3] = 0;
223  _spi->transfer(cmd, sizeof(cmd));
224 
225  //initiate the transfer
226  _spi->cs_release();
227  _spi->cs_assert();
228 
229  while(!ReadStatus()) ; //monitor the status register, wait until busy-flag is high
230 
231  // release SPI bus for use by other sensors
232  _spi->cs_release();
233  _spi_sem->give();
234 }

266 void DataFlash_APM2::BlockWrite (uint8_t BufferNum, uint16_t IntPageAdr,
267  const void *pHeader, uint8_t hdr_size,
268  const void *pBuffer, uint16_t size)
269 {
270  if (!_sem_take(1))
271  return;
272  // activate dataflash command decoder
273  _spi->cs_assert();
274 
275  uint8_t cmd[] = {
276  (uint8_t)(BufferNum?DF_BUFFER_2_WRITE:DF_BUFFER_1_WRITE),
277  0,
278  (uint8_t)(IntPageAdr>>8),
279  (uint8_t)(IntPageAdr)
280  };
281  _spi->transfer(cmd, sizeof(cmd));
282 
283  // transfer header, if any
284  if (hdr_size != 0) {
285  _spi->transfer((const uint8_t *)pHeader, hdr_size);
286  }
287 
288  // transfer data
289  _spi->transfer((const uint8_t *)pBuffer, size);
290 
291  // release SPI bus for use by other sensors
292  _spi->cs_release();
293  _spi_sem->give();
294 }

296 bool DataFlash_APM2::BlockRead(uint8_t BufferNum, uint16_t IntPageAdr, void *pBuffer, uint16_t size)
297 {
298  if (!_sem_take(1))
299  return false;
300 
301  // activate dataflash command decoder
302  _spi->cs_assert();
303 
304  if (BufferNum==0)
305  _spi->transfer(DF_BUFFER_1_READ);
306  else
307  _spi->transfer(DF_BUFFER_2_READ);
308 
309  _spi->transfer(0x00);
310  _spi->transfer((uint8_t)(IntPageAdr>>8)); //upper part of internal buffer address
311  _spi->transfer((uint8_t)(IntPageAdr)); //lower part of internal buffer address
312  _spi->transfer(0x00); //don't cares
313 
314  uint8_t *pData = (uint8_t *)pBuffer;
315  while (size--) {
316  *pData++ = _spi->transfer(0x00);
317  }
318 
319  // release SPI bus for use by other sensors
320  _spi->cs_release();
321 
322  _spi_sem->give();
323  return true;
324 }
