459  float &trim_roll,

461 {
462  uint8_t num_accels = min(get_accel_count(), INS_MAX_INSTANCES);
463  Vector3f samples[INS_MAX_INSTANCES][6];
464  Vector3f new_offsets[INS_MAX_INSTANCES];
465  Vector3f new_scaling[INS_MAX_INSTANCES];
466  Vector3f orig_offset[INS_MAX_INSTANCES];
467  Vector3f orig_scale[INS_MAX_INSTANCES];
468  uint8_t num_ok = 0;
469 
470  // exit immediately if calibration is already in progress
471  if (_calibrating) {
472  return false;
473  }
474 
475  _calibrating = true;
476 
477  /*
478  we do the accel calibration with no board rotation. This avoids
479  having to rotate readings during the calibration
480  */
481  enum Rotation saved_orientation = _board_orientation;
482  _board_orientation = ROTATION_NONE;
483 
484  for (uint8_t k=0; k<num_accels; k++) {
485  // backup original offsets and scaling
486  orig_offset[k] = _accel_offset[k].get();
487  orig_scale[k] = _accel_scale[k].get();
488 
489  // clear accelerometer offsets and scaling
490  _accel_offset[k] = Vector3f(0,0,0);
491  _accel_scale[k] = Vector3f(1,1,1);
492  }
493 
494  memset(samples, 0, sizeof(samples));
495 
496  // capture data from 6 positions
497  for (uint8_t i=0; i<6; i++) {
498  const prog_char_t *msg;
499 
500  // display message to user
501  switch ( i ) {
502  case 0:
503  msg = PSTR("level");
504  break;
505  case 1:
506  msg = PSTR("on its LEFT side");
507  break;
508  case 2:
509  msg = PSTR("on its RIGHT side");
510  break;
511  case 3:
512  msg = PSTR("nose DOWN");
513  break;
514  case 4:
515  msg = PSTR("nose UP");
516  break;
517  default: // default added to avoid compiler warning
518  case 5:
519  msg = PSTR("on its BACK");
520  break;
521  }
522  interact->printf_P(
523  PSTR("Place vehicle %S and press any key.\n"), msg);
524 
525  // wait for user input
526  if (!interact->blocking_read()) {
527  //No need to use interact->printf_P for an error, blocking_read does this when it fails
528  goto failed;
529  }
530 
531  const uint8_t update_dt_milliseconds = (uint8_t)(1000.0f/get_sample_rate()+0.5f);
532 
533  // wait 100ms for ins filter to rise
534  for (uint8_t k=0; k<100/update_dt_milliseconds; k++) {
535  wait_for_sample();
536  update();
537  hal.scheduler->delay(update_dt_milliseconds);
538  }
539 
540  uint32_t num_samples = 0;
541  while (num_samples < 400/update_dt_milliseconds) {
542  wait_for_sample();
543  // read samples from ins
544  update();
545  // capture sample
546  for (uint8_t k=0; k<num_accels; k++) {
547  Vector3f samp;
548  if(get_delta_velocity(k,samp) && _delta_velocity_dt[k] > 0) {
549  samp /= _delta_velocity_dt[k];
550  } else {
551  samp = get_accel(k);
552  }
553  samples[k][i] += samp;
554  if (!get_accel_health(k)) {
555  interact->printf_P(PSTR("accel[%u] not healthy"), (unsigned)k);
556  goto failed;
557  }
558  }
559  hal.scheduler->delay(update_dt_milliseconds);
560  num_samples++;
561  }
562  for (uint8_t k=0; k<num_accels; k++) {
563  samples[k][i] /= num_samples;
564  }
565  }
566 
567  // run the calibration routine
568  for (uint8_t k=0; k<num_accels; k++) {
569  if (!_check_sample_range(samples[k], saved_orientation, interact)) {
570  interact->printf_P(PSTR("Insufficient accel range"));
571  continue;
572  }
573 
574  bool success = _calibrate_accel(samples[k], new_offsets[k], new_scaling[k], saved_orientation);
575 
576  interact->printf_P(PSTR("Offsets[%u]: %.2f %.2f %.2f\n"),
577  (unsigned)k,
578  (double)new_offsets[k].x, (double)new_offsets[k].y, (double)new_offsets[k].z);
579  interact->printf_P(PSTR("Scaling[%u]: %.2f %.2f %.2f\n"),
580  (unsigned)k,
581  (double)new_scaling[k].x, (double)new_scaling[k].y, (double)new_scaling[k].z);
582  if (success) num_ok++;
583  }
584 
585  if (num_ok == num_accels) {
586  interact->printf_P(PSTR("Calibration successful\n"));
587 
588  for (uint8_t k=0; k<num_accels; k++) {
589  // set and save calibration
590  _accel_offset[k].set(new_offsets[k]);
591  _accel_scale[k].set(new_scaling[k]);
592  }
593  for (uint8_t k=num_accels; k<INS_MAX_INSTANCES; k++) {
594  // clear unused accelerometer's scaling and offsets
595  _accel_offset[k] = Vector3f(0,0,0);
596  _accel_scale[k] = Vector3f(0,0,0);
597  }
598  _save_parameters();

601  calculate the trims as well from primary accels
602  We use the original board rotation for this sample
603  */
604  Vector3f level_sample = samples[0][0];
605  level_sample.x *= new_scaling[0].x;
606  level_sample.y *= new_scaling[0].y;
607  level_sample.z *= new_scaling[0].z;
608  level_sample -= new_offsets[0];

770 void
771 AP_InertialSensor::_init_gyro()
772 {
773  uint8_t num_gyros = min(get_gyro_count(), INS_MAX_INSTANCES);
774  Vector3f last_average[INS_MAX_INSTANCES], best_avg[INS_MAX_INSTANCES];
775  Vector3f new_gyro_offset[INS_MAX_INSTANCES];
776  float best_diff[INS_MAX_INSTANCES];
777  bool converged[INS_MAX_INSTANCES];
778 
779  // exit immediately if calibration is already in progress
780  if (_calibrating) {
781  return;
782  }
783 
784  // record we are calibrating
785  _calibrating = true;
786 
787  // flash leds to tell user to keep the IMU still
788  AP_Notify::flags.initialising = true;
789 
790  // cold start
791  hal.console->print_P(PSTR("Init Gyro"));
792 
793  /*
794  we do the gyro calibration with no board rotation. This avoids
795  having to rotate readings during the calibration
796  */
797  enum Rotation saved_orientation = _board_orientation;
798  _board_orientation = ROTATION_NONE;
799 
800  // remove existing gyro offsets
801  for (uint8_t k=0; k<num_gyros; k++) {
802  _gyro_offset[k].set(Vector3f());
803  new_gyro_offset[k].zero();
804  best_diff[k] = 0;
805  last_average[k].zero();
806  converged[k] = false;
807  }
808 
809  for(int8_t c = 0; c < 5; c++) {
810  hal.scheduler->delay(5);
811  update();
812  }
813 
814  // the strategy is to average 50 points over 0.5 seconds, then do it
815  // again and see if the 2nd average is within a small margin of
816  // the first
817 
818  uint8_t num_converged = 0;
819 
820  // we try to get a good calibration estimate for up to 30 seconds
821  // if the gyros are stable, we should get it in 1 second
822  for (int16_t j = 0; j <= 30*4 && num_converged < num_gyros; j++) {
823  Vector3f gyro_sum[INS_MAX_INSTANCES], gyro_avg[INS_MAX_INSTANCES], gyro_diff[INS_MAX_INSTANCES];
824  Vector3f accel_start;
825  float diff_norm[INS_MAX_INSTANCES];
826  uint8_t i;
827 
828  memset(diff_norm, 0, sizeof(diff_norm));
829 
830  hal.console->print_P(PSTR("*"));
831 
832  for (uint8_t k=0; k<num_gyros; k++) {
833  gyro_sum[k].zero();
834  }
835  accel_start = get_accel(0);
836  for (i=0; i<50; i++) {
837  update();
838  for (uint8_t k=0; k<num_gyros; k++) {
839  gyro_sum[k] += get_gyro(k);
840  }
841  hal.scheduler->delay(5);
842  }
843 
844  Vector3f accel_diff = get_accel(0) - accel_start;
845  if (accel_diff.length() > 0.2f) {
846  // the accelerometers changed during the gyro sum. Skip
847  // this sample. This copes with doing gyro cal on a
848  // steadily moving platform. The value 0.2 corresponds
849  // with around 5 degrees/second of rotation.
850  continue;
851  }
852 
853  for (uint8_t k=0; k<num_gyros; k++) {
854  gyro_avg[k] = gyro_sum[k] / i;
855  gyro_diff[k] = last_average[k] - gyro_avg[k];
856  diff_norm[k] = gyro_diff[k].length();
857  }
858 
859  for (uint8_t k=0; k<num_gyros; k++) {
860  if (j == 0) {
861  best_diff[k] = diff_norm[k];
862  best_avg[k] = gyro_avg[k];
863  } else if (gyro_diff[k].length() < ToRad(0.1f)) {
864  // we want the average to be within 0.1 bit, which is 0.04 degrees/s
865  last_average[k] = (gyro_avg[k] * 0.5f) + (last_average[k] * 0.5f);
866  if (!converged[k] || last_average[k].length() < new_gyro_offset[k].length()) {
867  new_gyro_offset[k] = last_average[k];
868  }
869  if (!converged[k]) {
870  converged[k] = true;
871  num_converged++;
872  }
873  } else if (diff_norm[k] < best_diff[k]) {
874  best_diff[k] = diff_norm[k];
875  best_avg[k] = (gyro_avg[k] * 0.5f) + (last_average[k] * 0.5f);
876  }
877  last_average[k] = gyro_avg[k];
878  }
879  }
880 
881  // we've kept the user waiting long enough - use the best pair we
882  // found so far
883  hal.console->println();
884  for (uint8_t k=0; k<num_gyros; k++) {
885  if (!converged[k]) {
886  hal.console->printf_P(PSTR("gyro[%u] did not converge: diff=%f dps\n"),
887  (unsigned)k, (double)ToDeg(best_diff[k]));
888  _gyro_offset[k] = best_avg[k];
889  // flag calibration as failed for this gyro
890  _gyro_cal_ok[k] = false;
891  } else {
892  _gyro_cal_ok[k] = true;
893  _gyro_offset[k] = new_gyro_offset[k];
894  }
895  }
896 
897  // restore orientation
898  _board_orientation = saved_orientation;
899 
900  // record calibration complete
901  _calibrating = false;
902 
903  // stop flashing leds
904  AP_Notify::flags.initialising = false;
905 }

956 // _calibrate_model - perform low level accel calibration
957 // accel_sample are accelerometer samples collected in 6 different positions
958 // accel_offsets are output from the calibration routine
959 // accel_scale are output from the calibration routine
960 // returns true if successful
961 bool AP_InertialSensor::_calibrate_accel(const Vector3f accel_sample[6],
962  Vector3f& accel_offsets, Vector3f& accel_scale,
963  enum Rotation rotation)
964 {
965  int16_t i;
966  int16_t num_iterations = 0;
967  float eps = 0.000000001f;
968  float change = 100.0f;
969  float data[3];
970  float beta[6];
971  float delta[6];
972  float ds[6];
973  float JS[6][6];
974  bool success = true;
975 
976  // reset
977  beta[0] = beta[1] = beta[2] = 0;
978  beta[3] = beta[4] = beta[5] = 1.0f/GRAVITY_MSS;
979 
980  while( num_iterations < 20 && change > eps ) {
981  num_iterations++;
982 
983  _calibrate_reset_matrices(ds, JS);
984 
985  for( i=0; i<6; i++ ) {
986  data[0] = accel_sample[i].x;
987  data[1] = accel_sample[i].y;
988  data[2] = accel_sample[i].z;
989  _calibrate_update_matrices(ds, JS, beta, data);
990  }
991 
992  _calibrate_find_delta(ds, JS, delta);
993 
994  change = delta[0]*delta[0] +
995  delta[0]*delta[0] +
996  delta[1]*delta[1] +
997  delta[2]*delta[2] +
998  delta[3]*delta[3] / (beta[3]*beta[3]) +
999  delta[4]*delta[4] / (beta[4]*beta[4]) +
1000  delta[5]*delta[5] / (beta[5]*beta[5]);
1001 
1002  for( i=0; i<6; i++ ) {
1003  beta[i] -= delta[i];
1004  }
1005  }
1006 
1007  // copy results out
1008  accel_scale.x = beta[3] * GRAVITY_MSS;
1009  accel_scale.y = beta[4] * GRAVITY_MSS;
1010  accel_scale.z = beta[5] * GRAVITY_MSS;
1011  accel_offsets.x = beta[0] * accel_scale.x;
1012  accel_offsets.y = beta[1] * accel_scale.y;
1013  accel_offsets.z = beta[2] * accel_scale.z;

1017  success = false;

1021  success = false;
1022  }
1023 
1024  // return success or failure
1025  return success;

1028 void AP_InertialSensor::_calibrate_update_matrices(float dS[6], float JS[6][6],
1029  float beta[6], float data[3])
1030 {
1031  int16_t j, k;
1032  float dx, b;
1033  float residual = 1.0f;
1034  float jacobian[6];
1035 
1036  for( j=0; j<3; j++ ) {
1037  b = beta[3+j];
1038  dx = (float)data[j] - beta[j];
1039  residual -= b*b*dx*dx;
1040  jacobian[j] = 2.0f*b*b*dx;
1041  jacobian[3+j] = -2.0f*b*dx*dx;
1042  }
1043 
1044  for( j=0; j<6; j++ ) {
1045  dS[j] += jacobian[j]*residual;
1046  for( k=0; k<6; k++ ) {
1047  JS[j][k] += jacobian[j]*jacobian[k];
1048  }
1049  }

1065 void AP_InertialSensor::_calibrate_find_delta(float dS[6], float JS[6][6], float delta[6])
1066 {
1067  //Solve 6-d matrix equation JS*x = dS
1068  //first put in upper triangular form
1069  int16_t i,j,k;
1070  float mu;
1071 
1072  //make upper triangular
1073  for( i=0; i<6; i++ ) {
1074  //eliminate all nonzero entries below JS[i][i]
1075  for( j=i+1; j<6; j++ ) {
1076  mu = JS[i][j]/JS[i][i];
1077  if( !is_zero(mu) ) {
1078  dS[j] -= mu*dS[i];
1079  for( k=j; k<6; k++ ) {
1080  JS[k][j] -= mu*JS[k][i];
1081  }
1082  }
1083  }
1084  }
1085 
1086  //back-substitute
1087  for( i=5; i>=0; i-- ) {
1088  dS[i] /= JS[i][i];
1089  JS[i][i] = 1.0f;
1090 
1091  for( j=0; j<i; j++ ) {
1092  mu = JS[i][j];
1093  dS[j] -= mu*dS[i];
1094  JS[i][j] = 0.0f;
1095  }
1096  }
1097 
1098  for( i=0; i<6; i++ ) {
1099  delta[i] = dS[i];
1100  }

1184  determines the timing of the main loop in ardupilot.
1185 
1186  Ideally this function would return at exactly the rate given by the
1187  sample_rate argument given to AP_InertialSensor::init().
1188 
1189  The key output of this function is _delta_time, which is the time
1190  over which the gyro and accel integration will happen for this
1191  sample. We want that to be a constant time if possible, but if
1192  delays occur we need to cope with them. The long term sum of
1193  _delta_time should be exactly equal to the wall clock elapsed time
1194  */
1195 void AP_InertialSensor::wait_for_sample(void)
1196 {
1197  if (_have_sample) {
1198  // the user has called wait_for_sample() again without
1199  // consuming the sample with update()
1200  return;
1201  }
1202 
1203  uint32_t now = hal.scheduler->micros();
1204 
1205  if (_next_sample_usec == 0 && _delta_time <= 0) {
1206  // this is the first call to wait_for_sample()
1207  _last_sample_usec = now - _sample_period_usec;
1208  _next_sample_usec = now + _sample_period_usec;
1209  goto check_sample;
1210  }
1211 
1212  // see how long it is till the next sample is due
1213  if (_next_sample_usec - now <=_sample_period_usec) {
1214  // we're ahead on time, schedule next sample at expected period
1215  uint32_t wait_usec = _next_sample_usec - now;
1216  hal.scheduler->delay_microseconds_boost(wait_usec);
1217  uint32_t now2 = hal.scheduler->micros();
1218  if (now2+100 < _next_sample_usec) {
1219  timing_printf("shortsleep %u\n", (unsigned)(_next_sample_usec-now2));
1220  }
