137 /* try to synchronise simulation time with wall clock time, taking
138  into account desired speedup */
139 void Aircraft::sync_frame_time(void)
140 {
141  uint64_t now = get_wall_time_us();
142  uint64_t dt_us = now - last_wall_time_us;
143  if (dt_us < scaled_frame_time_us) {
144  usleep(scaled_frame_time_us - dt_us);
145  now = get_wall_time_us();
146 
147  if (now > last_wall_time_us && now - last_wall_time_us < 1.0e5) {
148  float rate = 1.0e6f/(now - last_wall_time_us);
149  achieved_rate_hz = (0.98f*achieved_rate_hz) + (0.02f*rate);
150  if (achieved_rate_hz < rate_hz * target_speedup) {
151  scaled_frame_time_us *= 0.999;
152  } else {
153  scaled_frame_time_us *= 1.001;
154  }
155  }
156 
157  }
158  last_wall_time_us = now;
159 }

172 /*
173  normal distribution random numbers
174  See
175  http://en.literateprograms.org/index.php?title=Special:DownloadCode/Box-Muller_transform_%28C%29&oldid=7011
176 */
177 double Aircraft::rand_normal(double mean, double stddev)
178 {
179  static double n2 = 0.0;
180  static int n2_cached = 0;
181  if (!n2_cached)
182  {
183  double x, y, r;
184  do
185  {
186  x = 2.0*rand()/RAND_MAX - 1;
187  y = 2.0*rand()/RAND_MAX - 1;
188 
189  r = x*x + y*y;
190  }
191  while (r == 0.0 || r > 1.0);
192  {
193  double d = sqrt(-2.0*log(r)/r);
194  double n1 = x*d;
195  n2 = y*d;
196  double result = n1*stddev + mean;
197  n2_cached = 1;
198  return result;
199  }
200  }
201  else
202  {
203  n2_cached = 0;
204  return n2*stddev + mean;
205  }
206 }
