55 // check heartbeat status
56 static void update_heartbeat(void)
57 {
58  static uint8_t last_hb_primary;
59  static uint8_t last_hb_backup;
60  static uint32_t last_primary_t;
61  static uint32_t last_backup_t;
62  uint8_t hb_primary = digitalRead(HEARTBEAT_PRIMARY_PIN);
63  uint8_t hb_backup = digitalRead(HEARTBEAT_BACKUP_PIN);
64  uint32_t tnow = millis();
65 // Serial.println(hb_primary);
66  // we consider the heartbeat to be OK if it changed in the
67  // last 0.4 seconds
68  const uint32_t hb_check_time = 400;
69  if (hb_primary != last_hb_primary) {
70  last_hb_primary = hb_primary;
71  last_primary_t = tnow;
72  //Serial.println("hb1 change");
73  }
74  if (hb_backup != last_hb_backup) {
75  last_hb_backup = hb_backup;
76  last_backup_t = tnow;
77  //Serial.println("hb2 change");
78  }
79  if (tnow < hb_check_time) {
80  // not enough time has passed for a measurement
81  return;
82  }
83  heartbeat_primary_ok = ((tnow - last_primary_t) < hb_check_time);
84  heartbeat_backup_ok = ((tnow - last_backup_t) < hb_check_time);
85 }

89 {
90  switch ((enum mux_mode)mode) {

168 void loop()
169 {
170  static uint32_t last_status_t;
171  uint32_t tnow = millis();
172  static uint8_t led_state;
173  static bool has_terminated = false;
174  static uint8_t termination_counter;
175  static uint16_t loop_counter;
176 
177  loop_counter++;
178 
179  // check for heartbeat
180  update_heartbeat();
181 
182  // see if we are in manual mode. Note that on the Uno,
183  // pulseIn is not very accurate. The +90 brings us much closer
184  // to the real pulse width
185  uint16_t mode_pwm = pulseIn(MODE_PWM_PIN, HIGH, 25000) + 90;
186  uint16_t manual_mode = (mode_pwm > 1750 && mode_pwm < 2100);
187 
188  // check the state of the terminate pins
189  uint8_t terminate_primary = digitalRead(TERMINATE_PRIMARY_PIN);
190  uint8_t terminate_backup = digitalRead(TERMINATE_BACKUP_PIN);
191 
192  // see if the OBC mode jumper is set
193  uint8_t obc_mode = digitalRead(OBC_MODE_PIN);
194 
195  if (tnow - last_status_t > 1000) {
196  last_status_t = tnow;
197  // show status once a second
198  Serial.print("OBC:"); Serial.print(obc_mode);
199  Serial.print(" Mode:"); Serial.print(mode_pwm);
200  Serial.print(" HB1:"); Serial.print(heartbeat_primary_ok);
201  Serial.print(" HB2:"); Serial.print(heartbeat_backup_ok);
202  Serial.print(" TERM1:"); Serial.print(terminate_primary);
203  Serial.print(" TERM2:"); Serial.print(terminate_backup);
204  Serial.print(" TERMINATED:"); Serial.print(has_terminated);
205  Serial.print(" LOOP:"); Serial.print(loop_counter);
206  Serial.println();
207  delayMicroseconds(5000);
208  loop_counter = 0;
209  // flash LED once a second so we know failsafe board
210  // is working
211  led_state = !led_state;
212  digitalWrite(LED_PIN, led_state);
213  }
214 
215  // allow reset via serial connection
216  if (Serial.available() > 0) {
217  char c = (char)Serial.read();
218  if (c == 'r') {
219  Serial.println("RESET BOARD");
220  has_terminated = false;
221  termination_counter = 0;
222  }
223  }
224 
225  // if we are not in OBC mode, and the mode control
226  // channel is high, then give RC control
227  if (!obc_mode && manual_mode) {
228  // give manual control via RC
229  set_mux_mode(MUX_MODE_RC);
230  return;
231  }
232 
233 #if BACKUP_AUTOPILOT_INSTALLED == 0
234  terminate_backup = false;
235 #endif
236 
237  // see if termination is set by a functioning autopilot
238  if (heartbeat_primary_ok && terminate_primary) {
239  termination_counter++;
240  } else if (heartbeat_backup_ok && terminate_backup) {
241  termination_counter++;
242  } else if (obc_mode && !heartbeat_primary_ok && !heartbeat_backup_ok) {
243  // if in OBC mode and neither autopilot is OK, then
244  // terminate
245  termination_counter++;
246  } else {
247  termination_counter = 0;
248  }
249 
250  // use the termination counter to debounce the termination
251  // pins
252  if (termination_counter > 10) {
253  termination_counter = 0;
254  has_terminated = true;
255  }
256 
257  // if we have terminated then setup the servos
258  if (has_terminated) {
259  set_servos_terminate(obc_mode);
260  return;
261  }
262 
263  bool heartbeat_ok = heartbeat_primary_ok;
264 #if BACKUP_AUTOPILOT_INSTALLED
265  if (heartbeat_backup_ok) {
266  heartbeat_ok = true;
267  }
268 #endif
269 
270  if (heartbeat_ok) {
271  // at least one autopilot is healthy
272  if (manual_mode) {
273  // we want manual/RC control
274  set_mux_mode(MUX_MODE_RC);
275  } else if (heartbeat_primary_ok) {
276  set_mux_mode(MUX_MODE_PRIMARY);
277  } else {
278  set_mux_mode(MUX_MODE_BACKUP);
279  }
280  } else {
281  // neither autopilot is OK
282  if (obc_mode) {
283  set_servos_terminate(obc_mode);
284  } else {
285  // give RC control
286  set_mux_mode(MUX_MODE_RC);
287  }
288  }
289 }
