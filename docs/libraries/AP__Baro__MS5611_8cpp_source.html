157 AP_Baro_MS5611::AP_Baro_MS5611(AP_Baro &baro, AP_SerialBus *serial, bool use_timer) :
158  AP_Baro_Backend(baro),
159  _serial(serial),
160  _updated(false),
161  _state(0),
162  _last_timer(0),
163  _use_timer(use_timer)
164 {
165  _instance = _frontend.register_sensor();
166  _serial->init();
167  if (!_serial->sem_take_blocking()){
168  hal.scheduler->panic(PSTR("PANIC: AP_Baro_MS5611: failed to take serial semaphore for init"));
169  }
170 
171  _serial->write(CMD_MS5611_RESET);
172  hal.scheduler->delay(4);
173 
174  // We read the factory calibration
175  // The on-chip CRC is not used
176  C1 = _serial->read_16bits(CMD_MS5611_PROM_C1);
177  C2 = _serial->read_16bits(CMD_MS5611_PROM_C2);
178  C3 = _serial->read_16bits(CMD_MS5611_PROM_C3);
179  C4 = _serial->read_16bits(CMD_MS5611_PROM_C4);
180  C5 = _serial->read_16bits(CMD_MS5611_PROM_C5);
181  C6 = _serial->read_16bits(CMD_MS5611_PROM_C6);
182 
183  if (!_check_crc()) {
184  hal.scheduler->panic(PSTR("Bad CRC on MS5611"));
185  }
186 
187  // Send a command to read Temp first
188  _serial->write(CMD_CONVERT_D2_OSR4096);
189  _last_timer = hal.scheduler->micros();
190  _state = 0;
191 
192  _s_D1 = 0;
193  _s_D2 = 0;
194  _d1_count = 0;
195  _d2_count = 0;
196 
197  _serial->sem_give();
198 
199  if (_use_timer) {
200  hal.scheduler->register_timer_process( AP_HAL_MEMBERPROC(&AP_Baro_MS5611::_timer));
201  }
202 }

207 bool AP_Baro_MS5611::_check_crc(void)
208 {
209  int16_t cnt;
210  uint16_t n_rem;
211  uint16_t crc_read;
212  uint8_t n_bit;
213  uint16_t n_prom[8] = { _serial->read_16bits(CMD_MS5611_PROM_Setup),
214  C1, C2, C3, C4, C5, C6,
215  _serial->read_16bits(CMD_MS5611_PROM_CRC) };
216  n_rem = 0x00;
217 
218  /* save the read crc */
219  crc_read = n_prom[7];
220 
221  /* remove CRC byte */
222  n_prom[7] = (0xFF00 & (n_prom[7]));
223 
224  for (cnt = 0; cnt < 16; cnt++) {
225  /* uneven bytes */
226  if (cnt & 1) {
227  n_rem ^= (uint8_t)((n_prom[cnt >> 1]) & 0x00FF);
228 
229  } else {
230  n_rem ^= (uint8_t)(n_prom[cnt >> 1] >> 8);
231  }
232 
233  for (n_bit = 8; n_bit > 0; n_bit--) {
234  if (n_rem & 0x8000) {
235  n_rem = (n_rem << 1) ^ 0x3000;
236 
237  } else {
238  n_rem = (n_rem << 1);
239  }
240  }
241  }
242 
243  /* final 4 bit remainder is CRC value */
244  n_rem = (0x000F & (n_rem >> 12));
245  n_prom[7] = crc_read;
246 
247  /* return true if CRCs match */
248  return (0x000F & crc_read) == (n_rem ^ 0x00);
249 }

252 /*
253  Read the sensor. This is a state machine
254  We read one time Temperature (state=1) and then 4 times Pressure (states 2-5)
255  temperature does not change so quickly...
256 */
257 void AP_Baro_MS5611::_timer(void)
258 {
259  // Throttle read rate to 100hz maximum.
260  if (hal.scheduler->micros() - _last_timer < 10000) {
261  return;
262  }
263 
264  if (!_serial->sem_take_nonblocking()) {
265  return;
266  }
267 
268  if (_state == 0) {
269  // On state 0 we read temp
270  uint32_t d2 = _serial->read_24bits(0);
271  if (d2 != 0) {
272  _s_D2 += d2;
273  _d2_count++;
274  if (_d2_count == 32) {
275  // we have summed 32 values. This only happens
276  // when we stop reading the barometer for a long time
277  // (more than 1.2 seconds)
278  _s_D2 >>= 1;
279  _d2_count = 16;
280  }
281  }
282  _state++;
283  _serial->write(CMD_CONVERT_D1_OSR4096); // Command to read pressure
284  } else {
285  uint32_t d1 = _serial->read_24bits(0);;
286  if (d1 != 0) {
287  // occasional zero values have been seen on the PXF
288  // board. These may be SPI errors, but safest to ignore
289  _s_D1 += d1;
290  _d1_count++;
291  if (_d1_count == 128) {
292  // we have summed 128 values. This only happens
293  // when we stop reading the barometer for a long time
294  // (more than 1.2 seconds)
295  _s_D1 >>= 1;
296  _d1_count = 64;
297  }
298  // Now a new reading exists
299  _updated = true;
300  }
301  _state++;
302  if (_state == 5) {
303  _serial->write(CMD_CONVERT_D2_OSR4096); // Command to read temperature
304  _state = 0;
305  } else {
306  _serial->write(CMD_CONVERT_D1_OSR4096); // Command to read pressure
307  }
308  }
309 
310  _last_timer = hal.scheduler->micros();
311  _serial->sem_give();
312 }

314 void AP_Baro_MS5611::update()
315 {
316  if (!_use_timer) {
317  // if we're not using the timer then accumulate one more time
318  // to cope with the calibration loop and minimise lag
319  accumulate();
320  }
321 
322  if (!_updated) {
323  return;
324  }
325  uint32_t sD1, sD2;
326  uint8_t d1count, d2count;
327 
328  // Suspend timer procs because these variables are written to
329  // in "_update".
330  hal.scheduler->suspend_timer_procs();
331  sD1 = _s_D1; _s_D1 = 0;
332  sD2 = _s_D2; _s_D2 = 0;
333  d1count = _d1_count; _d1_count = 0;
334  d2count = _d2_count; _d2_count = 0;
335  _updated = false;
336  hal.scheduler->resume_timer_procs();
337 
338  if (d1count != 0) {
339  D1 = ((float)sD1) / d1count;
340  }
341  if (d2count != 0) {
342  D2 = ((float)sD2) / d2count;
343  }
344  _calculate();
345 }

347 // Calculate Temperature and compensated Pressure in real units (Celsius degrees*100, mbar*100).
348 void AP_Baro_MS5611::_calculate()
349 {
350  float dT;
351  float TEMP;
352  float OFF;
353  float SENS;
354 
355  // Formulas from manufacturer datasheet
356  // sub -20c temperature compensation is not included
357 
358  // we do the calculations using floating point
359  // as this is much faster on an AVR2560, and also allows
360  // us to take advantage of the averaging of D1 and D1 over
361  // multiple samples, giving us more precision
362  dT = D2-(((uint32_t)C5)<<8);
363  TEMP = (dT * C6)/8388608;
364  OFF = C2 * 65536.0f + (C4 * dT) / 128;
365  SENS = C1 * 32768.0f + (C3 * dT) / 256;
366 
367  if (TEMP < 0) {
368  // second order temperature compensation when under 20 degrees C
369  float T2 = (dT*dT) / 0x80000000;
370  float Aux = TEMP*TEMP;
371  float OFF2 = 2.5f*Aux;
372  float SENS2 = 1.25f*Aux;
373  TEMP = TEMP - T2;
374  OFF = OFF - OFF2;
375  SENS = SENS - SENS2;
376  }
377 
378  float pressure = (D1*SENS/2097152 - OFF)/32768;
379  float temperature = (TEMP + 2000) * 0.01f;
380  _copy_to_frontend(_instance, pressure, temperature);
381 }
