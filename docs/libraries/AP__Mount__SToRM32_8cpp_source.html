26 // update mount position - should be called periodically
27 void AP_Mount_SToRM32::update()
28 {
29  // exit immediately if not initialised
30  if (!_initialised) {
31  return;
32  }
33 
34  // flag to trigger sending target angles to gimbal
35  bool resend_now = false;
36 
37  // update based on mount mode
38  switch(get_mode()) {
39  // move mount to a "retracted" position. To-Do: remove support and replace with a relaxed mode?
40  case MAV_MOUNT_MODE_RETRACT:
41  {
42  const Vector3f &target = _state._retract_angles.get();
43  _angle_ef_target_rad.x = ToRad(target.x);
44  _angle_ef_target_rad.y = ToRad(target.y);
45  _angle_ef_target_rad.z = ToRad(target.z);
46  }
47  break;
48 
49  // move mount to a neutral position, typically pointing forward
50  case MAV_MOUNT_MODE_NEUTRAL:
51  {
52  const Vector3f &target = _state._neutral_angles.get();
53  _angle_ef_target_rad.x = ToRad(target.x);
54  _angle_ef_target_rad.y = ToRad(target.y);
55  _angle_ef_target_rad.z = ToRad(target.z);
56  }
57  break;
58 
59  // point to the angles given by a mavlink message
60  case MAV_MOUNT_MODE_MAVLINK_TARGETING:
61  // do nothing because earth-frame angle targets (i.e. _angle_ef_target_rad) should have already been set by a MOUNT_CONTROL message from GCS
62  break;
63 
64  // RC radio manual angle control, but with stabilization from the AHRS
65  case MAV_MOUNT_MODE_RC_TARGETING:
66  // update targets using pilot's rc inputs
67  update_targets_from_rc();
68  resend_now = true;
69  break;
70 
71  // point mount to a GPS point given by the mission planner
72  case MAV_MOUNT_MODE_GPS_POINT:
73  if(_frontend._ahrs.get_gps().status() >= AP_GPS::GPS_OK_FIX_2D) {
74  calc_angle_to_location(_state._roi_target, _angle_ef_target_rad, true, true);
75  resend_now = true;
76  }
77  break;
78 
79  default:
80  // we do not know this mode so do nothing
81  break;
82  }
83 
84  // resend target angles at least once per second
85  if (resend_now || ((hal.scheduler->millis() - _last_send) > AP_MOUNT_STORM32_RESEND_MS)) {
86  send_do_mount_control(ToDeg(_angle_ef_target_rad.y), ToDeg(_angle_ef_target_rad.x), ToDeg(_angle_ef_target_rad.z), MAV_MOUNT_MODE_MAVLINK_TARGETING);
87  }
