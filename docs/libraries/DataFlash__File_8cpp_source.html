193 /* Write a block of data at current offset */
194 void DataFlash_File::WriteBlock(const void *pBuffer, uint16_t size)
195 {
196  if (_write_fd == -1 || !_initialised || _open_error || !_writes_enabled) {
197  return;
198  }
199  uint16_t _head;
200  uint16_t space = BUF_SPACE(_writebuf);
201  if (space < size) {
202  // discard the whole write, to keep the log consistent
203  perf_count(_perf_overruns);
204  return;
205  }
206 
207  if (_writebuf_tail < _head) {
208  // perform as single memcpy
209  assert(_writebuf_tail+size <= _writebuf_size);
210  memcpy(&_writebuf[_writebuf_tail], pBuffer, size);
211  BUF_ADVANCETAIL(_writebuf, size);
212  } else {
213  // perform as two memcpy calls
214  uint16_t n = _writebuf_size - _writebuf_tail;
215  if (n > size) n = size;
216  assert(_writebuf_tail+n <= _writebuf_size);
217  memcpy(&_writebuf[_writebuf_tail], pBuffer, n);
218  BUF_ADVANCETAIL(_writebuf, n);
219  pBuffer = (const void *)(((const uint8_t *)pBuffer) + n);
220  n = size - n;
221  if (n > 0) {
222  assert(_writebuf_tail+n <= _writebuf_size);
223  memcpy(&_writebuf[_writebuf_tail], pBuffer, n);
224  BUF_ADVANCETAIL(_writebuf, n);
225  }
226  }

311 int16_t DataFlash_File::get_log_data(uint16_t log_num, uint16_t page, uint32_t offset, uint16_t len, uint8_t *data)
312 {
313  if (!_initialised || _open_error) {
314  return -1;
315  }
316  if (_read_fd != -1 && log_num != _read_fd_log_num) {
317  ::close(_read_fd);
318  _read_fd = -1;
319  }
320  if (_read_fd == -1) {
321  char *fname = _log_file_name(log_num);
322  if (fname == NULL) {
323  return -1;
324  }
325  stop_logging();
326  _read_fd = ::open(fname, O_RDONLY);
327  if (_read_fd == -1) {
328  _open_error = true;
329  int saved_errno = errno;
330  ::printf("Log read open fail for %s - %s\n",
331  fname, strerror(saved_errno));
332  hal.console->printf("Log read open fail for %s - %s\n",
333  fname, strerror(saved_errno));
334  free(fname);
335  return -1;
336  }
337  free(fname);
338  _read_offset = 0;
339  _read_fd_log_num = log_num;
340  }
341  uint32_t ofs = page * (uint32_t)DATAFLASH_PAGE_SIZE + offset;
342 
343  /*
344  this rather strange bit of code is here to work around a bug
345  in file offsets in NuttX. Every few hundred blocks of reads
346  (starting at around 350k into a file) NuttX will get the
347  wrong offset for sequential reads. The offset it gets is
348  typically 128k earlier than it should be. It turns out that
349  calling lseek() with 0 offset and SEEK_CUR works around the
350  bug. We can remove this once we find the real bug.
351  */
352  if (ofs / 4096 != (ofs+len) / 4096) {
353  off_t seek_current = ::lseek(_read_fd, 0, SEEK_CUR);
354  if (seek_current != (off_t)_read_offset) {
355  ::lseek(_read_fd, _read_offset, SEEK_SET);
356  }
357  }
358 
359  if (ofs != _read_offset) {
360  ::lseek(_read_fd, ofs, SEEK_SET);
361  _read_offset = ofs;
362  }
363  int16_t ret = (int16_t)::read(_read_fd, data, len);
364  if (ret > 0) {
365  _read_offset += ret;
366  }
367  return ret;
368 }

466  Read the log and print it on port
467 */
468 void DataFlash_File::LogReadProcess(uint16_t log_num,
469  uint16_t start_page, uint16_t end_page,
470  void (*print_mode)(AP_HAL::BetterStream *port, uint8_t mode),
471  AP_HAL::BetterStream *port)
472 {
473  uint8_t log_step = 0;
474  if (!_initialised || _open_error) {
475  return;
476  }
477  if (_read_fd != -1) {
478  ::close(_read_fd);
479  _read_fd = -1;
480  }
481  char *fname = _log_file_name(log_num);
482  if (fname == NULL) {
483  return;
484  }
485  _read_fd = ::open(fname, O_RDONLY);
486  free(fname);
487  if (_read_fd == -1) {
488  return;
489  }
490  _read_fd_log_num = log_num;
491  _read_offset = 0;
492  if (start_page != 0) {
493  ::lseek(_read_fd, start_page * DATAFLASH_PAGE_SIZE, SEEK_SET);
494  _read_offset = start_page * DATAFLASH_PAGE_SIZE;
495  }
496 
497  uint8_t log_counter = 0;
498 
499  while (true) {
500  uint8_t data;
501  if (::read(_read_fd, &data, 1) != 1) {
502  // reached end of file
503  break;
504  }
505  _read_offset++;
506 
507  // This is a state machine to read the packets
508  switch(log_step) {
509  case 0:
510  if (data == HEAD_BYTE1) {
511  log_step++;
512  }
513  break;
514 
515  case 1:
516  if (data == HEAD_BYTE2) {
517  log_step++;
518  } else {
519  log_step = 0;
520  }
521  break;
522 
523  case 2:
524  log_step = 0;
525  _print_log_entry(data, print_mode, port);
526  log_counter++;
527  if (log_counter == 10) {
528  log_counter = 0;
529  ::lseek(_read_fd, 0, SEEK_CUR);
530  }
531  break;
532  }
533  if (_read_offset >= (end_page+1) * DATAFLASH_PAGE_SIZE) {
534  break;
535  }
536  }
537 
538  ::close(_read_fd);
539  _read_fd = -1;

603 void DataFlash_File::_io_timer(void)
604 {
605  uint16_t _tail;
606  if (_write_fd == -1 || !_initialised || _open_error) {
607  return;
608  }
609 
610  uint16_t nbytes = BUF_AVAILABLE(_writebuf);
611  if (nbytes == 0) {
612  return;
613  }
614  uint32_t tnow = hal.scheduler->micros();
615  if (nbytes < _writebuf_chunk &&
616  tnow - _last_write_time < 2000000UL) {
617  // write in 512 byte chunks, but always write at least once
618  // per 2 seconds if data is available
619  return;
620  }
621 
622  perf_begin(_perf_write);
623 
624  _last_write_time = tnow;
625  if (nbytes > _writebuf_chunk) {
626  // be kind to the FAT PX4 filesystem
627  nbytes = _writebuf_chunk;
628  }
629  if (_writebuf_head > _tail) {
630  // only write to the end of the buffer
631  nbytes = min(nbytes, _writebuf_size - _writebuf_head);
632  }
633 
634  // try to align writes on a 512 byte boundary to avoid filesystem
635  // reads
636  if ((nbytes + _write_offset) % 512 != 0) {
637  uint32_t ofs = (nbytes + _write_offset) % 512;
638  if (ofs < nbytes) {
639  nbytes -= ofs;
640  }
641  }
642 
643  assert(_writebuf_head+nbytes <= _writebuf_size);
644  ssize_t nwritten = ::write(_write_fd, &_writebuf[_writebuf_head], nbytes);
645  if (nwritten <= 0) {
646  perf_count(_perf_errors);
647  close(_write_fd);
648  _write_fd = -1;
649  _initialised = false;
650  } else {
651  _write_offset += nwritten;
652  /*
653  the best strategy for minimising corruption on microSD cards
654  seems to be to write in 4k chunks and fsync the file on each
655  chunk, ensuring the directory entry is updated after each
656  write.
657  */
658  BUF_ADVANCEHEAD(_writebuf, nwritten);
659 #if CONFIG_HAL_BOARD != HAL_BOARD_SITL && CONFIG_HAL_BOARD_SUBTYPE != HAL_BOARD_SUBTYPE_LINUX_NONE
660  ::fsync(_write_fd);
661 #endif
662  }
663  perf_end(_perf_write);
664 }

666 #endif // HAL_OS_POSIX_IO
