493 void AC_PosControl::get_stopping_point_xy(Vector3f &stopping_point) const
494 {
495  const Vector3f curr_pos = _inav.get_position();
496  Vector3f curr_vel = _inav.get_velocity();
497  float linear_distance; // the distance at which we swap from a linear to sqrt response
498  float linear_velocity; // the velocity above which we swap from a linear to sqrt response
499  float stopping_dist; // the distance within the vehicle can stop
500  float kP = _p_pos_xy.kP();
501 
502  // add velocity error to current velocity
503  if (is_active_xy()) {
504  curr_vel.x += _vel_error.x;
505  curr_vel.y += _vel_error.y;
506  }
507 
508  // calculate current velocity
509  float vel_total = pythagorous2(curr_vel.x, curr_vel.y);
510 
511  // avoid divide by zero by using current position if the velocity is below 10cm/s, kP is very low or acceleration is zero
512  if (kP <= 0.0f || _accel_cms <= 0.0f || is_zero(vel_total)) {
513  stopping_point.x = curr_pos.x;
514  stopping_point.y = curr_pos.y;
515  return;
516  }
517 
518  // calculate point at which velocity switches from linear to sqrt
519  linear_velocity = _accel_cms/kP;
520 
521  // calculate distance within which we can stop
522  if (vel_total < linear_velocity) {
523  stopping_dist = vel_total/kP;
524  } else {
525  linear_distance = _accel_cms/(2.0f*kP*kP);
526  stopping_dist = linear_distance + (vel_total*vel_total)/(2.0f*_accel_cms);
527  }
528 
529  // constrain stopping distance
530  stopping_dist = constrain_float(stopping_dist, 0, _leash);
531 
532  // convert the stopping distance into a stopping point using velocity vector
533  stopping_point.x = curr_pos.x + (stopping_dist * curr_vel.x / vel_total);
534  stopping_point.y = curr_pos.y + (stopping_dist * curr_vel.y / vel_total);
535 }

708 void AC_PosControl::pos_to_rate_xy(xy_mode mode, float dt, float ekfNavVelGainScaler)
709 {
710  Vector3f curr_pos = _inav.get_position();
711  float linear_distance; // the distance we swap between linear and sqrt velocity response
712  float kP = ekfNavVelGainScaler * _p_pos_xy.kP(); // scale gains to compensate for noisy optical flow measurement in the EKF
713 
714  // avoid divide by zero
715  if (kP <= 0.0f) {
716  _vel_target.x = 0.0f;
717  _vel_target.y = 0.0f;
718  }else{
719  // calculate distance error
720  _pos_error.x = _pos_target.x - curr_pos.x;
721  _pos_error.y = _pos_target.y - curr_pos.y;
722 
723  // constrain target position to within reasonable distance of current location
724  _distance_to_target = pythagorous2(_pos_error.x, _pos_error.y);
725  if (_distance_to_target > _leash && _distance_to_target > 0.0f) {
726  _pos_target.x = curr_pos.x + _leash * _pos_error.x/_distance_to_target;
727  _pos_target.y = curr_pos.y + _leash * _pos_error.y/_distance_to_target;
728  // re-calculate distance error
729  _pos_error.x = _pos_target.x - curr_pos.x;
730  _pos_error.y = _pos_target.y - curr_pos.y;
731  _distance_to_target = _leash;
732  }
733 
734  // calculate the distance at which we swap between linear and sqrt velocity response
735  linear_distance = _accel_cms/(2.0f*kP*kP);
736 
737  if (_distance_to_target > 2.0f*linear_distance) {
738  // velocity response grows with the square root of the distance
739  float vel_sqrt = safe_sqrt(2.0f*_accel_cms*(_distance_to_target-linear_distance));
740  _vel_target.x = vel_sqrt * _pos_error.x/_distance_to_target;
741  _vel_target.y = vel_sqrt * _pos_error.y/_distance_to_target;
742  }else{
743  // velocity response grows linearly with the distance
744  _vel_target.x = _p_pos_xy.kP() * _pos_error.x;
745  _vel_target.y = _p_pos_xy.kP() * _pos_error.y;
746  }
747 
748  if (mode == XY_MODE_POS_LIMITED_AND_VEL_FF) {
749  // this mode is for loiter - rate-limiting the position correction
750  // allows the pilot to always override the position correction in
751  // the event of a disturbance
752 
753  // scale velocity within limit
754  float vel_total = pythagorous2(_vel_target.x, _vel_target.y);
755  if (vel_total > POSCONTROL_VEL_XY_MAX_FROM_POS_ERR) {
756  _vel_target.x = POSCONTROL_VEL_XY_MAX_FROM_POS_ERR * _vel_target.x/vel_total;
757  _vel_target.y = POSCONTROL_VEL_XY_MAX_FROM_POS_ERR * _vel_target.y/vel_total;
758  }
759 
760  // add velocity feed-forward
761  _vel_target.x += _vel_desired.x;
762  _vel_target.y += _vel_desired.y;
763  } else {
764  if (mode == XY_MODE_POS_AND_VEL_FF) {
765  // add velocity feed-forward
766  _vel_target.x += _vel_desired.x;
767  _vel_target.y += _vel_desired.y;
768  }
769 
770  // scale velocity within speed limit
771  float vel_total = pythagorous2(_vel_target.x, _vel_target.y);
772  if (vel_total > _speed_cms) {
773  _vel_target.x = _speed_cms * _vel_target.x/vel_total;
774  _vel_target.y = _speed_cms * _vel_target.y/vel_total;
775  }
776  }
777  }

782 void AC_PosControl::rate_to_accel_xy(float dt, float ekfNavVelGainScaler)
783 {
784  const Vector3f &vel_curr = _inav.get_velocity(); // current velocity in cm/s
785  float accel_total; // total acceleration in cm/s/s
786  Vector2f vel_xy_p, vel_xy_i;
787 
788  // reset last velocity target to current target
789  if (_flags.reset_rate_to_accel_xy) {
790  _vel_last.x = _vel_target.x;
791  _vel_last.y = _vel_target.y;
792  _flags.reset_rate_to_accel_xy = false;
793  }
794 
795  // feed forward desired acceleration calculation
796  if (dt > 0.0f) {
797  if (!_flags.freeze_ff_xy) {
798  _accel_feedforward.x = (_vel_target.x - _vel_last.x)/dt;
799  _accel_feedforward.y = (_vel_target.y - _vel_last.y)/dt;
800  } else {
801  // stop the feed forward being calculated during a known discontinuity
802  _flags.freeze_ff_xy = false;
803  }
804  } else {
805  _accel_feedforward.x = 0.0f;
806  _accel_feedforward.y = 0.0f;
807  }
808 
809  // store this iteration's velocities for the next iteration
810  _vel_last.x = _vel_target.x;
811  _vel_last.y = _vel_target.y;
812 
813  // calculate velocity error
814  _vel_error.x = _vel_target.x - vel_curr.x;
815  _vel_error.y = _vel_target.y - vel_curr.y;
816 
817  // call pi controller
818  _pi_vel_xy.set_input(_vel_error);
819 
820  // get p
821  vel_xy_p = _pi_vel_xy.get_p();
822 
823  // update i term if we have not hit the accel or throttle limits OR the i term will reduce
824  if ((!_limit.accel_xy && !_motors.limit.throttle_upper)) {
825  vel_xy_i = _pi_vel_xy.get_i();
826  } else {
827  vel_xy_i = _pi_vel_xy.get_i_shrink();
828  }
829 
830  // combine feed forward accel with PID output from velocity error and scale PID output to compensate for optical flow measurement induced EKF noise
831  _accel_target.x = _accel_feedforward.x + (vel_xy_p.x + vel_xy_i.x) * ekfNavVelGainScaler;
832  _accel_target.y = _accel_feedforward.y + (vel_xy_p.y + vel_xy_i.y) * ekfNavVelGainScaler;
833 
834  // scale desired acceleration if it's beyond acceptable limit
835  // To-Do: move this check down to the accel_to_lean_angle method?
836  accel_total = pythagorous2(_accel_target.x, _accel_target.y);
837  if (accel_total > POSCONTROL_ACCEL_XY_MAX && accel_total > 0.0f) {
838  _accel_target.x = POSCONTROL_ACCEL_XY_MAX * _accel_target.x/accel_total;
839  _accel_target.y = POSCONTROL_ACCEL_XY_MAX * _accel_target.y/accel_total;
840  _limit.accel_xy = true; // unused
841  } else {
842  // reset accel limit flag
843  _limit.accel_xy = false;
844  }
845 }

849 void AC_PosControl::accel_to_lean_angles(float dt, float ekfNavVelGainScaler)
850 {
851  float accel_right, accel_forward;
852  float lean_angle_max = _attitude_control.lean_angle_max();
853 
854  // reset accel to current desired acceleration
855  if (_flags.reset_accel_to_lean_xy) {
856  _accel_target_jerk_limited.x = _accel_target.x;
857  _accel_target_jerk_limited.y = _accel_target.y;
858  _accel_target_filter.reset(Vector2f(_accel_target.x, _accel_target.y));
859  _flags.reset_accel_to_lean_xy = false;
860  }
861 
862  // apply jerk limit of 17 m/s^3 - equates to a worst case of about 100 deg/sec/sec
863  float max_delta_accel = dt * POSCONTROL_JERK_LIMIT_CMSSS;
864 
865  Vector2f accel_in(_accel_target.x, _accel_target.y);
866  Vector2f accel_change = accel_in-_accel_target_jerk_limited;
867  float accel_change_length = accel_change.length();
868 
869  if(accel_change_length > max_delta_accel) {
870  accel_change *= max_delta_accel/accel_change_length;
871  }
872  _accel_target_jerk_limited += accel_change;
873 
874  // lowpass filter on NE accel
875  _accel_target_filter.set_cutoff_frequency(min(_accel_xy_filt_hz, 5.0f*ekfNavVelGainScaler));
876  Vector2f accel_target_filtered = _accel_target_filter.apply(_accel_target_jerk_limited, dt);
877 
878  // rotate accelerations into body forward-right frame
879  accel_forward = accel_target_filtered.x*_ahrs.cos_yaw() + accel_target_filtered.y*_ahrs.sin_yaw();
880  accel_right = -accel_target_filtered.x*_ahrs.sin_yaw() + accel_target_filtered.y*_ahrs.cos_yaw();
881 
882  // update angle targets that will be passed to stabilize controller
883  _pitch_target = constrain_float(atanf(-accel_forward/(GRAVITY_MSS * 100))*(18000/M_PI_F),-lean_angle_max, lean_angle_max);
884  float cos_pitch_target = cosf(_pitch_target*M_PI_F/18000);
885  _roll_target = constrain_float(atanf(accel_right*cos_pitch_target/(GRAVITY_MSS * 100))*(18000/M_PI_F), -lean_angle_max, lean_angle_max);
886 }
