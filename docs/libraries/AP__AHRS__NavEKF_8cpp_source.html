65 void AP_AHRS_NavEKF::update(void)
66 {
67  // we need to restore the old DCM attitude values as these are
68  // used internally in DCM to calculate error values for gyro drift
69  // correction
70  roll = _dcm_attitude.x;
71  pitch = _dcm_attitude.y;
72  yaw = _dcm_attitude.z;
73  update_cd_values();
74 
75  AP_AHRS_DCM::update();
76 
77  // keep DCM attitude available for get_secondary_attitude()
78  _dcm_attitude(roll, pitch, yaw);
79 
80  if (!ekf_started) {
81  // wait 1 second for DCM to output a valid tilt error estimate
82  if (start_time_ms == 0) {
83  start_time_ms = hal.scheduler->millis();
84  }
85  if (hal.scheduler->millis() - start_time_ms > startup_delay_ms) {
86  ekf_started = EKF.InitialiseFilterDynamic();
87  }
88  }
89  if (ekf_started) {
90  EKF.UpdateFilter();
91  EKF.getRotationBodyToNED(_dcm_matrix);
92  if (using_EKF()) {
93  Vector3f eulers;
94  EKF.getEulerAngles(eulers);
95  roll = eulers.x;
96  pitch = eulers.y;
97  yaw = eulers.z;
98 
99  update_cd_values();
100  update_trig();
101 
102  // keep _gyro_bias for get_gyro_drift()
103  EKF.getGyroBias(_gyro_bias);
104  _gyro_bias = -_gyro_bias;
105 
106  // calculate corrected gryo estimate for get_gyro()
107  _gyro_estimate.zero();
108  uint8_t healthy_count = 0;
109  for (uint8_t i=0; i<_ins.get_gyro_count(); i++) {
110  if (_ins.get_gyro_health(i) && healthy_count < 2) {
111  _gyro_estimate += _ins.get_gyro(i);
112  healthy_count++;
113  }
114  }
115  if (healthy_count > 1) {
116  _gyro_estimate /= healthy_count;
117  }
118  _gyro_estimate += _gyro_bias;
119 
120  float abias1, abias2;
121  EKF.getAccelZBias(abias1, abias2);
122 
123  // update _accel_ef_ekf
124  for (uint8_t i=0; i<_ins.get_accel_count(); i++) {
125  Vector3f accel = _ins.get_accel(i);
126  if (i==0) {
127  accel.z -= abias1;
128  } else if (i==1) {
129  accel.z -= abias2;
130  }
131  if (_ins.get_accel_health(i)) {
132  _accel_ef_ekf[i] = _dcm_matrix * accel;
133  }
134  }
135 
136  if(_ins.get_accel_health(0) && _ins.get_accel_health(1)) {
137  float IMU1_weighting;
138  EKF.getIMU1Weighting(IMU1_weighting);
139  _accel_ef_ekf_blended = _accel_ef_ekf[0] * IMU1_weighting + _accel_ef_ekf[1] * (1.0f-IMU1_weighting);
140  } else {
141  _accel_ef_ekf_blended = _accel_ef_ekf[0];
142  }
143  }
144  }
145 }

313 bool AP_AHRS_NavEKF::using_EKF(void) const
314 {
315  uint8_t ekf_faults;
316  EKF.getFilterFaults(ekf_faults);
317  // If EKF is started we switch away if it reports unhealthy. This could be due to bad
318  // sensor data. If EKF reversion is inhibited, we only switch across if the EKF encounters
319  // an internal processing error, but not for bad sensor data.
320  bool ret = ekf_started && ((_ekf_use == EKF_USE_WITH_FALLBACK && EKF.healthy()) || (_ekf_use == EKF_USE_WITHOUT_FALLBACK && ekf_faults == 0));
321  if (!ret) {
322  return false;
323  }
324 
325  if (_vehicle_class == AHRS_VEHICLE_FIXED_WING ||
326  _vehicle_class == AHRS_VEHICLE_GROUND) {
327  nav_filter_status filt_state;
328  EKF.getFilterStatus(filt_state);
329  if (hal.util->get_soft_armed() && !filt_state.flags.using_gps && _gps.status() >= AP_GPS::GPS_OK_FIX_3D) {
330  // if the EKF is not fusing GPS and we have a 3D lock, then
331  // plane and rover would prefer to use the GPS position from
332  // DCM. This is a safety net while some issues with the EKF
333  // get sorted out
334  return false;
335  }
336  if (hal.util->get_soft_armed() && filt_state.flags.const_pos_mode) {
337  return false;
338  }
339  if (!filt_state.flags.attitude ||
340  !filt_state.flags.horiz_vel ||
341  !filt_state.flags.vert_vel ||
342  !filt_state.flags.horiz_pos_abs ||
343  !filt_state.flags.vert_pos) {
344  return false;
345  }
346  }
347  return ret;
348 }

350 /*
351  check if the AHRS subsystem is healthy
352 */
353 bool AP_AHRS_NavEKF::healthy(void) const
354 {
355  // If EKF is started we switch away if it reports unhealthy. This could be due to bad
356  // sensor data. If EKF reversion is inhibited, we only switch across if the EKF encounters
357  // an internal processing error, but not for bad sensor data.
358  if (_ekf_use != EKF_DO_NOT_USE) {
359  bool ret = ekf_started && EKF.healthy();
360  if (!ret) {
361  return false;
362  }
363  if ((_vehicle_class == AHRS_VEHICLE_FIXED_WING ||
364  _vehicle_class == AHRS_VEHICLE_GROUND) &&
365  !using_EKF()) {
366  // on fixed wing we want to be using EKF to be considered
367  // healthy if EKF is enabled
368  return false;
369  }
370  return true;
371  }
372  return AP_AHRS_DCM::healthy();
373 }
