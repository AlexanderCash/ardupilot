84  turn_angle is > 90 then a 90 degree turn distance is used, otherwise

115 void AP_L1_Control::_prevent_indecision(float &Nu)
116 {
117  const float Nu_limit = 0.9f*M_PI_F;
118  if (fabsf(Nu) > Nu_limit &&
119  fabsf(_last_Nu) > Nu_limit &&
120  fabsf(wrap_180_cd(_target_bearing_cd - _ahrs.yaw_sensor)) > 12000 &&
121  Nu * _last_Nu < 0.0f) {
122  // we are moving away from the target waypoint and pointing
123  // away from the waypoint (not flying backwards). The sign
124  // of Nu has also changed, which means we are
125  // oscillating in our decision about which way to go
126  Nu = _last_Nu;
127  }

130 // update L1 control for waypoint navigation
131 // this function costs about 3.5 milliseconds on a AVR2560
132 void AP_L1_Control::update_waypoint(const struct Location &prev_WP, const struct Location &next_WP)
133 {
134 
135  struct Location _current_loc;
136  float Nu;
137  float xtrackVel;
138  float ltrackVel;
139 
140  // Calculate L1 gain required for specified damping
141  float K_L1 = 4.0f * _L1_damping * _L1_damping;
142 
143  // Get current position and velocity
144  _ahrs.get_position(_current_loc);
145 
146  Vector2f _groundspeed_vector = _ahrs.groundspeed_vector();
147 
148  // update _target_bearing_cd
149  _target_bearing_cd = get_bearing_cd(_current_loc, next_WP);
150 
151  //Calculate groundspeed
152  float groundSpeed = _groundspeed_vector.length();
153  if (groundSpeed < 0.1f) {
154  // use a small ground speed vector in the right direction,
155  // allowing us to use the compass heading at zero GPS velocity
156  groundSpeed = 0.1f;
157  _groundspeed_vector = Vector2f(cosf(_ahrs.yaw), sinf(_ahrs.yaw)) * groundSpeed;
158  }
159 
160  // Calculate time varying control parameters
161  // Calculate the L1 length required for specified period
162  // 0.3183099 = 1/1/pipi
163  _L1_dist = 0.3183099f * _L1_damping * _L1_period * groundSpeed;
164 
165  // Calculate the NE position of WP B relative to WP A
166  Vector2f AB = location_diff(prev_WP, next_WP);
167 
168  // Check for AB zero length and track directly to the destination
169  // if too small
170  if (AB.length() < 1.0e-6f) {
171  AB = location_diff(_current_loc, next_WP);
172  if (AB.length() < 1.0e-6f) {
173  AB = Vector2f(cosf(_ahrs.yaw), sinf(_ahrs.yaw));
174  }
175  }
176  AB.normalize();
177 
178  // Calculate the NE position of the aircraft relative to WP A
179  Vector2f A_air = location_diff(prev_WP, _current_loc);
180 
181  // calculate distance to target track, for reporting
182  _crosstrack_error = AB % A_air;
183 
184  //Determine if the aircraft is behind a +-135 degree degree arc centred on WP A
185  //and further than L1 distance from WP A. Then use WP A as the L1 reference point
186  //Otherwise do normal L1 guidance
187  float WP_A_dist = A_air.length();
188  float alongTrackDist = A_air * AB;
189  if (WP_A_dist > _L1_dist && alongTrackDist/max(WP_A_dist, 1.0f) < -0.7071f)
190  {
191  //Calc Nu to fly To WP A
192  Vector2f A_air_unit = (A_air).normalized(); // Unit vector from WP A to aircraft
193  xtrackVel = _groundspeed_vector % (-A_air_unit); // Velocity across line
194  ltrackVel = _groundspeed_vector * (-A_air_unit); // Velocity along line
195  Nu = atan2f(xtrackVel,ltrackVel);
196  _nav_bearing = atan2f(-A_air_unit.y , -A_air_unit.x); // bearing (radians) from AC to L1 point
197 
198  } else { //Calc Nu to fly along AB line
199 
200  //Calculate Nu2 angle (angle of velocity vector relative to line connecting waypoints)
201  xtrackVel = _groundspeed_vector % AB; // Velocity cross track
202  ltrackVel = _groundspeed_vector * AB; // Velocity along track
203  float Nu2 = atan2f(xtrackVel,ltrackVel);
204  //Calculate Nu1 angle (Angle to L1 reference point)
205  float xtrackErr = A_air % AB;
206  float sine_Nu1 = xtrackErr/max(_L1_dist, 0.1f);
207  //Limit sine of Nu1 to provide a controlled track capture angle of 45 deg
208  sine_Nu1 = constrain_float(sine_Nu1, -0.7071f, 0.7071f);
209  float Nu1 = asinf(sine_Nu1);
210  Nu = Nu1 + Nu2;
211  _nav_bearing = atan2f(AB.y, AB.x) + Nu1; // bearing (radians) from AC to L1 point
212  }
213 
214  _prevent_indecision(Nu);
215  _last_Nu = Nu;
216 
217  //Limit Nu to +-pi
218  Nu = constrain_float(Nu, -1.5708f, +1.5708f);
219  _latAccDem = K_L1 * groundSpeed * groundSpeed / _L1_dist * sinf(Nu);
220 
221  // Waypoint capture status is always false during waypoint following
222  _WPcircle = false;
223 
224  _bearing_error = Nu; // bearing error angle (radians), +ve to left of track
225 }

227 // update L1 control for loitering
228 void AP_L1_Control::update_loiter(const struct Location &center_WP, float radius, int8_t loiter_direction)
229 {
230  struct Location _current_loc;
231 
232  // scale loiter radius with square of EAS2TAS to allow us to stay
233  // stable at high altitude
234  radius *= sq(_ahrs.get_EAS2TAS());
235 
236  // Calculate guidance gains used by PD loop (used during circle tracking)
237  float omega = (6.2832f / _L1_period);
238  float Kx = omega * omega;
239  float Kv = 2.0f * _L1_damping * omega;
240 
241  // Calculate L1 gain required for specified damping (used during waypoint capture)
242  float K_L1 = 4.0f * _L1_damping * _L1_damping;
243 
244  //Get current position and velocity
245  _ahrs.get_position(_current_loc);
246 
247  Vector2f _groundspeed_vector = _ahrs.groundspeed_vector();
248 
249  //Calculate groundspeed
250  float groundSpeed = max(_groundspeed_vector.length() , 1.0f);
251 
252 
253  // update _target_bearing_cd
254  _target_bearing_cd = get_bearing_cd(_current_loc, center_WP);
255 
256 
257  // Calculate time varying control parameters
258  // Calculate the L1 length required for specified period
259  // 0.3183099 = 1/pi
260  _L1_dist = 0.3183099f * _L1_damping * _L1_period * groundSpeed;
261 
262  //Calculate the NE position of the aircraft relative to WP A
263  Vector2f A_air = location_diff(center_WP, _current_loc);
264 
265  // Calculate the unit vector from WP A to aircraft
266  // protect against being on the waypoint and having zero velocity
267  // if too close to the waypoint, use the velocity vector
268  // if the velocity vector is too small, use the heading vector
269  Vector2f A_air_unit;
270  if (A_air.length() > 0.1f) {
271  A_air_unit = A_air.normalized();
272  } else {
273  if (_groundspeed_vector.length() < 0.1f) {
274  A_air_unit = Vector2f(cosf(_ahrs.yaw), sinf(_ahrs.yaw));
275  } else {
276  A_air_unit = _groundspeed_vector.normalized();
277  }
278  }
279 
280  //Calculate Nu to capture center_WP
281  float xtrackVelCap = A_air_unit % _groundspeed_vector; // Velocity across line - perpendicular to radial inbound to WP
282  float ltrackVelCap = - (_groundspeed_vector * A_air_unit); // Velocity along line - radial inbound to WP
283  float Nu = atan2f(xtrackVelCap,ltrackVelCap);
284 
285  _prevent_indecision(Nu);
286  _last_Nu = Nu;
287 
288  Nu = constrain_float(Nu, -M_PI_2, M_PI_2); //Limit Nu to +- Pi/2
289 
290  //Calculate lat accln demand to capture center_WP (use L1 guidance law)
291  float latAccDemCap = K_L1 * groundSpeed * groundSpeed / _L1_dist * sinf(Nu);
292 
293  //Calculate radial position and velocity errors
294  float xtrackVelCirc = -ltrackVelCap; // Radial outbound velocity - reuse previous radial inbound velocity
295  float xtrackErrCirc = A_air.length() - radius; // Radial distance from the loiter circle
296 
297  // keep crosstrack error for reporting
298  _crosstrack_error = xtrackErrCirc;
299 
300  //Calculate PD control correction to circle waypoint_ahrs.roll
301  float latAccDemCircPD = (xtrackErrCirc * Kx + xtrackVelCirc * Kv);
302 
303  //Calculate tangential velocity
304  float velTangent = xtrackVelCap * float(loiter_direction);
305 
306  //Prevent PD demand from turning the wrong way by limiting the command when flying the wrong way
307  if (ltrackVelCap < 0.0f && velTangent < 0.0f) {
308  latAccDemCircPD = max(latAccDemCircPD, 0.0f);
309  }
310 
311  // Calculate centripetal acceleration demand
312  float latAccDemCircCtr = velTangent * velTangent / max((0.5f * radius), (radius + xtrackErrCirc));
313 
314  //Sum PD control and centripetal acceleration to calculate lateral manoeuvre demand
315  float latAccDemCirc = loiter_direction * (latAccDemCircPD + latAccDemCircCtr);
316 
317  // Perform switchover between 'capture' and 'circle' modes at the
318  // point where the commands cross over to achieve a seamless transfer
319  // Only fly 'capture' mode if outside the circle
320  if (xtrackErrCirc > 0.0f && loiter_direction * latAccDemCap < loiter_direction * latAccDemCirc) {
321  _latAccDem = latAccDemCap;
322  _WPcircle = false;
323  _bearing_error = Nu; // angle between demanded and achieved velocity vector, +ve to left of track
324  _nav_bearing = atan2f(-A_air_unit.y , -A_air_unit.x); // bearing (radians) from AC to L1 point
325  } else {
326  _latAccDem = latAccDemCirc;
327  _WPcircle = true;
328  _bearing_error = 0.0f; // bearing error (radians), +ve to left of track
329  _nav_bearing = atan2f(-A_air_unit.y , -A_air_unit.x); // bearing (radians)from AC to L1 point
330  }
331 }

334 // update L1 control for heading hold navigation
335 void AP_L1_Control::update_heading_hold(int32_t navigation_heading_cd)
336 {
337  // Calculate normalised frequency for tracking loop
338  const float omegaA = 4.4428f/_L1_period; // sqrt(2)*pi/period
339  // Calculate additional damping gain
340 
341  int32_t Nu_cd;
342  float Nu;
343 
344  // copy to _target_bearing_cd and _nav_bearing
345  _target_bearing_cd = wrap_180_cd(navigation_heading_cd);
346  _nav_bearing = radians(navigation_heading_cd * 0.01f);
347 
348  Nu_cd = _target_bearing_cd - wrap_180_cd(_ahrs.yaw_sensor);
349  Nu_cd = wrap_180_cd(Nu_cd);
350  Nu = radians(Nu_cd * 0.01f);
351 
352  Vector2f _groundspeed_vector = _ahrs.groundspeed_vector();
353 
354  //Calculate groundspeed
355  float groundSpeed = _groundspeed_vector.length();
356 
357  // Calculate time varying control parameters
358  _L1_dist = groundSpeed / omegaA; // L1 distance is adjusted to maintain a constant tracking loop frequency
359  float VomegaA = groundSpeed * omegaA;
360 
361  // Waypoint capture status is always false during heading hold
362  _WPcircle = false;
363 
364  _crosstrack_error = 0;
365 
366  _bearing_error = Nu; // bearing error angle (radians), +ve to left of track
367 
368  // Limit Nu to +-pi
369  Nu = constrain_float(Nu, -M_PI_2, M_PI_2);
370  _latAccDem = 2.0f*sinf(Nu)*VomegaA;
371 }
