366 void AP_InertialSensor_MPU6000::_read_data_transaction() {
367  /* one resister address followed by seven 2-byte registers */
368  struct PACKED {
369  uint8_t cmd;
370  uint8_t int_status;
371  uint8_t v[14];
372  } rx, tx = { cmd : MPUREG_INT_STATUS | 0x80, };
373 
374  _spi->transaction((const uint8_t *)&tx, (uint8_t *)&rx, sizeof(rx));
375 
376  /*
377  detect a bad SPI bus transaction by looking for all 14 bytes
378  zero, or the wrong INT_STATUS register value. This is used to
379  detect a too high SPI bus speed.
380  */
381  uint8_t i;
382  for (i=0; i<14; i++) {
383  if (rx.v[i] != 0) break;
384  }
385  if ((rx.int_status&~0x6) != (_drdy_pin==NULL?0:BIT_RAW_RDY_INT) || i == 14) {
386  // likely a bad bus transaction
387  if (++_error_count > 4) {
388  _spi->set_bus_speed(AP_HAL::SPIDeviceDriver::SPI_SPEED_LOW);
389  }
390  }
391 
392 #define int16_val(v, idx) ((int16_t)(((uint16_t)v[2*idx] << 8) | v[2*idx+1]))
393 #if MPU6000_FAST_SAMPLING
394  _accel_filtered = _accel_filter.apply(Vector3f(int16_val(rx.v, 1),
395  int16_val(rx.v, 0),
396  -int16_val(rx.v, 2)));
397 
398  _gyro_filtered = _gyro_filter.apply(Vector3f(int16_val(rx.v, 5),
399  int16_val(rx.v, 4),
400  -int16_val(rx.v, 6)));
401 #else
402  _accel_sum.x += int16_val(rx.v, 1);
403  _accel_sum.y += int16_val(rx.v, 0);
404  _accel_sum.z -= int16_val(rx.v, 2);
405  _gyro_sum.x += int16_val(rx.v, 5);
406  _gyro_sum.y += int16_val(rx.v, 4);
407  _gyro_sum.z -= int16_val(rx.v, 6);
408 #endif
409  _sum_count++;
410 
411 #if !MPU6000_FAST_SAMPLING
412  if (_sum_count == 0) {
413  // rollover - v unlikely
414  _accel_sum.zero();
415  _gyro_sum.zero();
416  }
417 #endif
418 }

486 bool AP_InertialSensor_MPU6000::_hardware_init(void)
487 {
488  if (!_spi_sem->take(100)) {
489  hal.scheduler->panic(PSTR("MPU6000: Unable to get semaphore"));
490  }
491 
492  // initially run the bus at low speed (500kHz on APM2)
493  _spi->set_bus_speed(AP_HAL::SPIDeviceDriver::SPI_SPEED_LOW);
494 
495  // Chip reset
496  uint8_t tries;
497  for (tries = 0; tries<5; tries++) {
498  _register_write(MPUREG_PWR_MGMT_1, BIT_PWR_MGMT_1_DEVICE_RESET);
499  hal.scheduler->delay(100);
500 
501  // Wake up device and select GyroZ clock. Note that the
502  // MPU6000 starts up in sleep mode, and it can take some time
503  // for it to come out of sleep
504  _register_write(MPUREG_PWR_MGMT_1, BIT_PWR_MGMT_1_CLK_ZGYRO);
505  hal.scheduler->delay(5);
506 
507  // check it has woken up
508  if (_register_read(MPUREG_PWR_MGMT_1) == BIT_PWR_MGMT_1_CLK_ZGYRO) {
509  break;
510  }
511 #if MPU6000_DEBUG
512  _dump_registers();
513 #endif
514  }
515  if (tries == 5) {
516  hal.console->println_P(PSTR("Failed to boot MPU6000 5 times"));
517  _spi_sem->give();
518  return false;
519  }
520 
521  _register_write(MPUREG_PWR_MGMT_2, 0x00); // only used for wake-up in accelerometer only low power mode
522  hal.scheduler->delay(1);
523 
524  // Disable I2C bus (recommended on datasheet)

530 #else

532  // to minimise the effects of aliasing we choose a filter
533  // that is less than half of the sample rate

539  _sample_count = 4;
540  break;
541  case AP_InertialSensor::RATE_100HZ:

543  break;
544  case AP_InertialSensor::RATE_200HZ:

546  break;

549  }

551

554  _set_filter_register(256);
555 
556  // set sample rate to 1000Hz and apply a software filter
557  _register_write(MPUREG_SMPLRT_DIV, MPUREG_SMPLRT_1000HZ);
558 #else

562  // the requested rate to the application
563  _register_write(MPUREG_SMPLRT_DIV, MPUREG_SMPLRT_200HZ);
564 #endif
565  hal.scheduler->delay(1);
566 
567  _register_write(MPUREG_GYRO_CONFIG, BITS_GYRO_FS_2000DPS); // Gyro scale 2000/s
568  hal.scheduler->delay(1);
569 
570  // read the product ID rev c has 1/2 the sensitivity of rev d
571  _product_id = _register_read(MPUREG_PRODUCT_ID);
572  //Serial.printf("Product_ID= 0x%x\n", (unsigned) _mpu6000_product_id);
573 
574  if ((_product_id == MPU6000ES_REV_C4) ||
575  (_product_id == MPU6000ES_REV_C5) ||
576  (_product_id == MPU6000_REV_C4) ||
577  (_product_id == MPU6000_REV_C5)) {
578  // Accel scale 8g (4096 LSB/g)
579  // Rev C has different scaling than rev D
580  _register_write(MPUREG_ACCEL_CONFIG,1<<3);
581  } else {
582  // Accel scale 8g (4096 LSB/g)
583  _register_write(MPUREG_ACCEL_CONFIG,2<<3);
584  }
585  hal.scheduler->delay(1);
586 
587  // configure interrupt to fire when new data arrives
588  _register_write(MPUREG_INT_ENABLE, BIT_RAW_RDY_EN);
589  hal.scheduler->delay(1);
590 
591  // clear interrupt on any read, and hold the data ready pin high
592  // until we clear the interrupt
593  _register_write(MPUREG_INT_PIN_CFG, BIT_INT_RD_CLEAR | BIT_LATCH_INT_EN);
594 
595  // now that we have initialised, we set the SPI bus speed to high
596  // (8MHz on APM2)
597  _spi->set_bus_speed(AP_HAL::SPIDeviceDriver::SPI_SPEED_HIGH);
598 
599  _spi_sem->give();
600 
601  return true;
602 }
