120  /* Mode is to set the ISCn0 and ISCn1 bits.
121  * These correspond to the GPIO_INTERRUPT_ defs in AP_HAL.h */
122  if (!((mode == HAL_GPIO_INTERRUPT_LOW)||
123  (mode == HAL_GPIO_INTERRUPT_HIGH)||

133  EIMSK |= (1 << INT6);
