195 uint16_t AP_InertialSensor_LSM303D::_init_sensor( Sample_rate sample_rate )
196 {
197  if (_initialised) return _LSM303D_product_id;
198  _initialised = true;
199 
200  _spi = hal.spi->device(AP_HAL::SPIDevice_LSM303D);
201  _spi_sem = _spi->get_semaphore();
202 
203 // This device has mag and accel
204 #ifdef LSM303D_DRDY_X_PIN
205  _drdy_pin_x = hal.gpio->channel(LSM303D_DRDY_X_PIN);
206  _drdy_pin_x->mode(HAL_GPIO_INPUT);
207 #endif
208 
209 #ifdef LSM303D_DRDY_M_PIN
210  _drdy_pin_m = hal.gpio->channel(LSM303D_DRDY_M_PIN);
211  _drdy_pin_m->mode(HAL_GPIO_INPUT);
212 #endif
213 
214  hal.scheduler->suspend_timer_procs();
215 
216  // Test WHOAMI
217  uint8_t whoami = _register_read(ADDR_WHO_AM_I);
218  if (whoami != WHO_I_AM) {
219  // TODO: we should probably accept multiple chip
220  // revisions. This is the one on the PXF
221  hal.console->printf("LSM303D: unexpected WHOAMI 0x%x\n", (unsigned)whoami);
222  hal.scheduler->panic(PSTR("LSM303D: bad WHOAMI"));
223  }
224 
225  uint8_t tries = 0;
226  do {
227  bool success = _hardware_init(sample_rate);
228  if (success) {
229  hal.scheduler->delay(5+2);
230  if (!_spi_sem->take(100)) {
231  hal.scheduler->panic(PSTR("LSM303D: Unable to get semaphore"));
232  }
233  if (_data_ready()) {
234  _spi_sem->give();
235  break;
236  } else {
237  hal.console->println_P(
238  PSTR("LSM303D startup failed: no data ready"));
239  }
240  _spi_sem->give();
241  }
242  if (tries++ > 5) {
243  hal.scheduler->panic(PSTR("PANIC: failed to boot LSM303D 5 times"));
244  }
245  } while (1);
246 
247  hal.scheduler->resume_timer_procs();
248 
249 
250  /* read the first lot of data.
251  * _read_data_transaction requires the spi semaphore to be taken by
252  * its caller. */
253  _last_sample_time_micros = hal.scheduler->micros();
254  hal.scheduler->delay(10);
255  if (_spi_sem->take(100)) {
256  _read_data_transaction();
257  _spi_sem->give();
258  }
259 
260  // start the timer process to read samples
261  hal.scheduler->register_timer_process(AP_HAL_MEMBERPROC(&AP_InertialSensor_LSM303D::_poll_data));
262 
263 #if LSM303D_DEBUG
264  _dump_registers();
265 #endif
266  return _LSM303D_product_id;
267 }

350 void AP_InertialSensor_LSM303D::_poll_data(void)
351 {
352  if (hal.scheduler->in_timerprocess()) {
353  if (!_spi_sem->take_nonblocking()) {
354  /*
355  the semaphore being busy is an expected condition when the
356  mainline code is calling wait_for_sample() which will
357  grab the semaphore. We return now and rely on the mainline
358  code grabbing the latest sample.
359  */
360  return;
361  }
362  if (_data_ready()) {
363  _last_sample_time_micros = hal.scheduler->micros();
364  _read_data_transaction();
365  }
366  _spi_sem->give();
367  } else {
368  /* Synchronous read - take semaphore */
369  if (_spi_sem->take(10)) {
370  if (_data_ready()) {
371  _last_sample_time_micros = hal.scheduler->micros();
372  _read_data_transaction();
373  }
374  _spi_sem->give();
375  } else {
376  hal.scheduler->panic(
377  PSTR("PANIC: AP_InertialSensor_LSM303D::_poll_data "
378  "failed to take SPI semaphore synchronously"));
379  }
380  }

545 uint8_t AP_InertialSensor_LSM303D::accel_set_range(uint8_t max_g)

724 {

726  hal.scheduler->panic(PSTR("LSM303D: Unable to get semaphore"));
727  }
728 
729  // initially run the bus at low speed
730  _spi->set_bus_speed(AP_HAL::SPIDeviceDriver::SPI_SPEED_LOW);
731 
732  // ensure the chip doesn't interpret any other bus traffic as I2C
733  disable_i2c();
734 
735 
736  /* enable accel*/
737  _reg1_expected = REG1_X_ENABLE_A | REG1_Y_ENABLE_A | REG1_Z_ENABLE_A | REG1_BDU_UPDATE | REG1_RATE_800HZ_A;
738  _register_write(ADDR_CTRL_REG1, _reg1_expected);
739 
740  /* enable mag */
741  _reg7_expected = REG7_CONT_MODE_M;
742  _register_write(ADDR_CTRL_REG7, _reg7_expected);

744  _register_write(ADDR_CTRL_REG3, 0x04); // DRDY on ACCEL on INT1
745  _register_write(ADDR_CTRL_REG4, 0x04); // DRDY on MAG on INT2
746 
747  accel_set_range(LSM303D_ACCEL_DEFAULT_RANGE_G);
748  accel_set_samplerate(LSM303D_ACCEL_DEFAULT_RATE);
749

751  // we setup the anti-alias on-chip filter as 50Hz. We believe
752  // this operates in the analog domain, and is critical for
753  // anti-aliasing. The 2 pole software filter is designed to
754  // operate in conjunction with this on-chip filter
755  accel_set_onchip_lowpass_filter_bandwidth(LSM303D_ACCEL_DEFAULT_ONCHIP_FILTER_FREQ);
756 
757  mag_set_range(LSM303D_MAG_DEFAULT_RANGE_GA);
758  mag_set_samplerate(LSM303D_MAG_DEFAULT_RATE);
