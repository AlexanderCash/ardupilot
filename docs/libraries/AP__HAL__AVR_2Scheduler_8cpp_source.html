163 void AVRScheduler::_timer_isr_event() {
164  // we enable the interrupt again immediately and also enable
165  // interrupts. This allows other time critical interrupts to
166  // run (such as the serial receive interrupt). We catch the
167  // timer calls taking too long using _in_timer_call.
168  // This approach also gives us a nice uniform spacing between
169  // timer calls
170 
171  TCNT2 = _timer2_reset_value;
172  sei();
173  _run_timer_procs(true);
174 }

176 void AVRScheduler::_run_timer_procs(bool called_from_isr) {
177 
178  if (_in_timer_proc) {
179  // the timer calls took longer than the period of the
180  // timer. This is bad, and may indicate a serious
181  // driver failure. We can't just call the drivers
182  // again, as we could run out of stack. So we only
183  // call the _failsafe call. It's job is to detect if
184  // the drivers or the main loop are indeed dead and to
185  // activate whatever failsafe it thinks may help if
186  // need be. We assume the failsafe code can't
187  // block. If it does then we will recurse and die when
188  // we run out of stack
189  if (_failsafe != NULL) {
190  _failsafe();
191  }
192  return;
193  }
194 
195  _in_timer_proc = true;
196 
197  if (!_timer_suspended) {
198  // now call the timer based drivers
199  for (int i = 0; i < _num_timer_procs; i++) {
200  if (_timer_proc[i] != NULL) {
201  _timer_proc[i]();
202  }
203  }
204  } else if (called_from_isr) {
205  _timer_event_missed = true;
206  }
207 
208  // and the failsafe, if one is setup
209  if (_failsafe != NULL) {
210  _failsafe();
211  }
212 
213  _in_timer_proc = false;
214 }
