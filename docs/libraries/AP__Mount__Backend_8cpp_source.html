44 // control_msg - process MOUNT_CONTROL messages received from GCS
45 void AP_Mount_Backend::control_msg(mavlink_message_t *msg)
46 {
47  __mavlink_mount_control_t packet;
48  mavlink_msg_mount_control_decode(msg, &packet);
49 
50  // interpret message fields based on mode
51  switch (_frontend.get_mode(_instance)) {
52  case MAV_MOUNT_MODE_RETRACT:
53  case MAV_MOUNT_MODE_NEUTRAL:
54  // do nothing with request if mount is retracted or in neutral position
55  break;
56 
57  // set earth frame target angles from mavlink message
58  case MAV_MOUNT_MODE_MAVLINK_TARGETING:
59  set_angle_targets(packet.input_b*0.01f, packet.input_a*0.01f, packet.input_c*0.01f);
60  break;
61 
62  // Load neutral position and start RC Roll,Pitch,Yaw control with stabilization
63  case MAV_MOUNT_MODE_RC_TARGETING:
64  // do nothing if pilot is controlling the roll, pitch and yaw
65  break;
66 
67  // set lat, lon, alt position targets from mavlink message
68  case MAV_MOUNT_MODE_GPS_POINT:
69  Location target_location;
70  memset(&target_location, 0, sizeof(target_location));
71  target_location.lat = packet.input_a;
72  target_location.lng = packet.input_b;
73  target_location.alt = packet.input_c;
74  target_location.flags.relative_alt = true;
75  set_roi_target(target_location);
76  break;
77 
78  default:
79  // do nothing
80  break;
81  }
82 }

85 void AP_Mount_Backend::update_targets_from_rc()

87 #define rc_ch(i) RC_Channel::rc_channel(i-1)

89  uint8_t roll_rc_in = _state._roll_rc_in;
90  uint8_t tilt_rc_in = _state._tilt_rc_in;
91  uint8_t pan_rc_in = _state._pan_rc_in;
92 
93  // if joystick_speed is defined then pilot input defines a rate of change of the angle
94  if (_frontend._joystick_speed) {
95  // allow pilot speed position input to come directly from an RC_Channel
96  if (roll_rc_in && rc_ch(roll_rc_in)) {

98  constrain_float(_angle_ef_target_rad.x, radians(_state._roll_angle_min*0.01f), radians(_state._roll_angle_max*0.01f));
99  }
100  if (tilt_rc_in && (rc_ch(tilt_rc_in))) {

102  constrain_float(_angle_ef_target_rad.y, radians(_state._tilt_angle_min*0.01f), radians(_state._tilt_angle_max*0.01f));
103  }
104  if (pan_rc_in && (rc_ch(pan_rc_in))) {
105  _angle_ef_target_rad.z += rc_ch(pan_rc_in)->norm_input_dz() * 0.0001f * _frontend._joystick_speed;
106  constrain_float(_angle_ef_target_rad.z, radians(_state._pan_angle_min*0.01f), radians(_state._pan_angle_max*0.01f));
107  }

134 // calc_angle_to_location - calculates the earth-frame roll, tilt and pan angles (and radians) to point at the given target
135 void AP_Mount_Backend::calc_angle_to_location(const struct Location &target, Vector3f& angles_to_target_rad, bool calc_tilt, bool calc_pan)
136 {
137  float GPS_vector_x = (target.lng-_frontend._current_loc.lng)*cosf(ToRad((_frontend._current_loc.lat+target.lat)*0.00000005f))*0.01113195f;
138  float GPS_vector_y = (target.lat-_frontend._current_loc.lat)*0.01113195f;
139  float GPS_vector_z = (target.alt-_frontend._current_loc.alt); // baro altitude(IN CM) should be adjusted to known home elevation before take off (Set altimeter).
140  float target_distance = 100.0f*pythagorous2(GPS_vector_x, GPS_vector_y); // Careful , centimeters here locally. Baro/alt is in cm, lat/lon is in meters.
141 
142  // initialise all angles to zero
143  angles_to_target_rad.zero();
144 
145  // tilt calcs
146  if (calc_tilt) {
147  angles_to_target_rad.y = atan2f(GPS_vector_z, target_distance);
148  }
149 
150  // pan calcs
151  if (calc_pan) {
152  // calc absolute heading and then onvert to vehicle relative yaw
153  angles_to_target_rad.z = wrap_PI(atan2f(GPS_vector_x, GPS_vector_y) - _frontend._ahrs.yaw);
154  }
