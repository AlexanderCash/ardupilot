262 // rate_bf_to_motor_yaw - ask the rate controller to calculate the motor outputs to achieve the target rate in centi-degrees / second
263 float AC_AttitudeControl_Heli::rate_bf_to_motor_yaw(float rate_target_cds)
264 {
265  float pd,i,ff; // used to capture pid values for logging
266  float current_rate; // this iteration's rate
267  float rate_error; // simply target_rate - current_rate
268  float yaw_out;
269 
270  // get current rate
271  // To-Do: make getting gyro rates more efficient?
272  current_rate = (_ahrs.get_gyro().z * AC_ATTITUDE_CONTROL_DEGX100);
273 
274  // calculate error and call pid controller
275  rate_error = rate_target_cds - current_rate;
276 
277  // send input to PID controller
278  _pid_rate_yaw.set_input_filter_all(rate_error);
279 
280  // get p and d
281  pd = _pid_rate_yaw.get_p() + _pid_rate_yaw.get_d();
282 
283  // get i term
284  i = _pid_rate_yaw.get_integrator();
285 
286  // update i term as long as we haven't breached the limits or the I term will certainly reduce
287  if (!_flags_heli.limit_yaw || ((i>0&&rate_error<0)||(i<0&&rate_error>0))) {
288  if (((AP_MotorsHeli&)_motors).motor_runup_complete()) {
289  i = _pid_rate_yaw.get_i();
290  } else {
291  i = ((AC_HELI_PID&)_pid_rate_yaw).get_leaky_i(AC_ATTITUDE_HELI_RATE_INTEGRATOR_LEAK_RATE); // If motor is not running use leaky I-term to avoid excessive build-up
292  }
293  }
294 
295  ff = yaw_feedforward_filter.apply(((AC_HELI_PID&)_pid_rate_yaw).get_ff(rate_target_cds), _dt);
296 
297  // add feed forward
298  yaw_out = pd + i + ff;
299 
300  // constrain output and update limit flag
301  if (fabsf(yaw_out) > AC_ATTITUDE_RATE_YAW_CONTROLLER_OUT_MAX) {
302  yaw_out = constrain_float(yaw_out,-AC_ATTITUDE_RATE_YAW_CONTROLLER_OUT_MAX,AC_ATTITUDE_RATE_YAW_CONTROLLER_OUT_MAX);
303  _flags_heli.limit_yaw = true;
304  }else{
305  _flags_heli.limit_yaw = false;
306  }
307 
308  // output to motors
309  return yaw_out;
310 }
