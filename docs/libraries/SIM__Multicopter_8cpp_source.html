105 MultiCopter::MultiCopter(const char *home_str, const char *frame_str) :
106  Aircraft(home_str, frame_str),
107  frame(NULL),
108  hover_throttle(0.51),
109  terminal_velocity(15.0),
110  terminal_rotation_rate(4*radians(360.0))
111 {
112  for (uint8_t i=0; i<sizeof(supported_frames)/sizeof(supported_frames[0]); i++) {
113  if (strcasecmp(frame_str, supported_frames[i].name) == 0) {
114  frame = &supported_frames[i];
115  }
116  }
117  if (frame == NULL) {
118  printf("Frame '%s' not found", frame_str);
119  exit(1);
120  }
121  /*
122  scaling from total motor power to Newtons. Allows the copter
123  to hover against gravity when each motor is at hover_throttle
124  */
125  mass = 1.5;
126  thrust_scale = (mass * GRAVITY_MSS) / (frame->num_motors * hover_throttle);
127 
128  frame_height = 0.1;
129 }

131 /*
132  update the multicopter simulation by one time step
133  */
134 void MultiCopter::update(const struct sitl_input &input)
135 {
136  float motor_speed[frame->num_motors];
137 
138  for (uint8_t i=0; i<frame->num_motors; i++) {
139  uint16_t servo = input.servos[frame->motors[i].servo-1];
140  // assume 1000 to 2000 PWM range
141  if (servo <= 1000) {
142  motor_speed[i] = 0;
143  } else {
144  motor_speed[i] = (servo-1000) / 1000.0f;
145  }
146  }
147 
148  // how much time has passed?
149  float delta_time = frame_time_us * 1.0e-6f;
150 
151  // rotational acceleration, in rad/s/s, in body frame
152  Vector3f rot_accel;
153  float thrust = 0.0f; // newtons
154 
155  for (uint8_t i=0; i<frame->num_motors; i++) {
156  rot_accel.x += -radians(5000.0) * sinf(radians(frame->motors[i].angle)) * motor_speed[i];
157  rot_accel.y += radians(5000.0) * cosf(radians(frame->motors[i].angle)) * motor_speed[i];
158  if (frame->motors[i].clockwise) {
159  rot_accel.z -= motor_speed[i] * radians(400.0);
160  } else {
161  rot_accel.z += motor_speed[i] * radians(400.0);
162  }
163  thrust += motor_speed[i] * thrust_scale; // newtons
164  }
165 
166  // rotational air resistance
167  rot_accel.x -= gyro.x * radians(5000.0) / terminal_rotation_rate;
168  rot_accel.y -= gyro.y * radians(5000.0) / terminal_rotation_rate;
169  rot_accel.z -= gyro.z * radians(400.0) / terminal_rotation_rate;
170 
171  // update rotational rates in body frame
172  gyro += rot_accel * delta_time;
173 
174  // update attitude
175  dcm.rotate(gyro * delta_time);
176  dcm.normalize();
177 
178  // air resistance
179  Vector3f air_resistance = -velocity_ef * (GRAVITY_MSS/terminal_velocity);
180 
181  accel_body = Vector3f(0, 0, -thrust / mass);
182  Vector3f accel_earth = dcm * accel_body;
183 
184  accel_earth += Vector3f(0, 0, GRAVITY_MSS);
185  accel_earth += air_resistance;
186 
187  // if we're on the ground, then our vertical acceleration is limited
188  // to zero. This effectively adds the force of the ground on the aircraft
189  if (on_ground(position) && accel_earth.z > 0) {
190  accel_earth.z = 0;
191  }
192 
193  // work out acceleration as seen by the accelerometers. It sees the kinematic
194  // acceleration (ie. real movement), plus gravity
195  accel_body = dcm.transposed() * (accel_earth + Vector3f(0, 0, -GRAVITY_MSS));
196 
197  // add some noise
198  add_noise(thrust / (thrust_scale * frame->num_motors));
199 
200  // new velocity vector
201  velocity_ef += accel_earth * delta_time;
202 
203  // new position vector
204  Vector3f old_position = position;
205  position += velocity_ef * delta_time;
206 
207  // constrain height to the ground
208  if (on_ground(position)) {
209  if (!on_ground(old_position)) {
210  printf("Hit ground at %f m/s\n", velocity_ef.z);
211 
212  velocity_ef.zero();
213 
214  // zero roll/pitch, but keep yaw
215  float r, p, y;
216  dcm.to_euler(&r, &p, &y);
217  dcm.from_euler(0, 0, y);
218 
219  position.z = -(ground_level + frame_height - home.alt*0.01f);
220  }
221  }
222 
223  // update lat/lon/altitude
224  update_position();
225 }
