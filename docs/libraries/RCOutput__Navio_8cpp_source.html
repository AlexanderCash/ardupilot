102 void LinuxRCOutput_Navio::set_freq(uint32_t chmask, uint16_t freq_hz)
103 {
104 
105  /* Correctly finish last pulses */
106  for (int i = 0; i < PWM_CHAN_COUNT; i++) {
107  write(i, _pulses_buffer[i]);
108  }
109 
110  if (!_i2c_sem->take(10)) {
111  return;
112  }
113 
114  /* Shutdown before sleeping.
115  * see p.14 of PCA9685 product datasheet
116  */
117  hal.i2c->writeRegister(PCA9685_ADDRESS, PCA9685_RA_ALL_LED_OFF_H, PCA9685_ALL_LED_OFF_H_SHUT);
118 
119  /* Put PCA9685 to sleep (required to write prescaler) */
120  hal.i2c->writeRegister(PCA9685_ADDRESS, PCA9685_RA_MODE1, PCA9685_MODE1_SLEEP_BIT);
121 
122  /* Calculate and write prescale value to match frequency */
123  uint8_t prescale = round(24576000.f / 4096.f / freq_hz) - 1;
124  hal.i2c->writeRegister(PCA9685_ADDRESS, PCA9685_RA_PRE_SCALE, prescale);
125 
126  /* Enable external clocking */
127  hal.i2c->writeRegister(PCA9685_ADDRESS, PCA9685_RA_MODE1,
128  PCA9685_MODE1_SLEEP_BIT | PCA9685_MODE1_EXTCLK_BIT);
129 
130  /* Restart the device to apply new settings and enable auto-incremented write */
131  hal.i2c->writeRegister(PCA9685_ADDRESS, PCA9685_RA_MODE1,
132  PCA9685_MODE1_RESTART_BIT | PCA9685_MODE1_AI_BIT);
133  _frequency = freq_hz;
134 
135  _i2c_sem->give();
136 }
