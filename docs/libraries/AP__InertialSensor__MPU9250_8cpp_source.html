205 /*
206  initialise the sensor
207  */
208 bool AP_InertialSensor_MPU9250::_init_sensor(void)
209 {
210  _spi = hal.spi->device(AP_HAL::SPIDevice_MPU9250);
211  _spi_sem = _spi->get_semaphore();
212 
213  // we need to suspend timers to prevent other SPI drivers grabbing
214  // the bus while we do the long initialisation
215  hal.scheduler->suspend_timer_procs();
216 
217  uint8_t whoami = _register_read(MPUREG_WHOAMI);
218  if (whoami != MPUREG_WHOAMI_MPU9250 && whoami != MPUREG_WHOAMI_MPU9255) {
219  hal.console->printf("MPU9250: unexpected WHOAMI 0x%x\n", (unsigned)whoami);
220  return false;
221  }
222 
223  uint8_t tries = 0;
224  do {
225  bool success = _hardware_init();
226  if (success) {
227  hal.scheduler->delay(10);
228  if (!_spi_sem->take(100)) {
229  hal.console->printf("MPU9250: Unable to get semaphore");
230  return false;
231  }
232  uint8_t status = _register_read(MPUREG_INT_STATUS);
233  if ((status & BIT_RAW_RDY_INT) != 0) {
234  _spi_sem->give();
235  break;
236  }
237  _spi_sem->give();
238  }
239  if (tries++ > 5) {
240  return false;
241  }
242  } while (1);
243 
244  hal.scheduler->resume_timer_procs();
245 
246  _gyro_instance = _imu.register_gyro();
247  _accel_instance = _imu.register_accel();
248 
249  _product_id = AP_PRODUCT_ID_MPU9250;
250 
251  // start the timer process to read samples
252  hal.scheduler->register_timer_process(AP_HAL_MEMBERPROC(&AP_InertialSensor_MPU9250::_poll_data));
253 
254 #if MPU9250_DEBUG
255  _dump_registers();
256 #endif
257  return true;
258 }

260 /*
261  update the accel and gyro vectors
262  */
263 bool AP_InertialSensor_MPU9250::update( void )
264 {
265  // pull the data from the timer shared data buffer
266  uint8_t idx = _shared_data_idx;
267  Vector3f gyro = _shared_data[idx]._gyro_filtered;
268  Vector3f accel = _shared_data[idx]._accel_filtered;
269 
270  _have_sample_available = false;
271 
272  accel *= MPU9250_ACCEL_SCALE_1G;
273  gyro *= GYRO_SCALE;
274 
275  // rotate for bbone default
276  accel.rotate(ROTATION_ROLL_180_YAW_90);
277  gyro.rotate(ROTATION_ROLL_180_YAW_90);
278 
279 #if CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_LINUX_PXF
280  // PXF has an additional YAW 180
281  accel.rotate(ROTATION_YAW_180);
282  gyro.rotate(ROTATION_YAW_180);
283 #elif CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_LINUX_NAVIO
284  // NavIO has different orientation, assuming RaspberryPi is right
285  // way up, and PWM pins on NavIO are at the back of the aircraft
286  accel.rotate(ROTATION_ROLL_180_YAW_90);
287  gyro.rotate(ROTATION_ROLL_180_YAW_90);
288 #elif CONFIG_HAL_BOARD_SUBTYPE == HAL_BOARD_SUBTYPE_LINUX_BBBMINI
289  accel.rotate(ROTATION_ROLL_180);
290  gyro.rotate(ROTATION_ROLL_180);
291 #endif
292 
293  _publish_gyro(_gyro_instance, gyro);
294  _publish_accel(_accel_instance, accel);
295 
296  if (_last_accel_filter_hz != _accel_filter_cutoff()) {
297  _set_accel_filter(_accel_filter_cutoff());
298  _last_accel_filter_hz = _accel_filter_cutoff();
299  }
300 
301  if (_last_gyro_filter_hz != _gyro_filter_cutoff()) {
302  _set_gyro_filter(_gyro_filter_cutoff());
303  _last_gyro_filter_hz = _gyro_filter_cutoff();
304  }
305 
306  return true;
307 }

314 void AP_InertialSensor_MPU9250::_poll_data(void)
315 {
316  if (!_spi_sem->take_nonblocking()) {
317  /*
318  the semaphore being busy is an expected condition when the
319  mainline code is calling wait_for_sample() which will
320  grab the semaphore. We return now and rely on the mainline
321  code grabbing the latest sample.
322  */
323  return;
324  }
325  _read_data_transaction();
326  _spi_sem->give();
327 }

330 /*
331  read from the data registers and update filtered data
332  */
333 void AP_InertialSensor_MPU9250::_read_data_transaction()
334 {
335  /* one resister address followed by seven 2-byte registers */
336  struct PACKED {
337  uint8_t cmd;
338  uint8_t int_status;
339  uint8_t v[14];
340  } rx, tx = { cmd : MPUREG_INT_STATUS | 0x80, };
341 
342  _spi->transaction((const uint8_t *)&tx, (uint8_t *)&rx, sizeof(rx));
343 
344 #define int16_val(v, idx) ((int16_t)(((uint16_t)v[2*idx] << 8) | v[2*idx+1]))
345 
346  Vector3f _accel_filtered = _accel_filter.apply(Vector3f(int16_val(rx.v, 1),
347  int16_val(rx.v, 0),
348  -int16_val(rx.v, 2)));
349 
350  Vector3f _gyro_filtered = _gyro_filter.apply(Vector3f(int16_val(rx.v, 5),
351  int16_val(rx.v, 4),
352  -int16_val(rx.v, 6)));
353  // update the shared buffer
354  uint8_t idx = _shared_data_idx ^ 1;
355  _shared_data[idx]._accel_filtered = _accel_filtered;
356  _shared_data[idx]._gyro_filtered = _gyro_filtered;
357  _shared_data_idx = idx;
358 
359  _have_sample_available = true;
360 }

408 /*
409  initialise the sensor configuration registers
410  */
411 bool AP_InertialSensor_MPU9250::_hardware_init(void)
412 {
413  if (!_spi_sem->take(100)) {
414  hal.console->printf("MPU9250: Unable to get semaphore");
415  return false;
416  }
417 
418  // initially run the bus at low speed
419  _spi->set_bus_speed(AP_HAL::SPIDeviceDriver::SPI_SPEED_LOW);
420 
421  // Chip reset
422  uint8_t tries;
423  for (tries = 0; tries<5; tries++) {
424  hal.scheduler->delay(100);
425 
426  // Wake up device and select GyroZ clock. Note that the
427  // MPU6000 starts up in sleep mode, and it can take some time
428  // for it to come out of sleep
429  _register_write(MPUREG_PWR_MGMT_1, BIT_PWR_MGMT_1_CLK_ZGYRO);
430  hal.scheduler->delay(5);
431 
432  // check it has woken up
433  if (_register_read(MPUREG_PWR_MGMT_1) == BIT_PWR_MGMT_1_CLK_ZGYRO) {
434  break;
435  }
436 #if MPU9250_DEBUG
437  _dump_registers();
438 #endif
439  }
440  if (tries == 5) {
441  hal.console->println_P(PSTR("Failed to boot MPU9250 5 times"));
442  _spi_sem->give();
443  return false;
444  }
445 
446  _register_write(MPUREG_PWR_MGMT_2, 0x00); // only used for wake-up in accelerometer only low power mode
447 
448  // used no filter of 256Hz on the sensor, then filter using
449  // the 2-pole software filter
450  _register_write(MPUREG_CONFIG, BITS_DLPF_CFG_256HZ_NOLPF2);
451 
452  // set sample rate to 1kHz, and use the 2 pole filter to give the
453  // desired rate
454  _register_write(MPUREG_SMPLRT_DIV, MPUREG_SMPLRT_1000HZ);
455  _register_write(MPUREG_GYRO_CONFIG, BITS_GYRO_FS_2000DPS); // Gyro scale 2000º/s
456 
457  // RM-MPU-9250A-00.pdf, pg. 15, select accel full scale 16g
458  _register_write(MPUREG_ACCEL_CONFIG,3<<3);
459 
460  // configure interrupt to fire when new data arrives
461  _register_write(MPUREG_INT_ENABLE, BIT_RAW_RDY_EN);
462 
463  // clear interrupt on any read, and hold the data ready pin high
464  // until we clear the interrupt
465  _register_write(MPUREG_INT_PIN_CFG, BIT_INT_RD_CLEAR | BIT_LATCH_INT_EN);
466 
467  // now that we have initialised, we set the SPI bus speed to high
468  // (8MHz on APM2)
469  _spi->set_bus_speed(AP_HAL::SPIDeviceDriver::SPI_SPEED_HIGH);
470 
471  _spi_sem->give();
472 
473  return true;
474 }
