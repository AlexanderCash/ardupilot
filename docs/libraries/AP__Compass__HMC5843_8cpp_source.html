111 // Read Sensor data
112 bool AP_Compass_HMC5843::read_raw()
113 {
114  uint8_t buff[6];
115 
116  if (hal.i2c->readRegisters(COMPASS_ADDRESS, 0x03, 6, buff) != 0) {
117  hal.i2c->setHighSpeed(false);
118  _retry_time = hal.scheduler->millis() + 1000;
119  _i2c_sem->give();
120  return false;
121  }
122 
123  int16_t rx, ry, rz;
124  rx = (((int16_t)buff[0]) << 8) | buff[1];
125  if (_product_id == AP_COMPASS_TYPE_HMC5883L) {
126  rz = (((int16_t)buff[2]) << 8) | buff[3];
127  ry = (((int16_t)buff[4]) << 8) | buff[5];
128  } else {
129  ry = (((int16_t)buff[2]) << 8) | buff[3];
130  rz = (((int16_t)buff[4]) << 8) | buff[5];
131  }
132  if (rx == -4096 || ry == -4096 || rz == -4096) {
133  // no valid data available
134  return false;
135  }
136 
137  _mag_x = -rx;
138  _mag_y = ry;
139  _mag_z = -rz;
140 
141  return true;
142 }

145 // accumulate a reading from the magnetometer
146 void AP_Compass_HMC5843::accumulate(void)
147 {
148  if (!_initialised) {
149  // someone has tried to enable a compass for the first time
150  // mid-flight .... we can't do that yet (especially as we won't
151  // have the right orientation!)
152  return;
153  }
154  uint32_t tnow = hal.scheduler->micros();
155  if (_accum_count != 0 && (tnow - _last_accum_time) < 13333) {
156  // the compass gets new data at 75Hz
157  return;
158  }
159 
160  if (!_i2c_sem->take(1)) {
161  // the bus is busy - try again later
162  return;
163  }
164  bool result = read_raw();
165  _i2c_sem->give();
166 
167  if (result) {
168  // the _mag_N values are in the range -2048 to 2047, so we can
169  // accumulate up to 15 of them in an int16_t. Let's make it 14
170  // for ease of calculation. We expect to do reads at 10Hz, and
171  // we get new data at most 75Hz, so we don't expect to
172  // accumulate more than 8 before a read
173  _mag_x_accum += _mag_x;
174  _mag_y_accum += _mag_y;
175  _mag_z_accum += _mag_z;
176  _accum_count++;
177  if (_accum_count == 14) {
178  _mag_x_accum /= 2;
179  _mag_y_accum /= 2;
180  _mag_z_accum /= 2;
181  _accum_count = 7;
182  }
183  _last_accum_time = tnow;
184  }
185 }

202 bool
203 AP_Compass_HMC5843::init()
204 {
205  int numAttempts = 0, good_count = 0;
206  bool success = false;
207  uint8_t calibration_gain = 0x20;
208  uint16_t expected_x = 715;
209  uint16_t expected_yz = 715;
210  float gain_multiple = 1.0;
211 
212  hal.scheduler->suspend_timer_procs();
213  hal.scheduler->delay(10);
214 
215  _i2c_sem = hal.i2c->get_semaphore();
216  if (!_i2c_sem->take(HAL_SEMAPHORE_BLOCK_FOREVER)) {
217  hal.scheduler->panic(PSTR("Failed to get HMC5843 semaphore"));
218  }
219 
220  // determine if we are using 5843 or 5883L
221  _base_config = 0;
222  if (!write_register(ConfigRegA, SampleAveraging_8<<5 | DataOutputRate_75HZ<<2 | NormalOperation) ||
223  !read_register(ConfigRegA, &_base_config)) {
224  _i2c_sem->give();
225  hal.scheduler->resume_timer_procs();
226  return false;
227  }
228  if ( _base_config == (SampleAveraging_8<<5 | DataOutputRate_75HZ<<2 | NormalOperation)) {
229  // a 5883L supports the sample averaging config
230  _product_id = AP_COMPASS_TYPE_HMC5883L;
231  calibration_gain = 0x60;
232  /*
233  note that the HMC5883 datasheet gives the x and y expected
234  values as 766 and the z as 713. Experiments have shown the x
235  axis is around 766, and the y and z closer to 713.
236  */
237  expected_x = 766;
238  expected_yz = 713;
239  gain_multiple = 660.0f / 1090; // adjustment for runtime vs calibration gain
240  } else if (_base_config == (NormalOperation | DataOutputRate_75HZ<<2)) {
241  _product_id = AP_COMPASS_TYPE_HMC5843;
242  } else {
243  // not behaving like either supported compass type
244  _i2c_sem->give();
245  hal.scheduler->resume_timer_procs();
246  return false;
247  }
248 
249  calibration[0] = 0;
250  calibration[1] = 0;
251  calibration[2] = 0;
252 
253  while ( success == 0 && numAttempts < 25 && good_count < 5)
254  {
255  // record number of attempts at initialisation
256  numAttempts++;
257 
258  // force positiveBias (compass should return 715 for all channels)
259  if (!write_register(ConfigRegA, PositiveBiasConfig))
260  continue; // compass not responding on the bus
261  hal.scheduler->delay(50);
262 
263  // set gains
264  if (!write_register(ConfigRegB, calibration_gain) ||
265  !write_register(ModeRegister, SingleConversion))
266  continue;
267 
268  // read values from the compass
269  hal.scheduler->delay(50);
270  if (!read_raw())
271  continue; // we didn't read valid values
272 
273  hal.scheduler->delay(10);
274 
275  float cal[3];
276 
277  // hal.console->printf_P(PSTR("mag %d %d %d\n"), _mag_x, _mag_y, _mag_z);
278  cal[0] = fabsf(expected_x / (float)_mag_x);
279  cal[1] = fabsf(expected_yz / (float)_mag_y);
280  cal[2] = fabsf(expected_yz / (float)_mag_z);
281 
282  // hal.console->printf_P(PSTR("cal=%.2f %.2f %.2f\n"), cal[0], cal[1], cal[2]);
283 
284  // we throw away the first two samples as the compass may
285  // still be changing its state from the application of the
286  // strap excitation. After that we accept values in a
287  // reasonable range
288  if (numAttempts > 2 &&
289  cal[0] > 0.7f && cal[0] < 1.35f &&
290  cal[1] > 0.7f && cal[1] < 1.35f &&
291  cal[2] > 0.7f && cal[2] < 1.35f) {
292  // hal.console->printf_P(PSTR("cal=%.2f %.2f %.2f good\n"), cal[0], cal[1], cal[2]);
293  good_count++;
294  calibration[0] += cal[0];
295  calibration[1] += cal[1];
296  calibration[2] += cal[2];
297  }
298 
299 #if 0
300  /* useful for debugging */
301  hal.console->printf_P(PSTR("MagX: %d MagY: %d MagZ: %d\n"), (int)_mag_x, (int)_mag_y, (int)_mag_z);
302  hal.console->printf_P(PSTR("CalX: %.2f CalY: %.2f CalZ: %.2f\n"), cal[0], cal[1], cal[2]);
303 #endif
304  }
305 
306  if (good_count >= 5) {
307  /*
308  The use of gain_multiple below is incorrect, as the gain
309  difference between 2.5Ga mode and 1Ga mode is already taken
310  into account by the expected_x and expected_yz values. We
311  are not going to fix it however as it would mean all
312  APM1/APM2 users redoing their compass calibration. The
313  impact is that the values we report on APM1/APM2 are lower
314  than they should be (by a multiple of about 0.6). This
315  doesn't have any impact other than the learned compass
316  offsets
317  */
318  calibration[0] = calibration[0] * gain_multiple / good_count;
319  calibration[1] = calibration[1] * gain_multiple / good_count;
320  calibration[2] = calibration[2] * gain_multiple / good_count;
321  success = true;
322  } else {
323  /* best guess */
324  calibration[0] = 1.0;
325  calibration[1] = 1.0;
326  calibration[2] = 1.0;
327  }
328 
329  // leave test mode
330  if (!re_initialise()) {
331  _i2c_sem->give();
332  hal.scheduler->resume_timer_procs();
333  return false;
334  }
335 
336  _i2c_sem->give();
337  hal.scheduler->resume_timer_procs();
338  _initialised = true;
339 
340  // perform an initial read
341  read();
342 
343 #if 0
344  hal.console->printf_P(PSTR("CalX: %.2f CalY: %.2f CalZ: %.2f\n"),
345  calibration[0], calibration[1], calibration[2]);
346 #endif
347 
348  if (success) {
349  // register the compass instance in the frontend
350  _compass_instance = register_compass();
351  }
352 
353  return success;
354 }

356 // Read Sensor data
357 void AP_Compass_HMC5843::read()
358 {
359  if (!_initialised) {
360  // someone has tried to enable a compass for the first time
361  // mid-flight .... we can't do that yet (especially as we won't
362  // have the right orientation!)
363  return;
364  }
365  if (_retry_time != 0) {
366  if (hal.scheduler->millis() < _retry_time) {
367  return;
368  }
369  if (!re_initialise()) {
370  _retry_time = hal.scheduler->millis() + 1000;
371  hal.i2c->setHighSpeed(false);
372  return;
373  }
374  }
375 
376  if (_accum_count == 0) {
377  accumulate();
378  if (_retry_time != 0) {
379  hal.i2c->setHighSpeed(false);
380  return;
381  }
382  }
383 
384  Vector3f field(_mag_x_accum * calibration[0],
385  _mag_y_accum * calibration[1],
386  _mag_z_accum * calibration[2]);
387  field /= _accum_count;
388 
389  _accum_count = 0;
390  _mag_x_accum = _mag_y_accum = _mag_z_accum = 0;
391 
392  // rotate to the desired orientation
393  if (_product_id == AP_COMPASS_TYPE_HMC5883L) {
394  field.rotate(ROTATION_YAW_90);
395  }
396 
397  publish_field(field, _compass_instance);
398  _retry_time = 0;
399 }
