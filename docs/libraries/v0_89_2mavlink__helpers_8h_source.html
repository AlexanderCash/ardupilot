395 MAVLINK_HELPER uint8_t put_bitfield_n_by_index(int32_t b, uint8_t bits, uint8_t packet_index, uint8_t bit_index, uint8_t* r_bit_index, uint8_t* buffer)
396 {
397  uint16_t bits_remain = bits;
398  // Transform number into network order
399  int32_t v;
400  uint8_t i_bit_index, i_byte_index, curr_bits_n;
401 #if MAVLINK_NEED_BYTE_SWAP
402  union {
403  int32_t i;
404  uint8_t b[4];
405  } bin, bout;
406  bin.i = b;
407  bout.b[0] = bin.b[3];
408  bout.b[1] = bin.b[2];
409  bout.b[2] = bin.b[1];
410  bout.b[3] = bin.b[0];
411  v = bout.i;
412 #else
413  v = b;
414 #endif
415 
416  // buffer in
417  // 01100000 01000000 00000000 11110001
418  // buffer out
419  // 11110001 00000000 01000000 01100000
420 
421  // Existing partly filled byte (four free slots)
422  // 0111xxxx
423 
424  // Mask n free bits
425  // 00001111 = 2^0 + 2^1 + 2^2 + 2^3 = 2^n - 1
426  // = ((uint32_t)(1 << n)) - 1; // = 2^n - 1
427 
428  // Shift n bits into the right position
429  // out = in >> n;
430 
431  // Mask and shift bytes
432  i_bit_index = bit_index;
433  i_byte_index = packet_index;
434  if (bit_index > 0)
435  {
436  // If bits were available at start, they were available
437  // in the byte before the current index
438  i_byte_index--;
439  }
440 
441  // While bits have not been packed yet
442  while (bits_remain > 0)
443  {
444  // Bits still have to be packed
445  // there can be more than 8 bits, so
446  // we might have to pack them into more than one byte
447 
448  // First pack everything we can into the current 'open' byte
449  //curr_bits_n = bits_remain << 3; // Equals bits_remain mod 8
450  //FIXME
451  if (bits_remain <= (uint8_t)(8 - i_bit_index))
452  {
453  // Enough space
454  curr_bits_n = (uint8_t)bits_remain;
455  }
456  else
457  {
458  curr_bits_n = (8 - i_bit_index);
459  }
460 
461  // Pack these n bits into the current byte
462  // Mask out whatever was at that position with ones (xxx11111)
463  buffer[i_byte_index] &= (0xFF >> (8 - curr_bits_n));
464  // Put content to this position, by masking out the non-used part
465  buffer[i_byte_index] |= ((0x00 << curr_bits_n) & v);
466 
467  // Increment the bit index
468  i_bit_index += curr_bits_n;
469 
470  // Now proceed to the next byte, if necessary
471  bits_remain -= curr_bits_n;
472  if (bits_remain > 0)
473  {
474  // Offer another 8 bits / one byte
475  i_byte_index++;
476  i_bit_index = 0;
477  }
478  }
479 
480  *r_bit_index = i_bit_index;
481  // If a partly filled byte is present, mark this as consumed
482  if (i_bit_index != 7) i_byte_index++;
483  return i_byte_index - packet_index;
484 }
