235  switch (size) {

238  static uint32_t count;

252  // garbage input
253  return;

255

258  // prevent bad inputs from SIM from corrupting our state
259  double *v = &_sitl->state.latitude;
260  for (uint8_t i=0; i<17; i++) {
261  if (isinf(v[i]) || isnan(v[i]) || fabs(v[i]) > 1.0e10) {
262  v[i] = 0;
263  }
264  }
265  }

267 
268  count++;
269  if (hal.scheduler->millis() - last_report > 1000) {

271  count = 0;
272  last_report = hal.scheduler->millis();
273  }

275

277  // a packet giving the receiver PWM inputs
278  uint8_t i;

324  apply servo rate filtering
325  This allows simulation of servo lag
326  */
327 void SITL_State::_apply_servo_filter(float deltat)
328 {
329  if (_sitl->servo_rate < 1.0f) {
330  // no limit
331  return;
332  }
333  // 1000 usec == 90 degrees
334  uint16_t max_change = deltat * _sitl->servo_rate * 1000 / 90;
335  if (max_change == 0) {
336  max_change = 1;
337  }
338  for (uint8_t i=0; i<11; i++) {
339  int16_t change = (int16_t)pwm_output[i] - (int16_t)last_pwm_output[i];
340  if (change > max_change) {
341  pwm_output[i] = last_pwm_output[i] + max_change;
342  } else if (change < -max_change) {
343  pwm_output[i] = last_pwm_output[i] - max_change;
344  }
345  }
346 }

350  create sitl_input structure for sending to FDM

352 void SITL_State::_simulator_servos(Aircraft::sitl_input &input)
353 {
354  static uint32_t last_update_usec;
355 
356  /* this maps the registers used for PWM outputs. The RC
357  * driver updates these whenever it wants the channel output
358  * to change */
359  uint8_t i;
360 
361  if (last_update_usec == 0) {
362  for (i=0; i<11; i++) {
363  pwm_output[i] = 1000;
364  }
365  if (_vehicle == ArduPlane) {
366  pwm_output[0] = pwm_output[1] = pwm_output[3] = 1500;
367  pwm_output[7] = 1800;
368  }
369  if (_vehicle == APMrover2) {
370  pwm_output[0] = pwm_output[1] = pwm_output[2] = pwm_output[3] = 1500;
371  pwm_output[7] = 1800;
372  }
373  for (i=0; i<11; i++) {
374  last_pwm_output[i] = pwm_output[i];
375  }
376  }
377 
378  // output at chosen framerate
379  uint32_t now = hal.scheduler->micros();
380  float deltat = (now - last_update_usec) * 1.0e-6f;
381  last_update_usec = now;
382 
383  _apply_servo_filter(deltat);
384 
385  for (i=0; i<11; i++) {
386  if (pwm_output[i] == 0xFFFF) {
387  input.servos[i] = 0;
388  } else {
389  input.servos[i] = pwm_output[i];
390  }
391  last_pwm_output[i] = pwm_output[i];
392  }
393 
394  if (_vehicle == ArduPlane) {
395  // add in engine multiplier
396  if (input.servos[2] > 1000) {
397  input.servos[2] = ((input.servos[2]-1000) * _sitl->engine_mul) + 1000;
398  if (input.servos[2] > 2000) input.servos[2] = 2000;
399  }
400  _motors_on = ((input.servos[2]-1000)/1000.0f) > 0;
401  } else if (_vehicle == APMrover2) {
402  // add in engine multiplier
403  if (input.servos[2] != 1500) {
404  input.servos[2] = ((input.servos[2]-1500) * _sitl->engine_mul) + 1500;
405  if (input.servos[2] > 2000) input.servos[2] = 2000;
406  if (input.servos[2] < 1000) input.servos[2] = 1000;
407  }
408  _motors_on = ((input.servos[2]-1500)/500.0f) != 0;
409  } else {
410  _motors_on = false;
411  // apply engine multiplier to first motor
412  input.servos[0] = ((input.servos[0]-1000) * _sitl->engine_mul) + 1000;
413  // run checks on each motor
414  for (i=0; i<4; i++) {
415  // check motors do not exceed their limits
416  if (input.servos[i] > 2000) input.servos[i] = 2000;
417  if (input.servos[i] < 1000) input.servos[i] = 1000;
418  // update motor_on flag
419  if ((input.servos[i]-1000)/1000.0f > 0) {
420  _motors_on = true;
421  }
422  }
423  }
424 
425  float throttle = _motors_on?(input.servos[2]-1000) / 1000.0f:0;
426  // lose 0.7V at full throttle
427  float voltage = _sitl->batt_voltage - 0.7f*throttle;
428 
429  // assume 50A at full throttle
430  _current = 50.0f * throttle;
431 
432  // assume 3DR power brick
433  voltage_pin_value = ((voltage / 10.1f) / 5.0f) * 1024;
434  current_pin_value = ((_current / 17.0f) / 5.0f) * 1024;
435 }
