52 /*
53  return turning circle (diameter) in meters for steering angle proportion in degrees
54 */
55 float Rover::turn_circle(float steering)
56 {
57  if (fabsf(steering) < 1.0e-6) {
58  return 0;
59  }
60  return turning_circle * sinf(radians(35)) / sinf(radians(steering*35));
61 }

63 /*
64  return yaw rate in degrees/second given steering_angle and speed
65 */
66 float Rover::calc_yaw_rate(float steering, float speed)
67 {
68  if (skid_steering) {
69  return steering * skid_turn_rate;
70  }
71  if (fabsf(steering) < 1.0e-6 or fabsf(speed) < 1.0e-6) {
72  return 0;
73  }
74  float d = turn_circle(steering);
75  float c = M_PI_F * d;
76  float t = c / speed;
77  float rate = 360.0f / t;
78  return rate;
79 }

91 /*
92  update the rover simulation by one time step
93  */
94 void Rover::update(const struct sitl_input &input)
95 {
96  float steering, throttle;
97 
98  // if in skid steering mode the steering and throttle values are used for motor1 and motor2
99  if (skid_steering) {
100  float motor1 = 2*((input.servos[0]-1000)/1000.0f - 0.5f);
101  float motor2 = 2*((input.servos[2]-1000)/1000.0f - 0.5f);
102  steering = motor1 - motor2;
103  throttle = 0.5*(motor1 + motor2);
104  } else {
105  steering = 2*((input.servos[0]-1000)/1000.0f - 0.5f);
106  throttle = 2*((input.servos[2]-1000)/1000.0f - 0.5f);
107  }
108 
109  // how much time has passed?
110  float delta_time = frame_time_us * 1.0e-6f;
111 
112  // speed in m/s in body frame
113  Vector3f velocity_body = dcm.transposed() * velocity_ef;
114 
115  // speed along x axis, +ve is forward
116  float speed = velocity_body.x;
117 
118  // yaw rate in degrees/s
119  float yaw_rate = calc_yaw_rate(steering, speed);
120 
121  // target speed with current throttle
122  float target_speed = throttle * max_speed;
123 
124  // linear acceleration in m/s/s - very crude model
125  float accel = max_accel * (target_speed - speed) / max_speed;
126 
127  gyro = Vector3f(0,0,radians(yaw_rate));
128 
129  // update attitude
130  dcm.rotate(gyro * delta_time);
131  dcm.normalize();
132 
133  // accel in body frame due to motor
134  accel_body = Vector3f(accel, 0, 0);
135 
136  // add in accel due to direction change
137  accel_body.y += radians(yaw_rate) * speed;
138 
139  // now in earth frame
140  Vector3f accel_earth = dcm * accel_body;
141  accel_earth += Vector3f(0, 0, GRAVITY_MSS);
142 
143  // we are on the ground, so our vertical accel is zero
144  accel_earth.z = 0;
145 
146  // work out acceleration as seen by the accelerometers. It sees the kinematic
147  // acceleration (ie. real movement), plus gravity
148  accel_body = dcm.transposed() * (accel_earth + Vector3f(0, 0, -GRAVITY_MSS));
149 
150  // new velocity vector
151  velocity_ef += accel_earth * delta_time;
152 
153  // new position vector
154  position += velocity_ef * delta_time;
155  position.z = -home.alt*0.01f;
156 
157  // update lat/lon/altitude
158  update_position();
159 }
