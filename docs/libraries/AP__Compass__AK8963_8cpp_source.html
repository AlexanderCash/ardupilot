271 bool AP_Compass_AK8963_MPU9250::read_raw()
272 {
273  uint8_t rx[14] = {0};
274 
275  const uint8_t count = 9;
276  _backend->read(MPUREG_EXT_SENS_DATA_00, rx, count);
277 
278  uint8_t st2 = rx[8]; /* End data read by reading ST2 register */
279 
280 #define int16_val(v, idx) ((int16_t)(((uint16_t)v[2*idx + 1] << 8) | v[2*idx]))
281 
282  if(!(st2 & 0x08)) {
283  _mag_x = (float) int16_val(rx, 1);
284  _mag_y = (float) int16_val(rx, 2);
285  _mag_z = (float) int16_val(rx, 3);
286 
287  if (is_zero(_mag_x) && is_zero(_mag_y) && is_zero(_mag_z)) {
288  return false;
289  }
290 
291  return true;
292  } else {
293  return false;
294  }

319 bool AP_Compass_AK8963::_self_test()
320 {
321  bool success = false;
322 
323  /* see AK8963.pdf p.19 */
324 
325  /* Set power-down mode */
326  _register_write(AK8963_CNTL1, AK8963_POWERDOWN_MODE | _magnetometer_adc_resolution);
327 
328  /* Turn the internal magnetic field on */
329  _register_write(AK8963_ASTC, AK8983_SELFTEST_MAGNETIC_FIELD_ON);
330 
331  /* Register value to self-test mode in 14-bit */
332  _register_write(AK8963_CNTL1, AK8963_SELFTEST_MODE | _magnetometer_adc_resolution);
333 
334  _start_conversion();
335  hal.scheduler->delay(20);
336  read_raw();
337 
338  float hx = _mag_x;
339  float hy = _mag_y;
340  float hz = _mag_z;
341 
342  error("AK8963's SELF-TEST STARTED\n");
343 
344  switch (_magnetometer_adc_resolution) {
345  bool hx_is_in_range;
346  bool hy_is_in_range;
347  bool hz_is_in_range;
348  case AK8963_14BIT_ADC:
349  hx_is_in_range = (hx >= - 50) && (hx <= 50);
350  hy_is_in_range = (hy >= - 50) && (hy <= 50);
351  hz_is_in_range = (hz >= - 800) && (hz <= -200);
352  if (hx_is_in_range && hy_is_in_range && hz_is_in_range) {
353  success = true;
354  }
355  break;
356  case AK8963_16BIT_ADC:
357  hx_is_in_range = (hx >= -200) && (hx <= 200);
358  hy_is_in_range = (hy >= -200) && (hy <= 200);
359  hz_is_in_range = (hz >= -3200) && (hz <= -800);
360  if (hx_is_in_range && hy_is_in_range && hz_is_in_range) {
361  success = true;
362  }
363  break;
364  default:
365  success = false;
366  hal.scheduler->panic(PSTR("Wrong AK8963's ADC resolution selected"));
367  break;
368  }
369 
370  error("AK8963's SELF-TEST ENDED: %f %f %f\n", hx, hy, hz);
371 
372  /* Turn the internal magnetic field off */
373  _register_write(AK8963_ASTC, 0x0);
374 
375  /* Register value to continuous measurement in 14-bit */
376  _register_write(AK8963_CNTL1, AK8963_POWERDOWN_MODE | _magnetometer_adc_resolution);
377 
378  return success;
379 }

381 bool AP_Compass_AK8963::init()
382 {
383  hal.scheduler->suspend_timer_procs();
384  if (!_backend->sem_take_blocking()) {
385  error("_spi_sem->take failed\n");
386  return false;
387  }
388 
389 
390  if (!_backend_init()) {
391  _backend->sem_give();
392  return false;
393  }
394 
395  _register_write(AK8963_CNTL2, AK8963_RESET); /* Reset AK8963 */
396 
397  hal.scheduler->delay(1000);
398 
399  int id_mismatch_count;
400  uint8_t deviceid;
401  for (id_mismatch_count = 0; id_mismatch_count < 5; id_mismatch_count++) {
402  _register_read(AK8963_WIA, 0x01, &deviceid); /* Read AK8963's id */
403 
404  if (deviceid == AK8963_Device_ID) {
405  break;
406  }
407 
408  error("trying to read AK8963's ID once more...\n");
409  _backend_reset();
410  hal.scheduler->delay(100);
411  _dump_registers();
412  }
413 
414  if (id_mismatch_count == 5) {
415  _initialised = false;
416  hal.console->printf("WRONG AK8963 DEVICE ID: 0x%x\n", (unsigned)deviceid);
417  hal.scheduler->panic(PSTR("AK8963: bad DEVICE ID"));
418  }
419 
420  _calibrate();
421 
422  _initialised = true;
423 
424 #if AK8963_SELFTEST
425  if (_self_test()) {
426  _initialised = true;
427  } else {
428  _initialised = false;
429  }
430 #endif
431 
432  /* Register value to continuous measurement */
433  _register_write(AK8963_CNTL1, AK8963_CONTINUOUS_MODE2 | _magnetometer_adc_resolution);
434 
435  _backend->sem_give();
436 
437  // register the compass instance in the frontend
438  _compass_instance = register_compass();
439 
440  hal.scheduler->resume_timer_procs();
441  hal.scheduler->register_timer_process( AP_HAL_MEMBERPROC(&AP_Compass_AK8963::_update));
442 
443  _start_conversion();
444 
445  _initialised = true;
446  return _initialised;
447 }
