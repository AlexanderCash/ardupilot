8 /*
9  get a speed scaling number for control surfaces. This is applied to
10  PIDs to change the scaling of the PID with speed. At high speed we
11  move the surfaces less, and at low speeds we move them more.
12  */
13 static float get_speed_scaler(void)
14 {
15  float aspeed, speed_scaler;
16  if (ahrs.airspeed_estimate(&aspeed)) {
17  if (aspeed > auto_state.highest_airspeed) {
18  auto_state.highest_airspeed = aspeed;
19  }
20  if (aspeed > 0) {
21  speed_scaler = g.scaling_speed / aspeed;
22  } else {
23  speed_scaler = 2.0;
24  }
25  speed_scaler = constrain_float(speed_scaler, 0.5f, 2.0f);
26  } else {
27  if (channel_throttle->servo_out > 0) {
28  speed_scaler = 0.5f + ((float)THROTTLE_CRUISE / channel_throttle->servo_out / 2.0f); // First order taylor expansion of square root
29  // Should maybe be to the 2/7 power, but we aren't goint to implement that...
30  }else{
31  speed_scaler = 1.67f;
32  }
33  // This case is constrained tighter as we don't have real speed info
34  speed_scaler = constrain_float(speed_scaler, 0.6f, 1.67f);
35  }
36  return speed_scaler;
37 }

67 /*
68  this is the main roll stabilization function. It takes the
69  previously set nav_roll calculates roll servo_out to try to
70  stabilize the plane at the given roll
71  */
72 static void stabilize_roll(float speed_scaler)
73 {
74  if (fly_inverted()) {
75  // we want to fly upside down. We need to cope with wrap of
76  // the roll_sensor interfering with wrap of nav_roll, which
77  // would really confuse the PID code. The easiest way to
78  // handle this is to ensure both go in the same direction from
79  // zero
80  nav_roll_cd += 18000;
81  if (ahrs.roll_sensor < 0) nav_roll_cd -= 36000;
82  }
83 
84  bool disable_integrator = false;
85  if (control_mode == STABILIZE && channel_roll->control_in != 0) {
86  disable_integrator = true;
87  }
88  channel_roll->servo_out = rollController.get_servo_out(nav_roll_cd - ahrs.roll_sensor,
89  speed_scaler,
90  disable_integrator);
91 }

93 /*
94  this is the main pitch stabilization function. It takes the
95  previously set nav_pitch and calculates servo_out values to try to
96  stabilize the plane at the given attitude.
97  */
98 static void stabilize_pitch(float speed_scaler)
99 {
100  int8_t force_elevator = takeoff_tail_hold();
101  if (force_elevator != 0) {
102  // we are holding the tail down during takeoff. Just covert
103  // from a percentage to a -4500..4500 centidegree angle
104  channel_pitch->servo_out = 45*force_elevator;
105  return;
106  }
107  int32_t demanded_pitch = nav_pitch_cd + g.pitch_trim_cd + channel_throttle->servo_out * g.kff_throttle_to_pitch;
108  bool disable_integrator = false;
109  if (control_mode == STABILIZE && channel_pitch->control_in != 0) {
110  disable_integrator = true;
111  }
112  channel_pitch->servo_out = pitchController.get_servo_out(demanded_pitch - ahrs.pitch_sensor,
113  speed_scaler,
114  disable_integrator);
115 }

117 /*
118  perform stick mixing on one channel
119  This type of stick mixing reduces the influence of the auto
120  controller as it increases the influence of the users stick input,
121  allowing the user full deflection if needed
122  */
123 static void stick_mix_channel(RC_Channel *channel, int16_t &servo_out)
124 {
125  float ch_inf;
126 
127  ch_inf = (float)channel->radio_in - (float)channel->radio_trim;
128  ch_inf = fabsf(ch_inf);
129  ch_inf = min(ch_inf, 400.0f);
130  ch_inf = ((400.0f - ch_inf) / 400.0f);
131  servo_out *= ch_inf;
132  servo_out += channel->pwm_to_angle();
133 }

153 /*
154  this gives the user control of the aircraft in stabilization modes
155  using FBW style controls
156  */
157 static void stabilize_stick_mixing_fbw()
158 {
159  if (!stick_mixing_enabled() ||
160  control_mode == ACRO ||
161  control_mode == FLY_BY_WIRE_A ||
162  control_mode == AUTOTUNE ||
163  control_mode == FLY_BY_WIRE_B ||
164  control_mode == CRUISE ||
165  control_mode == TRAINING ||
166  (control_mode == AUTO && g.auto_fbw_steer)) {
167  return;
168  }
169  // do FBW style stick mixing. We don't treat it linearly
170  // however. For inputs up to half the maximum, we use linear
171  // addition to the nav_roll and nav_pitch. Above that it goes
172  // non-linear and ends up as 2x the maximum, to ensure that
173  // the user can direct the plane in any direction with stick
174  // mixing.
175  float roll_input = channel_roll->norm_input();
176  if (roll_input > 0.5f) {
177  roll_input = (3*roll_input - 1);
178  } else if (roll_input < -0.5f) {
179  roll_input = (3*roll_input + 1);
180  }
181  nav_roll_cd += roll_input * roll_limit_cd;
182  nav_roll_cd = constrain_int32(nav_roll_cd, -roll_limit_cd, roll_limit_cd);
183 
184  float pitch_input = channel_pitch->norm_input();
185  if (fabsf(pitch_input) > 0.5f) {
186  pitch_input = (3*pitch_input - 1);
187  }
188  if (fly_inverted()) {
189  pitch_input = -pitch_input;
190  }
191  if (pitch_input > 0) {
192  nav_pitch_cd += pitch_input * aparm.pitch_limit_max_cd;
193  } else {
194  nav_pitch_cd += -(pitch_input * pitch_limit_min_cd);
195  }
196  nav_pitch_cd = constrain_int32(nav_pitch_cd, pitch_limit_min_cd, aparm.pitch_limit_max_cd.get());
197 }

277 /*
278  this is the ACRO mode stabilization function. It does rate
279  stabilization on roll and pitch axes
280  */
281 static void stabilize_acro(float speed_scaler)
282 {
283  float roll_rate = (channel_roll->control_in/4500.0f) * g.acro_roll_rate;
284  float pitch_rate = (channel_pitch->control_in/4500.0f) * g.acro_pitch_rate;
285 
286  /*
287  check for special roll handling near the pitch poles
288  */
289  if (g.acro_locking && is_zero(roll_rate)) {
290  /*
291  we have no roll stick input, so we will enter "roll locked"
292  mode, and hold the roll we had when the stick was released
293  */
294  if (!acro_state.locked_roll) {
295  acro_state.locked_roll = true;
296  acro_state.locked_roll_err = 0;
297  } else {
298  acro_state.locked_roll_err += ahrs.get_gyro().x * G_Dt;
299  }
300  int32_t roll_error_cd = -ToDeg(acro_state.locked_roll_err)*100;
301  nav_roll_cd = ahrs.roll_sensor + roll_error_cd;
302  // try to reduce the integrated angular error to zero. We set
303  // 'stabilze' to true, which disables the roll integrator
304  channel_roll->servo_out = rollController.get_servo_out(roll_error_cd,
305  speed_scaler,
306  true);
307  } else {
308  /*
309  aileron stick is non-zero, use pure rate control until the
310  user releases the stick
311  */
312  acro_state.locked_roll = false;
313  channel_roll->servo_out = rollController.get_rate_out(roll_rate, speed_scaler);
314  }
315 
316  if (g.acro_locking && is_zero(pitch_rate)) {
317  /*
318  user has zero pitch stick input, so we lock pitch at the
319  point they release the stick
320  */
321  if (!acro_state.locked_pitch) {
322  acro_state.locked_pitch = true;
323  acro_state.locked_pitch_cd = ahrs.pitch_sensor;
324  }
325  // try to hold the locked pitch. Note that we have the pitch
326  // integrator enabled, which helps with inverted flight
327  nav_pitch_cd = acro_state.locked_pitch_cd;
328  channel_pitch->servo_out = pitchController.get_servo_out(nav_pitch_cd - ahrs.pitch_sensor,
329  speed_scaler,
330  false);
331  } else {
332  /*
333  user has non-zero pitch input, use a pure rate controller
334  */
335  acro_state.locked_pitch = false;
336  channel_pitch->servo_out = pitchController.get_rate_out(pitch_rate, speed_scaler);
337  }
338 
339  /*
340  manual rudder for now
341  */
342  steering_control.steering = steering_control.rudder = rudder_input;
343 }

345 /*
346  main stabilization function for all 3 axes
347  */
348 static void stabilize()
349 {
350  if (control_mode == MANUAL) {
351  // nothing to do
352  return;
353  }
354  float speed_scaler = get_speed_scaler();
355 
356  if (control_mode == TRAINING) {
357  stabilize_training(speed_scaler);
358  } else if (control_mode == ACRO) {
359  stabilize_acro(speed_scaler);
360  } else {
361  if (g.stick_mixing == STICK_MIXING_FBW && control_mode != STABILIZE) {
362  stabilize_stick_mixing_fbw();
363  }
364  stabilize_roll(speed_scaler);
365  stabilize_pitch(speed_scaler);
366  if (g.stick_mixing == STICK_MIXING_DIRECT || control_mode == STABILIZE) {
367  stabilize_stick_mixing_direct();
368  }
369  stabilize_yaw(speed_scaler);
370  }
371 
372  /*
373  see if we should zero the attitude controller integrators.
374  */
375  if (channel_throttle->control_in == 0 &&
376  relative_altitude_abs_cm() < 500 &&
377  fabsf(barometer.get_climb_rate()) < 0.5f &&
378  gps.ground_speed() < 3) {
379  // we are low, with no climb rate, and zero throttle, and very
380  // low ground speed. Zero the attitude controller
381  // integrators. This prevents integrator buildup pre-takeoff.
382  rollController.reset_I();
383  pitchController.reset_I();
384  yawController.reset_I();
385 
386  // if moving very slowly also zero the steering integrator
387  if (gps.ground_speed() < 1) {
388  steerController.reset_I();
389  }
390  }
391 }

443 /*
444  calculate yaw control for ground steering
445  */
446 static void calc_nav_yaw_ground(void)
447 {
448  if (gps.ground_speed() < 1 &&
449  channel_throttle->control_in == 0 &&
450  flight_stage != AP_SpdHgtControl::FLIGHT_TAKEOFF) {
451  // manual rudder control while still
452  steer_state.locked_course = false;
453  steer_state.locked_course_err = 0;
454  steering_control.steering = rudder_input;
455  return;
456  }
457

459  if (flight_stage == AP_SpdHgtControl::FLIGHT_TAKEOFF) {
460  steer_rate = 0;
461  }
462  if (!is_zero(steer_rate)) {
463  // pilot is giving rudder input
464  steer_state.locked_course = false;
465  } else if (!steer_state.locked_course) {
466  // pilot has released the rudder stick or we are still - lock the course
467  steer_state.locked_course = true;
468  if (flight_stage != AP_SpdHgtControl::FLIGHT_TAKEOFF) {
469  steer_state.locked_course_err = 0;
470  }
471  }
472  if (!steer_state.locked_course) {
473  // use a rate controller at the pilot specified rate
474  steering_control.steering = steerController.get_steering_out_rate(steer_rate);
475  } else {
476  // use a error controller on the summed error
477  int32_t yaw_error_cd = -ToDeg(steer_state.locked_course_err)*100;
478  steering_control.steering = steerController.get_steering_out_angle_error(yaw_error_cd);
479  }
480  steering_control.steering = constrain_int16(steering_control.steering, -4500, 4500);
481 }

508 * Throttle slew limit
509 *****************************************/
510 static void throttle_slew_limit(int16_t last_throttle)
511 {
512  uint8_t slewrate = aparm.throttle_slewrate;
513  if (control_mode==AUTO && auto_state.takeoff_complete == false && g.takeoff_throttle_slewrate != 0) {
514  slewrate = g.takeoff_throttle_slewrate;
515  }
516  // if slew limit rate is set to zero then do not slew limit
517  if (slewrate) {
518  // limit throttle change by the given percentage per second
519  float temp = slewrate * G_Dt * 0.01f * fabsf(channel_throttle->radio_max - channel_throttle->radio_min);
520  // allow a minimum change of 1 PWM per cycle
521  if (temp < 1) {
522  temp = 1;
523  }
524  channel_throttle->radio_out = constrain_int16(channel_throttle->radio_out, last_throttle - temp, last_throttle + temp);
525  }

531 static void flap_slew_limit(int8_t &last_value, int8_t &new_value)
532 {

534  // if slew limit rate is set to zero then do not slew limit

548 /* We want to suppress the throttle if we think we are on the ground and in an autopilot controlled throttle mode.
549 
550  Disable throttle if following conditions are met:
551  * 1 - We are in Circle mode (which we use for short term failsafe), or in FBW-B or higher
552  * AND
553  * 2 - Our reported altitude is within 10 meters of the home altitude.
554  * 3 - Our reported speed is under 5 meters per second.
555  * 4 - We are not performing a takeoff in Auto mode or takeoff speed/accel not yet reached
556  * OR
557  * 5 - Home location is not set
558 */
559 static bool suppress_throttle(void)
560 {
561  if (!throttle_suppressed) {
562  // we've previously met a condition for unsupressing the throttle
563  return false;
564  }
565  if (!auto_throttle_mode) {
566  // the user controls the throttle
567  throttle_suppressed = false;
568  return false;
569  }
570 
571  if (control_mode==AUTO && g.auto_fbw_steer) {
572  // user has throttle control
573  return false;
574  }
575 
576  if (control_mode==AUTO &&
577  auto_state.takeoff_complete == false) {
578  if (auto_takeoff_check()) {
579  // we're in auto takeoff
580  throttle_suppressed = false;
581  return false;
582  }
583  // keep throttle suppressed
584  return true;
585  }
586 
587  if (relative_altitude_abs_cm() >= 1000) {
588  // we're more than 10m from the home altitude
589  throttle_suppressed = false;
590  gcs_send_text_fmt(PSTR("Throttle unsuppressed - altitude %.2f"),
591  (double)(relative_altitude_abs_cm()*0.01f));
592  return false;
593  }
594 
595  if (gps.status() >= AP_GPS::GPS_OK_FIX_2D &&
596  gps.ground_speed() >= 5) {
597  // if we have an airspeed sensor, then check it too, and
598  // require 5m/s. This prevents throttle up due to spiky GPS
599  // groundspeed with bad GPS reception
600  if ((!ahrs.airspeed_sensor_enabled()) || airspeed.get_airspeed() >= 5) {
601  // we're moving at more than 5 m/s
602  gcs_send_text_fmt(PSTR("Throttle unsuppressed - speed %.2f airspeed %.2f"),
603  (double)gps.ground_speed(),
604  (double)airspeed.get_airspeed());
605  throttle_suppressed = false;
606  return false;
607  }
608  }
609 
610  // throttle remains suppressed
611  return true;
612 }

614 /*
615  implement a software VTail or elevon mixer. There are 4 different mixing modes
616  */
617 static void channel_output_mixer(uint8_t mixing_type, int16_t &chan1_out, int16_t &chan2_out)
618 {
619  int16_t c1, c2;
620  int16_t v1, v2;
621 
622  // first get desired elevator and rudder as -500..500 values
623  c1 = chan1_out - 1500;
624  c2 = chan2_out - 1500;
625 
626  v1 = (c1 - c2) * g.mixing_gain;
627  v2 = (c1 + c2) * g.mixing_gain;
628 
629  // now map to mixed output
630  switch (mixing_type) {

689 * Set the flight control servos based on the current calculated values

694 
695  /*

697  */

700  // wheel to the rudder just in case the barometer has drifted
701  // a lot

703  } else if (!RC_Channel_aux::function_assigned(RC_Channel_aux::k_steering)) {
704  // we are within the ground steering altitude but don't have a
705  // dedicated steering channel. Set the rudder to the ground
706  // steering output

708  }

710 
711  // clear ground_steering to ensure manual control if the yaw stabilizer doesn't run
712  steering_control.ground_steering = false;
713 
714  RC_Channel_aux::set_servo_out(RC_Channel_aux::k_rudder, steering_control.rudder);
715  RC_Channel_aux::set_servo_out(RC_Channel_aux::k_steering, steering_control.steering);
716 
717  if (control_mode == MANUAL) {
718  // do a direct pass through of radio values
719  if (g.mix_mode == 0 || g.elevon_output != MIXING_DISABLED) {
720  channel_roll->radio_out = channel_roll->radio_in;
721  channel_pitch->radio_out = channel_pitch->radio_in;
722  } else {
723  channel_roll->radio_out = channel_roll->read();
724  channel_pitch->radio_out = channel_pitch->read();
725  }
726  channel_throttle->radio_out = channel_throttle->radio_in;
727  channel_rudder->radio_out = channel_rudder->radio_in;

730  // main input channel, but using the 2nd channels dead

732  // pwm_to_angle_dz() to ensure we don't trim the value for the
733  // deadzone of the main aileron channel, otherwise the 2nd
734  // aileron won't quite follow the first one
735  RC_Channel_aux::set_servo_out(RC_Channel_aux::k_aileron, channel_roll->pwm_to_angle_dz(0));
736  RC_Channel_aux::set_servo_out(RC_Channel_aux::k_elevator, channel_pitch->pwm_to_angle_dz(0));
737

739  // input channel setup in your transmitter for manual control
740  // of the 2nd aileron

742  RC_Channel_aux::copy_radio_in_out(RC_Channel_aux::k_elevator_with_input);
743 
744  if (g.mix_mode == 0 && g.elevon_output == MIXING_DISABLED) {
745  // set any differential spoilers to follow the elevons in
746  // manual mode.
747  RC_Channel_aux::set_radio(RC_Channel_aux::k_dspoiler1, channel_roll->radio_out);
748  RC_Channel_aux::set_radio(RC_Channel_aux::k_dspoiler2, channel_pitch->radio_out);
749  }
750  } else {
751  if (g.mix_mode == 0) {
752  // both types of secondary aileron are slaved to the roll servo out
753  RC_Channel_aux::set_servo_out(RC_Channel_aux::k_aileron, channel_roll->servo_out);
754  RC_Channel_aux::set_servo_out(RC_Channel_aux::k_aileron_with_input, channel_roll->servo_out);
755 
756  // both types of secondary elevator are slaved to the pitch servo out
757  RC_Channel_aux::set_servo_out(RC_Channel_aux::k_elevator, channel_pitch->servo_out);
758  RC_Channel_aux::set_servo_out(RC_Channel_aux::k_elevator_with_input, channel_pitch->servo_out);
759  }else{
760  /*Elevon mode*/
761  float ch1;
762  float ch2;
763  ch1 = channel_pitch->servo_out - (BOOL_TO_SIGN(g.reverse_elevons) * channel_roll->servo_out);
764  ch2 = channel_pitch->servo_out + (BOOL_TO_SIGN(g.reverse_elevons) * channel_roll->servo_out);
765 
766  /* Differential Spoilers
767  If differential spoilers are setup, then we translate
768  rudder control into splitting of the two ailerons on
769  the side of the aircraft where we want to induce
770  additional drag.
771  */
772  if (RC_Channel_aux::function_assigned(RC_Channel_aux::k_dspoiler1) && RC_Channel_aux::function_assigned(RC_Channel_aux::k_dspoiler2)) {
773  float ch3 = ch1;
774  float ch4 = ch2;
775  if ( BOOL_TO_SIGN(g.reverse_elevons) * channel_rudder->servo_out < 0) {
776  ch1 += abs(channel_rudder->servo_out);
777  ch3 -= abs(channel_rudder->servo_out);
778  } else {
779  ch2 += abs(channel_rudder->servo_out);
780  ch4 -= abs(channel_rudder->servo_out);
781  }
782  RC_Channel_aux::set_servo_out(RC_Channel_aux::k_dspoiler1, ch3);
783  RC_Channel_aux::set_servo_out(RC_Channel_aux::k_dspoiler2, ch4);
784  }

787  channel_roll->radio_out = elevon.trim1 + (BOOL_TO_SIGN(g.reverse_ch1_elevon) * (ch1 * 500.0f/ SERVO_MAX));
788  channel_pitch->radio_out = elevon.trim2 + (BOOL_TO_SIGN(g.reverse_ch2_elevon) * (ch2 * 500.0f/ SERVO_MAX));
789  }
790 
791  // push out the PWM values
792  if (g.mix_mode == 0) {
793  channel_roll->calc_pwm();
794  channel_pitch->calc_pwm();
795  }
796  channel_rudder->calc_pwm();
797 
798 #if THROTTLE_OUT == 0
799  channel_throttle->servo_out = 0;
800 #else

802  uint8_t min_throttle = aparm.throttle_min.get();
803  uint8_t max_throttle = aparm.throttle_max.get();
804  if (control_mode == AUTO && flight_stage == AP_SpdHgtControl::FLIGHT_LAND_FINAL) {
805  min_throttle = 0;
806  }
807  if (control_mode == AUTO && flight_stage == AP_SpdHgtControl::FLIGHT_TAKEOFF) {

829  }

1017 /*
1018  calculate a new aerodynamic_load_factor and limit nav_roll_cd to
1019  ensure that the load factor does not take us below the sustainable
1020  airspeed
1021  */
1022 static void update_load_factor(void)
1023 {
1024  float demanded_roll = fabsf(nav_roll_cd*0.01f);
1025  if (demanded_roll > 85) {
1026  // limit to 85 degrees to prevent numerical errors
1027  demanded_roll = 85;
1028  }
1029  aerodynamic_load_factor = 1.0f / safe_sqrt(cosf(radians(demanded_roll)));
1030 
1031  if (!aparm.stall_prevention) {
1032  // stall prevention is disabled
1033  return;
1034  }
1035  if (fly_inverted()) {
1036  // no roll limits when inverted
1037  return;
1038  }
1039 
1040  float max_load_factor = smoothed_airspeed / aparm.airspeed_min;
1041  if (max_load_factor <= 1) {
1042  // our airspeed is below the minimum airspeed. Limit roll to
1043  // 25 degrees
1044  nav_roll_cd = constrain_int32(nav_roll_cd, -2500, 2500);
1045  } else if (max_load_factor < aerodynamic_load_factor) {
1046  // the demanded nav_roll would take us past the aerodymamic
1047  // load limit. Limit our roll to a bank angle that will keep
1048  // the load within what the airframe can handle. We always
1049  // allow at least 25 degrees of roll however, to ensure the
1050  // aircraft can be maneuvered with a bad airspeed estimate. At
1051  // 25 degrees the load factor is 1.1 (10%)
1052  int32_t roll_limit = degrees(acosf(sq(1.0f / max_load_factor)))*100;
1053  if (roll_limit < 2500) {
1054  roll_limit = 2500;
1055  }
1056  nav_roll_cd = constrain_int32(nav_roll_cd, -roll_limit, roll_limit);
1057  }
