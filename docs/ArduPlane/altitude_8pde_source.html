22 /*
23  adjust altitude target depending on mode
24  */
25 static void adjust_altitude_target()
26 {
27  if (control_mode == FLY_BY_WIRE_B ||
28  control_mode == CRUISE) {
29  return;
30  }
31  if (flight_stage == AP_SpdHgtControl::FLIGHT_LAND_FINAL) {
32  // in land final TECS uses TECS_LAND_SINK as a target sink
33  // rate, and ignores the target altitude
34  set_target_altitude_location(next_WP_loc);
35  } else if (flight_stage == AP_SpdHgtControl::FLIGHT_LAND_APPROACH) {
36  setup_landing_glide_slope();
37  } else if (nav_controller->reached_loiter_target()) {
38  // once we reach a loiter target then lock to the final
39  // altitude target
40  set_target_altitude_location(next_WP_loc);
41  } else if (target_altitude.offset_cm != 0 &&
42  !location_passed_point(current_loc, prev_WP_loc, next_WP_loc)) {
43  // control climb/descent rate
44  set_target_altitude_proportion(next_WP_loc, 1.0f-auto_state.wp_proportion);
45 
46  // stay within the range of the start and end locations in altitude
47  constrain_target_altitude_location(next_WP_loc, prev_WP_loc);
48  } else if (mission.get_current_do_cmd().id != MAV_CMD_CONDITION_CHANGE_ALT) {
49  set_target_altitude_location(next_WP_loc);
50  }
51 
52  altitude_error_cm = calc_altitude_error_cm();
53 }

55 /*
56  setup for a gradual glide slope to the next waypoint, if appropriate
57  */
58 static void setup_glide_slope(void)
59 {
60  // establish the distance we are travelling to the next waypoint,
61  // for calculating out rate of change of altitude
62  auto_state.wp_distance = get_distance(current_loc, next_WP_loc);
63  auto_state.wp_proportion = location_path_proportion(current_loc,
64  prev_WP_loc, next_WP_loc);
65 
66  /*
67  work out if we will gradually change altitude, or try to get to
68  the new altitude as quickly as possible.
69  */
70  switch (control_mode) {
71  case RTL:
72  case GUIDED:
73  /* glide down slowly if above target altitude, but ascend more
74  rapidly if below it. See
75  https://github.com/diydrones/ardupilot/issues/39
76  */
77  if (above_location_current(next_WP_loc)) {
78  set_offset_altitude_location(next_WP_loc);
79  } else {
80  reset_offset_altitude();
81  }
82  break;
83 
84  case AUTO:
85  // we only do glide slide handling in AUTO when above 20m or
86  // when descending. The 20 meter threshold is arbitrary, and
87  // is basically to prevent situations where we try to slowly
88  // gain height at low altitudes, potentially hitting
89  // obstacles.
90  if (adjusted_relative_altitude_cm() > 2000 || above_location_current(next_WP_loc)) {
91  set_offset_altitude_location(next_WP_loc);
92  } else {
93  reset_offset_altitude();
94  }
95  break;
96  default:
97  reset_offset_altitude();
98  break;
99  }
100 }

133  setting up for altitude hold, such as when releasing elevator in

137 {

173  */
174 static void set_target_altitude_location(const Location &loc)
175 {

178  target_altitude.amsl_cm += home.alt;

200 /*
201  return relative to home target altitude in centimeters. Used for
202  altitude control libraries
203  */
204 static int32_t relative_target_altitude_cm(void)
205 {
206 #if AP_TERRAIN_AVAILABLE
207  float relative_home_height;
208  if (target_altitude.terrain_following &&
209  terrain.height_relative_home_equivalent(target_altitude.terrain_alt_cm*0.01f,
210  relative_home_height, true)) {
211  // add lookahead adjustment the target altitude
212  target_altitude.lookahead = lookahead_adjustment();
213  relative_home_height += target_altitude.lookahead;
214 
215  // correct for rangefinder data
216  relative_home_height += rangefinder_correction();
217 
218  // we are following terrain, and have terrain data for the
219  // current location. Use it.
220  return relative_home_height*100;
221  }
222 #endif
223  int32_t relative_alt = target_altitude.amsl_cm - home.alt;
224  relative_alt += int32_t(g.alt_offset)*100;
225  relative_alt += rangefinder_correction() * 100;
226  return relative_alt;
227 }

243 /*
244  change target altitude by a proportion of the target altitude offset
245  (difference in height to next WP from previous WP). proportion
246  should be between 0 and 1.
247 
248  When proportion is zero we have reached the destination. When
249  proportion is 1 we are at the starting waypoint.
250 
251  Note that target_altitude is setup initially based on the
252  destination waypoint
253  */
254 static void set_target_altitude_proportion(const Location &loc, float proportion)
255 {
256  set_target_altitude_location(loc);
257  proportion = constrain_float(proportion, 0.0f, 1.0f);
258  change_target_altitude(-target_altitude.offset_cm*proportion);
259  //rebuild the glide slope if we are above it and supposed to be climbing
260  if(g.glide_slope_threshold > 0) {
261  if(target_altitude.offset_cm > 0 && calc_altitude_error_cm() < -100 * g.glide_slope_threshold) {
262  set_target_altitude_location(loc);
263  set_offset_altitude_location(loc);
264  change_target_altitude(-target_altitude.offset_cm*proportion);
265  //adjust the new target offset altitude to reflect that we are partially already done
266  if(proportion > 0.0f)
267  target_altitude.offset_cm = ((float)target_altitude.offset_cm)/proportion;
268  }
269  }
270 }

333 /*
334  reset the altitude offset used for glide slopes, based on difference
335  between altitude at a destination and current altitude. If
336  destination is above the current altitude then the result is
337  positive.
338  */
339 static void set_offset_altitude_location(const Location &loc)
340 {
341  target_altitude.offset_cm = loc.alt - current_loc.alt;
342 
343 #if AP_TERRAIN_AVAILABLE
344  /*
345  if this location has the terrain_alt flag set and we know the
346  terrain altitude of our current location then treat it as a
347  terrain altitude
348  */
349  float height;
350  if (loc.flags.terrain_alt &&
351  target_altitude.terrain_following &&
352  terrain.height_above_terrain(height, true)) {
353  target_altitude.offset_cm = target_altitude.terrain_alt_cm - (height * 100);
354  }
355 #endif
356 
357  if (flight_stage != AP_SpdHgtControl::FLIGHT_LAND_APPROACH &&
358  flight_stage != AP_SpdHgtControl::FLIGHT_LAND_FINAL) {
359  // if we are within GLIDE_SLOPE_MIN meters of the target altitude
360  // then reset the offset to not use a glide slope. This allows for
361  // more accurate flight of missions where the aircraft may lose or
362  // gain a bit of altitude near waypoint turn points due to local
363  // terrain changes
364  if (g.glide_slope_min <= 0 ||
365  labs(target_altitude.offset_cm)*0.01f < g.glide_slope_min) {
366  target_altitude.offset_cm = 0;
367  }
368  }
369 }

371 /*
372  return true if current_loc is above loc. Used for glide slope
373  calculations.
374 
375  "above" is simple if we are not terrain following, as it just means
376  the pressure altitude of one is above the other.
377 
378  When in terrain following mode "above" means the over-the-terrain
379  current altitude is above the over-the-terrain alt of loc. It is
380  quite possible for current_loc to be "above" loc when it is at a
381  lower pressure altitude, if current_loc is in a low part of the
382  terrain
383  */
384 static bool above_location_current(const Location &loc)
385 {
386 #if AP_TERRAIN_AVAILABLE
387  float terrain_alt;
388  if (loc.flags.terrain_alt &&
389  terrain.height_above_terrain(terrain_alt, true)) {
390  float loc_alt = loc.alt*0.01f;
391  if (!loc.flags.relative_alt) {
392  loc_alt -= home.alt*0.01f;
393  }
394  return terrain_alt > loc_alt;
395  }
396 #endif
397 
398  float loc_alt_cm = loc.alt;
399  if (!loc.flags.relative_alt) {
400  loc_alt_cm -= home.alt;
401  }
402  return current_loc.alt > loc_alt_cm;
403 }

460 /*
461  work out target altitude adjustment from terrain lookahead
462  */
463 static float lookahead_adjustment(void)
464 {
465 #if AP_TERRAIN_AVAILABLE
466  int32_t bearing_cd;
467  int16_t distance;
468  // work out distance and bearing to target
469  if (control_mode == FLY_BY_WIRE_B) {
470  // there is no target waypoint in FBWB, so use yaw as an approximation
471  bearing_cd = ahrs.yaw_sensor;
472  distance = g.terrain_lookahead;
473  } else if (!nav_controller->reached_loiter_target()) {
474  bearing_cd = nav_controller->target_bearing_cd();
475  distance = constrain_float(auto_state.wp_distance, 0, g.terrain_lookahead);
476  } else {
477  // no lookahead when loitering
478  bearing_cd = 0;
479  distance = 0;
480  }
481  if (distance <= 0) {
482  // no lookahead
483  return 0;
484  }
485 
486 
487  float groundspeed = ahrs.groundspeed();
488  if (groundspeed < 1) {
489  // we're not moving
490  return 0;
491  }
492  // we need to know the climb ratio. We use 50% of the maximum
493  // climb rate so we are not constantly at 100% throttle and to
494  // give a bit more margin on terrain
495  float climb_ratio = 0.5f * SpdHgt_Controller->get_max_climbrate() / groundspeed;
496 
497  if (climb_ratio <= 0) {
498  // lookahead makes no sense for negative climb rates
499  return 0;
500  }
501 
502  // ask the terrain code for the lookahead altitude change
503  float lookahead = terrain.lookahead(bearing_cd*0.01f, distance, climb_ratio);
504 
505  if (target_altitude.offset_cm < 0) {
506  // we are heading down to the waypoint, so we don't need to
507  // climb as much
508  lookahead += target_altitude.offset_cm*0.01f;
509  }
510 
511  // constrain lookahead to a reasonable limit
512  return constrain_float(lookahead, 0, 1000.0f);
513 #else
514  return 0;
515 #endif
516 }

519 /*
520  correct target altitude using rangefinder data. Returns offset in
521  meters to correct target altitude. A positive number means we need
522  to ask the speed/height controller to fly higher
523  */
524 static float rangefinder_correction(void)
525 {
526  if (hal.scheduler->millis() - rangefinder_state.last_correction_time_ms > 5000) {
527  // we haven't had any rangefinder data for 5s - don't use it
528  return 0;
529  }
530 
531  // for now we only support the rangefinder for landing
532  bool using_rangefinder = (g.rangefinder_landing &&
533  control_mode == AUTO &&
534  (flight_stage == AP_SpdHgtControl::FLIGHT_LAND_APPROACH ||
535  flight_stage == AP_SpdHgtControl::FLIGHT_LAND_FINAL));
536  if (!using_rangefinder) {
537  return 0;
538  }
539 
540  return rangefinder_state.correction;
541 }

544 /*
545  update the offset between rangefinder height and terrain height
546  */
547 static void rangefinder_height_update(void)
548 {
549  uint16_t distance_cm = rangefinder.distance_cm();
550  float height_estimate = 0;
551  if ((rangefinder.status() == RangeFinder::RangeFinder_Good) && home_is_set != HOME_UNSET) {
552  // correct the range for attitude (multiply by DCM.c.z, which
553  // is cos(roll)*cos(pitch))
554  height_estimate = distance_cm * 0.01f * ahrs.get_dcm_matrix().c.z;
555 
556  // we consider ourselves to be fully in range when we have 10
557  // good samples (0.2s)
558  if (rangefinder_state.in_range_count < 10) {
559  rangefinder_state.in_range_count++;
560  } else {
561  rangefinder_state.in_range = true;
562  }
563  } else {
564  rangefinder_state.in_range_count = 0;
565  rangefinder_state.in_range = false;
566  }
567 
568  if (rangefinder_state.in_range) {
569  // base correction is the difference between baro altitude and
570  // rangefinder estimate
571  float correction = relative_altitude() - height_estimate;
572 
573 #if AP_TERRAIN_AVAILABLE
574  // if we are terrain following then correction is based on terrain data
575  float terrain_altitude;
576  if ((target_altitude.terrain_following || g.terrain_follow) &&
577  terrain.height_above_terrain(terrain_altitude, true)) {
578  correction = terrain_altitude - height_estimate;
579  }
580 #endif
581 
582  // remember the last correction. Use a low pass filter unless
583  // the old data is more than 5 seconds old
584  if (hal.scheduler->millis() - rangefinder_state.last_correction_time_ms > 5000) {
585  rangefinder_state.correction = correction;
586  } else {
587  rangefinder_state.correction = 0.8f*rangefinder_state.correction + 0.2f*correction;
588  }
589  rangefinder_state.last_correction_time_ms = hal.scheduler->millis();
590  }
