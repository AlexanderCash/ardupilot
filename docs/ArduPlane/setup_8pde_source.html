78 static int8_t setup_set(uint8_t argc, const Menu::arg *argv)
79 {
80  int8_t value_int8;
81  int16_t value_int16;
82 
83  AP_Param *param;
84  enum ap_var_type p_type;
85 
86  if(argc!=3)
87  {
88  cliSerial->printf_P(PSTR("Invalid command. Usage: set <name> <value>\n"));
89  return 0;
90  }
91 
92  param = AP_Param::find(argv[1].str, &p_type);
93  if(!param)
94  {
95  cliSerial->printf_P(PSTR("Param not found: %s\n"), argv[1].str);
96  return 0;
97  }
98 
99  switch(p_type)
100  {
101  case AP_PARAM_INT8:
102  value_int8 = (int8_t)(argv[2].i);
103  if(argv[2].i!=value_int8)
104  {
105  cliSerial->printf_P(PSTR("Value out of range for type INT8\n"));
106  return 0;
107  }
108  ((AP_Int8*)param)->set_and_save(value_int8);
109  break;
110  case AP_PARAM_INT16:
111  value_int16 = (int16_t)(argv[2].i);
112  if(argv[2].i!=value_int16)
113  {
114  cliSerial->printf_P(PSTR("Value out of range for type INT16\n"));
115  return 0;
116  }
117  ((AP_Int16*)param)->set_and_save(value_int16);
118  break;
119 
120  //int32 and float don't need bounds checking, just use the value provoded by Menu::arg
121  case AP_PARAM_INT32:
122  ((AP_Int32*)param)->set_and_save(argv[2].i);
123  break;
124  case AP_PARAM_FLOAT:
125  ((AP_Float*)param)->set_and_save(argv[2].f);
126  break;
127  default:
128  cliSerial->printf_P(PSTR("Cannot set parameter of type %d.\n"), p_type);
129  break;
130  }
131 
132  return 0;
133 }
