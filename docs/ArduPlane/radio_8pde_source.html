127 static void read_radio()
128 {
129  if (!hal.rcin->new_input()) {
130  control_failsafe(channel_throttle->radio_in);
131  return;
132  }
133 
134  failsafe.last_valid_rc_ms = hal.scheduler->millis();
135 
136  elevon.ch1_temp = channel_roll->read();
137  elevon.ch2_temp = channel_pitch->read();
138  uint16_t pwm_roll, pwm_pitch;
139 
140  if (g.mix_mode == 0) {
141  pwm_roll = elevon.ch1_temp;
142  pwm_pitch = elevon.ch2_temp;
143  }else{
144  pwm_roll = BOOL_TO_SIGN(g.reverse_elevons) * (BOOL_TO_SIGN(g.reverse_ch2_elevon) * int16_t(elevon.ch2_temp - elevon.trim2) - BOOL_TO_SIGN(g.reverse_ch1_elevon) * int16_t(elevon.ch1_temp - elevon.trim1)) / 2 + 1500;
145  pwm_pitch = (BOOL_TO_SIGN(g.reverse_ch2_elevon) * int16_t(elevon.ch2_temp - elevon.trim2) + BOOL_TO_SIGN(g.reverse_ch1_elevon) * int16_t(elevon.ch1_temp - elevon.trim1)) / 2 + 1500;
146  }
147 
148  RC_Channel::set_pwm_all();
149 
150  if (control_mode == TRAINING) {
151  // in training mode we don't want to use a deadzone, as we
152  // want manual pass through when not exceeding attitude limits
153  channel_roll->set_pwm_no_deadzone(pwm_roll);
154  channel_pitch->set_pwm_no_deadzone(pwm_pitch);
155  channel_throttle->set_pwm_no_deadzone(channel_throttle->read());
156  channel_rudder->set_pwm_no_deadzone(channel_rudder->read());
157  } else {
158  channel_roll->set_pwm(pwm_roll);
159  channel_pitch->set_pwm(pwm_pitch);
160  }
161 
162  control_failsafe(channel_throttle->radio_in);
163 
164  channel_throttle->servo_out = channel_throttle->control_in;
165 
166  if (g.throttle_nudge && channel_throttle->servo_out > 50) {
167  float nudge = (channel_throttle->servo_out - 50) * 0.02f;
168  if (ahrs.airspeed_sensor_enabled()) {
169  airspeed_nudge_cm = (aparm.airspeed_max * 100 - g.airspeed_cruise_cm) * nudge;
170  } else {
171  throttle_nudge = (aparm.throttle_max - aparm.throttle_cruise) * nudge;
172  }
173  } else {
174  airspeed_nudge_cm = 0;
175  throttle_nudge = 0;
176  }
177 
178  rudder_arm_check();
179 
180  if (g.rudder_only != 0) {
181  // in rudder only mode we discard rudder input and get target
182  // attitude from the roll channel.
183  rudder_input = 0;
184  } else {
185  rudder_input = channel_rudder->control_in;
186  }
187 }

189 static void control_failsafe(uint16_t pwm)
190 {
191  if (hal.scheduler->millis() - failsafe.last_valid_rc_ms > 1000 || rc_failsafe_active()) {
192  // we do not have valid RC input. Set all primary channel
193  // control inputs to the trim value and throttle to min
194  channel_roll->radio_in = channel_roll->radio_trim;
195  channel_pitch->radio_in = channel_pitch->radio_trim;
196  channel_rudder->radio_in = channel_rudder->radio_trim;
197 
198  // note that we don't set channel_throttle->radio_in to radio_trim,
199  // as that would cause throttle failsafe to not activate
200 
201  channel_roll->control_in = 0;
202  channel_pitch->control_in = 0;
203  channel_rudder->control_in = 0;
204  channel_throttle->control_in = 0;
205  }
206 
207  if(g.throttle_fs_enabled == 0)
208  return;
209 
210  if (g.throttle_fs_enabled) {
211  if (rc_failsafe_active()) {
212  // we detect a failsafe from radio
213  // throttle has dropped below the mark
214  failsafe.ch3_counter++;
215  if (failsafe.ch3_counter == 10) {
216  gcs_send_text_fmt(PSTR("MSG FS ON %u"), (unsigned)pwm);
217  failsafe.ch3_failsafe = true;
218  AP_Notify::flags.failsafe_radio = true;
219  }
220  if (failsafe.ch3_counter > 10) {
221  failsafe.ch3_counter = 10;
222  }
223 
224  }else if(failsafe.ch3_counter > 0) {
225  // we are no longer in failsafe condition
226  // but we need to recover quickly
227  failsafe.ch3_counter--;
228  if (failsafe.ch3_counter > 3) {
229  failsafe.ch3_counter = 3;
230  }
231  if (failsafe.ch3_counter == 1) {
232  gcs_send_text_fmt(PSTR("MSG FS OFF %u"), (unsigned)pwm);
233  } else if(failsafe.ch3_counter == 0) {
234  failsafe.ch3_failsafe = false;
235  AP_Notify::flags.failsafe_radio = false;
236  }
237  }
238  }

241 static void trim_control_surfaces()
242 {
243  read_radio();
244  int16_t trim_roll_range = (channel_roll->radio_max - channel_roll->radio_min)/5;
245  int16_t trim_pitch_range = (channel_pitch->radio_max - channel_pitch->radio_min)/5;
246  if (channel_roll->radio_in < channel_roll->radio_min+trim_roll_range ||
247  channel_roll->radio_in > channel_roll->radio_max-trim_roll_range ||
248  channel_pitch->radio_in < channel_pitch->radio_min+trim_pitch_range ||
249  channel_pitch->radio_in > channel_pitch->radio_max-trim_pitch_range) {
250  // don't trim for extreme values - if we attempt to trim so
251  // there is less than 20 percent range left then assume the
252  // sticks are not properly centered. This also prevents
253  // problems with starting APM with the TX off
254  return;
255  }
256 
257  // Store control surface trim values
258  // ---------------------------------
259  if(g.mix_mode == 0) {
260  if (channel_roll->radio_in != 0) {
261  channel_roll->radio_trim = channel_roll->radio_in;
262  }
263  if (channel_pitch->radio_in != 0) {
264  channel_pitch->radio_trim = channel_pitch->radio_in;
265  }
266 
267  // the secondary aileron/elevator is trimmed only if it has a
268  // corresponding transmitter input channel, which k_aileron
269  // doesn't have
270  RC_Channel_aux::set_radio_trim(RC_Channel_aux::k_aileron_with_input);
271  RC_Channel_aux::set_radio_trim(RC_Channel_aux::k_elevator_with_input);
272  } else{
273  if (elevon.ch1_temp != 0) {
274  elevon.trim1 = elevon.ch1_temp;
275  }
276  if (elevon.ch2_temp != 0) {
277  elevon.trim2 = elevon.ch2_temp;
278  }
279  //Recompute values here using new values for elevon1_trim and elevon2_trim
280  //We cannot use radio_in[CH_ROLL] and radio_in[CH_PITCH] values from read_radio() because the elevon trim values have changed
281  uint16_t center = 1500;
282  channel_roll->radio_trim = center;
283  channel_pitch->radio_trim = center;
284  }
285  if (channel_rudder->radio_in != 0) {
286  channel_rudder->radio_trim = channel_rudder->radio_in;
287  }
288 
289  // save to eeprom
290  channel_roll->save_eeprom();
291  channel_pitch->save_eeprom();
292  channel_rudder->save_eeprom();
293 }
