66 static int8_t
67 dump_log(uint8_t argc, const Menu::arg *argv)
68 {
69  int16_t dump_log;
70  uint16_t dump_log_start;
71  uint16_t dump_log_end;
72  uint16_t last_log_num;
73 
74  // check that the requested log number can be read
75  dump_log = argv[1].i;
76  last_log_num = DataFlash.find_last_log();
77 
78  if (dump_log == -2) {
79  DataFlash.DumpPageInfo(cliSerial);
80  return(-1);
81  } else if (dump_log <= 0) {
82  cliSerial->printf_P(PSTR("dumping all\n"));
83  Log_Read(0, 1, 0);
84  return(-1);
85  } else if ((argc != 2)
86  || ((uint16_t)dump_log > last_log_num))
87  {
88  cliSerial->printf_P(PSTR("bad log number\n"));
89  return(-1);
90  }
91 
92  DataFlash.get_log_boundaries(dump_log, dump_log_start, dump_log_end);
93  Log_Read((uint16_t)dump_log, dump_log_start, dump_log_end);
94  return 0;
95 }

144  }
145 
146  if (!strcasecmp_P(argv[0].str, PSTR("enable"))) {
147  g.log_bitmask.set_and_save(g.log_bitmask | bits);
