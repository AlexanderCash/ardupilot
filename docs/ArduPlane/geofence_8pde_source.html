87 /*
88  * allocate and fill the geofence state structure
89  */
90 static void geofence_load(void)
91 {
92  uint8_t i;
93 
94  if (geofence_state == NULL) {
95  if (hal.util->available_memory() < 512 + sizeof(struct GeofenceState)) {
96  // too risky to enable as we could run out of stack
97  goto failed;
98  }
99  geofence_state = (struct GeofenceState *)calloc(1, sizeof(struct GeofenceState));
100  if (geofence_state == NULL) {
101  // not much we can do here except disable it
102  goto failed;
103  }
104 
105  geofence_state->boundary = (Vector2l *)calloc(sizeof(Vector2l), max_fencepoints());
106  if (geofence_state->boundary == NULL) {
107  free(geofence_state);
108  geofence_state = NULL;
109  goto failed;
110  }
111 
112  geofence_state->old_switch_position = 254;
113  }
114 
115  if (g.fence_total <= 0) {
116  g.fence_total.set(0);
117  return;
118  }
119 
120  for (i=0; i<g.fence_total; i++) {
121  geofence_state->boundary[i] = get_fence_point_with_index(i);
122  }
123  geofence_state->num_points = i;
124 
125  if (!Polygon_complete(&geofence_state->boundary[1], geofence_state->num_points-1)) {
126  // first point and last point must be the same
127  goto failed;
128  }
129  if (Polygon_outside(geofence_state->boundary[0], &geofence_state->boundary[1], geofence_state->num_points-1)) {
130  // return point needs to be inside the fence
131  goto failed;
132  }
133 
134  geofence_state->boundary_uptodate = true;
135  geofence_state->fence_triggered = false;
136 
137  gcs_send_text_P(SEVERITY_LOW,PSTR("geo-fence loaded"));
138  gcs_send_message(MSG_FENCE_STATUS);
139  return;
140 
141 failed:
142  g.fence_action.set(FENCE_ACTION_NONE);
143  gcs_send_text_P(SEVERITY_HIGH,PSTR("geo-fence setup error"));
144 }

275 /*
276  * check if we have breached the geo-fence
277  */
278 static void geofence_check(bool altitude_check_only)
279 {
280  if (!geofence_enabled()) {
281  // switch back to the chosen control mode if still in
282  // GUIDED to the return point
283  if (geofence_state != NULL &&
284  (g.fence_action == FENCE_ACTION_GUIDED || g.fence_action == FENCE_ACTION_GUIDED_THR_PASS) &&
285  control_mode == GUIDED &&
286  geofence_present() &&
287  geofence_state->boundary_uptodate &&
288  geofence_state->old_switch_position == oldSwitchPosition &&
289  guided_WP_loc.lat == geofence_state->guided_lat &&
290  guided_WP_loc.lng == geofence_state->guided_lng) {
291  geofence_state->old_switch_position = 254;
292  set_mode(get_previous_mode());
293  }
294  return;
295  }
296 
297  /* allocate the geo-fence state if need be */
298  if (geofence_state == NULL || !geofence_state->boundary_uptodate) {
299  geofence_load();
300  if (!geofence_enabled()) {
301  // may have been disabled by load
302  return;
303  }
304  }
305 
306  bool outside = false;
307  uint8_t breach_type = FENCE_BREACH_NONE;
308  struct Location loc;
309 
310  // Never trigger a fence breach in the final stage of landing
311  if (flight_stage == AP_SpdHgtControl::FLIGHT_LAND_FINAL) {
312  return;
313  }
314 
315  if (geofence_state->floor_enabled && geofence_check_minalt()) {
316  outside = true;
317  breach_type = FENCE_BREACH_MINALT;
318  } else if (geofence_check_maxalt()) {
319  outside = true;
320  breach_type = FENCE_BREACH_MAXALT;
321  } else if (!altitude_check_only && ahrs.get_position(loc)) {
322  Vector2l location;
323  location.x = loc.lat;
324  location.y = loc.lng;
325  outside = Polygon_outside(location, &geofence_state->boundary[1], geofence_state->num_points-1);
326  if (outside) {
327  breach_type = FENCE_BREACH_BOUNDARY;
328  }
329  }
330 
331  if (!outside) {
332  if (geofence_state->fence_triggered && !altitude_check_only) {
333  // we have moved back inside the fence
334  geofence_state->fence_triggered = false;
335  gcs_send_text_P(SEVERITY_LOW,PSTR("geo-fence OK"));
336  #if FENCE_TRIGGERED_PIN > 0
337  hal.gpio->pinMode(FENCE_TRIGGERED_PIN, HAL_GPIO_OUTPUT);
338  hal.gpio->write(FENCE_TRIGGERED_PIN, 0);
339  #endif
340  gcs_send_message(MSG_FENCE_STATUS);
341  }
342  // we're inside, all is good with the world
343  return;
344  }
345 
346  // we are outside the fence
347  if (geofence_state->fence_triggered &&
348  (control_mode == GUIDED || g.fence_action == FENCE_ACTION_REPORT)) {
349  // we have already triggered, don't trigger again until the
350  // user disables/re-enables using the fence channel switch
351  return;
352  }
353 
354  // we are outside, and have not previously triggered.
355  geofence_state->fence_triggered = true;
356  geofence_state->breach_count++;
357  geofence_state->breach_time = millis();
358  geofence_state->breach_type = breach_type;
359 
360  #if FENCE_TRIGGERED_PIN > 0
361  hal.gpio->pinMode(FENCE_TRIGGERED_PIN, HAL_GPIO_OUTPUT);
362  hal.gpio->write(FENCE_TRIGGERED_PIN, 1);
363  #endif
364 
365  gcs_send_text_P(SEVERITY_LOW,PSTR("geo-fence triggered"));
366  gcs_send_message(MSG_FENCE_STATUS);
367 
368  // see what action the user wants
369  switch (g.fence_action) {
370  case FENCE_ACTION_REPORT:
371  break;
372 
373  case FENCE_ACTION_GUIDED:
374  case FENCE_ACTION_GUIDED_THR_PASS:
375  if (g.fence_ret_rally != 0) { //return to a rally point
376  guided_WP_loc = rally.calc_best_rally_or_home_location(current_loc, get_RTL_altitude());
377 
378  } else { //return to fence return point, not a rally point
379  if (g.fence_retalt > 0) {
380  //fly to the return point using fence_retalt
381  guided_WP_loc.alt = home.alt + 100.0f*g.fence_retalt;
382  } else if (g.fence_minalt >= g.fence_maxalt) {
383  // invalid min/max, use RTL_altitude
384  guided_WP_loc.alt = home.alt + g.RTL_altitude_cm;
385  } else {
386  // fly to the return point, with an altitude half way between
387  // min and max
388  guided_WP_loc.alt = home.alt + 100.0f*(g.fence_minalt + g.fence_maxalt)/2;
389  }
390  guided_WP_loc.options = 0;
391  guided_WP_loc.lat = geofence_state->boundary[0].x;
392  guided_WP_loc.lng = geofence_state->boundary[0].y;
393  }
394  geofence_state->guided_lat = guided_WP_loc.lat;
395  geofence_state->guided_lng = guided_WP_loc.lng;
396  geofence_state->old_switch_position = oldSwitchPosition;
397 
398  setup_terrain_target_alt(guided_WP_loc);
399 
400  set_guided_WP();
401 
402  // make sure we don't auto trim the surfaces on this mode change
403  int8_t saved_auto_trim = g.auto_trim;
404  g.auto_trim.set(0);
405 
406  set_mode(GUIDED);
407  g.auto_trim.set(saved_auto_trim);
408 
409  if (g.fence_action == FENCE_ACTION_GUIDED_THR_PASS) {
410  guided_throttle_passthru = true;
411  }
412  break;
413  }
