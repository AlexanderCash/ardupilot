312 static void set_mode(enum FlightMode mode)
313 {
314  if(control_mode == mode) {
315  // don't switch modes if we are already in the correct mode.
316  return;
317  }
318  if(g.auto_trim > 0 && control_mode == MANUAL)
319  trim_control_surfaces();
320 
321  // perform any cleanup required for prev flight mode
322  exit_mode(control_mode);
323 
324  // cancel inverted flight
325  auto_state.inverted_flight = false;
326 
327  // don't cross-track when starting a mission
328  auto_state.next_wp_no_crosstrack = true;
329 
330  // reset landing check
331  auto_state.checked_for_autoland = false;
332 
333  // reset go around command
334  auto_state.commanded_go_around = false;
335 
336  // zero locked course
337  steer_state.locked_course_err = 0;
338 
339  // set mode
340  previous_mode = control_mode;
341  control_mode = mode;
342 
343  if (previous_mode == AUTOTUNE && control_mode != AUTOTUNE) {
344  // restore last gains
345  autotune_restore();
346  }
347 
348  // zero initial pitch and highest airspeed on mode change
349  auto_state.highest_airspeed = 0;
350  auto_state.initial_pitch_cd = ahrs.pitch_sensor;
351 
352  // disable taildrag takeoff on mode change
353  auto_state.fbwa_tdrag_takeoff_mode = false;
354 
355  switch(control_mode)
356  {
357  case INITIALISING:
358  auto_throttle_mode = true;
359  break;
360 
361  case MANUAL:
362  case STABILIZE:
363  case TRAINING:
364  case FLY_BY_WIRE_A:
365  auto_throttle_mode = false;
366  break;
367 
368  case AUTOTUNE:
369  auto_throttle_mode = false;
370  autotune_start();
371  break;
372 
373  case ACRO:
374  auto_throttle_mode = false;
375  acro_state.locked_roll = false;
376  acro_state.locked_pitch = false;
377  break;
378 
379  case CRUISE:
380  auto_throttle_mode = true;
381  cruise_state.locked_heading = false;
382  cruise_state.lock_timer_ms = 0;
383  set_target_altitude_current();
384  break;
385 
386  case FLY_BY_WIRE_B:
387  auto_throttle_mode = true;
388  set_target_altitude_current();
389  break;
390 
391  case CIRCLE:
392  // the altitude to circle at is taken from the current altitude
393  auto_throttle_mode = true;
394  next_WP_loc.alt = current_loc.alt;
395  break;
396 
397  case AUTO:
398  auto_throttle_mode = true;
399  next_WP_loc = prev_WP_loc = current_loc;
400  // start or resume the mission, based on MIS_AUTORESET
401  mission.start_or_resume();
402  break;
403 
404  case RTL:
405  auto_throttle_mode = true;
406  prev_WP_loc = current_loc;
407  do_RTL();
408  break;
409 
410  case LOITER:
411  auto_throttle_mode = true;
412  do_loiter_at_location();
413  break;
414 
415  case GUIDED:
416  auto_throttle_mode = true;
417  guided_throttle_passthru = false;
418  /*
419  when entering guided mode we set the target as the current
420  location. This matches the behaviour of the copter code
421  */
422  guided_WP_loc = current_loc;
423  set_guided_WP();
424  break;
425  }
426 
427  // start with throttle suppressed in auto_throttle modes
428  throttle_suppressed = auto_throttle_mode;
429 
430  if (should_log(MASK_LOG_MODE))
431  DataFlash.Log_Write_Mode(control_mode);
432 
433  // reset attitude integrators on mode change
434  rollController.reset_I();
435  pitchController.reset_I();
436  yawController.reset_I();
437  steerController.reset_I();
438 }

476 static void check_long_failsafe()
477 {
478  uint32_t tnow = millis();
479  // only act on changes
480  // -------------------
481  if(failsafe.state != FAILSAFE_LONG && failsafe.state != FAILSAFE_GCS) {
482  if (failsafe.state == FAILSAFE_SHORT &&
483  (tnow - failsafe.ch3_timer_ms) > g.long_fs_timeout*1000) {
484  failsafe_long_on_event(FAILSAFE_LONG);
485  } else if (g.gcs_heartbeat_fs_enabled != GCS_FAILSAFE_OFF &&
486  failsafe.last_heartbeat_ms != 0 &&
487  (tnow - failsafe.last_heartbeat_ms) > g.long_fs_timeout*1000) {
488  failsafe_long_on_event(FAILSAFE_GCS);
489  } else if (g.gcs_heartbeat_fs_enabled == GCS_FAILSAFE_HB_RSSI &&
490  gcs[0].last_radio_status_remrssi_ms != 0 &&
491  (tnow - gcs[0].last_radio_status_remrssi_ms) > g.long_fs_timeout*1000) {
492  failsafe_long_on_event(FAILSAFE_GCS);
493  }
494  } else {
495  // We do not change state but allow for user to change mode
496  if (failsafe.state == FAILSAFE_GCS &&
497  (tnow - failsafe.last_heartbeat_ms) < g.short_fs_timeout*1000) {
498  failsafe.state = FAILSAFE_NONE;
499  } else if (failsafe.state == FAILSAFE_LONG &&
500  !failsafe.ch3_failsafe) {
501  failsafe.state = FAILSAFE_NONE;
502  }
503  }
504 }
