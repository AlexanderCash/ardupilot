428 static bool verify_takeoff()
429 {
430  if (ahrs.yaw_initialised() && steer_state.hold_course_cd == -1) {
431  const float min_gps_speed = 5;
432  if (auto_state.takeoff_speed_time_ms == 0 &&
433  gps.status() >= AP_GPS::GPS_OK_FIX_3D &&
434  gps.ground_speed() > min_gps_speed) {
435  auto_state.takeoff_speed_time_ms = hal.scheduler->millis();
436  }
437  if (auto_state.takeoff_speed_time_ms != 0 &&
438  hal.scheduler->millis() - auto_state.takeoff_speed_time_ms >= 2000) {
439  // once we reach sufficient speed for good GPS course
440  // estimation we save our current GPS ground course
441  // corrected for summed yaw to set the take off
442  // course. This keeps wings level until we are ready to
443  // rotate, and also allows us to cope with arbitary
444  // compass errors for auto takeoff
445  float takeoff_course = wrap_PI(radians(gps.ground_course_cd()*0.01f)) - steer_state.locked_course_err;
446  takeoff_course = wrap_PI(takeoff_course);
447  steer_state.hold_course_cd = wrap_360_cd(degrees(takeoff_course)*100);
448  gcs_send_text_fmt(PSTR("Holding course %ld at %.1fm/s (%.1f)"),
449  steer_state.hold_course_cd,
450  (double)gps.ground_speed(),
451  (double)degrees(steer_state.locked_course_err));
452  }
453  }
454 
455  if (steer_state.hold_course_cd != -1) {
456  // call navigation controller for heading hold
457  nav_controller->update_heading_hold(steer_state.hold_course_cd);
458  } else {
459  nav_controller->update_level_flight();
460  }
461 
462  // see if we have reached takeoff altitude
463  int32_t relative_alt_cm = adjusted_relative_altitude_cm();
464  if (relative_alt_cm > auto_state.takeoff_altitude_rel_cm) {
465  gcs_send_text_fmt(PSTR("Takeoff complete at %.2fm"),
466  (double)(relative_alt_cm*0.01f));
467  steer_state.hold_course_cd = -1;
468  auto_state.takeoff_complete = true;
469  next_WP_loc = prev_WP_loc = current_loc;
470 
471 #if GEOFENCE_ENABLED == ENABLED
472  if (g.fence_autoenable > 0) {
473  if (! geofence_set_enabled(true, AUTO_TOGGLED)) {
474  gcs_send_text_P(SEVERITY_HIGH, PSTR("Enable fence failed (cannot autoenable"));
475  } else {
476  gcs_send_text_P(SEVERITY_HIGH, PSTR("Fence enabled. (autoenabled)"));
477  }
478  }
479 #endif
480 
481  // don't cross-track on completion of takeoff, as otherwise we
482  // can end up doing too sharp a turn
483  auto_state.next_wp_no_crosstrack = true;
484  return true;
485  } else {
486  return false;
487  }
488 }

494 static bool verify_nav_wp(const AP_Mission::Mission_Command& cmd)
495 {
496  steer_state.hold_course_cd = -1;
497 
498  if (auto_state.no_crosstrack) {
499  nav_controller->update_waypoint(current_loc, next_WP_loc);
500  } else {
501  nav_controller->update_waypoint(prev_WP_loc, next_WP_loc);
502  }
503 
504  // see if the user has specified a maximum distance to waypoint
505  if (g.waypoint_max_radius > 0 &&
506  auto_state.wp_distance > (uint16_t)g.waypoint_max_radius) {
507  if (location_passed_point(current_loc, prev_WP_loc, next_WP_loc)) {
508  // this is needed to ensure completion of the waypoint
509  prev_WP_loc = current_loc;
510  }
511  return false;
512  }
513 
514  float acceptance_distance = nav_controller->turn_distance(g.waypoint_radius, auto_state.next_turn_angle);
515  if (cmd.p1 > 0) {
516  // allow user to override acceptance radius
517  acceptance_distance = cmd.p1;
518  }
519 
520  if (auto_state.wp_distance <= acceptance_distance) {
521  gcs_send_text_fmt(PSTR("Reached Waypoint #%i dist %um"),
522  (unsigned)mission.get_current_nav_cmd().index,
523  (unsigned)get_distance(current_loc, next_WP_loc));
524  return true;
525  }
526 
527  // have we flown past the waypoint?
528  if (location_passed_point(current_loc, prev_WP_loc, next_WP_loc)) {
529  gcs_send_text_fmt(PSTR("Passed Waypoint #%i dist %um"),
530  (unsigned)mission.get_current_nav_cmd().index,
531  (unsigned)get_distance(current_loc, next_WP_loc));
532  return true;
533  }
534 
535  return false;
