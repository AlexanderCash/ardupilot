830 void loop()
831 {
832  // wait for an INS sample
833  ins.wait_for_sample();
834 
835  uint32_t timer = hal.scheduler->micros();
836 
837  delta_us_fast_loop = timer - fast_loopTimer_us;
838  G_Dt = delta_us_fast_loop * 1.0e-6f;
839 
840  if (delta_us_fast_loop > G_Dt_max && fast_loopTimer_us != 0) {
841  G_Dt_max = delta_us_fast_loop;
842  }
843 
844  if (delta_us_fast_loop < G_Dt_min || G_Dt_min == 0) {
845  G_Dt_min = delta_us_fast_loop;
846  }
847  fast_loopTimer_us = timer;
848 
849  mainLoop_count++;
850 
851  // tell the scheduler one tick has passed
852  scheduler.tick();
853 
854  // run all the tasks that are due to run. Note that we only
855  // have to call this once per loop, as the tasks are scheduled
856  // in multiples of the main loop tick. So if they don't run on
857  // the first call to the scheduler they won't run on a later
858  // call until scheduler.tick() is called again
859  uint32_t remaining = (timer + 20000) - hal.scheduler->micros();
860  if (remaining > 19500) {
861  remaining = 19500;
862  }
863  scheduler.run(remaining);
864 }

1250  effective_mode = FLY_BY_WIRE_A;

1253  if (effective_mode != AUTO) {

1261  handle_auto_mode();

1265  case LOITER:
1266  case GUIDED:
1267  calc_nav_roll();
1268  calc_nav_pitch();
1269  calc_throttle();
1270  break;
1271 
1272  case TRAINING: {
1273  training_manual_roll = false;
1274  training_manual_pitch = false;
1275

1277  // we set target roll to the limit
1278  if (ahrs.roll_sensor >= roll_limit_cd) {
1279  nav_roll_cd = roll_limit_cd;
1280  } else if (ahrs.roll_sensor <= -roll_limit_cd) {
1281  nav_roll_cd = -roll_limit_cd;

1288  // we set target pitch to the limit

1290  nav_pitch_cd = aparm.pitch_limit_max_cd;
1291  } else if (ahrs.pitch_sensor <= pitch_limit_min_cd) {
1292  nav_pitch_cd = pitch_limit_min_cd;
1293  } else {
1294  training_manual_pitch = true;
1295  nav_pitch_cd = 0;
1296  }
1297  if (fly_inverted()) {
1298  nav_pitch_cd = -nav_pitch_cd;
1299  }
1300  break;
1301  }
1302 
1303  case ACRO: {
1304  // handle locked/unlocked control
1305  if (acro_state.locked_roll) {
1306  nav_roll_cd = acro_state.locked_roll_err;
1307  } else {
1308  nav_roll_cd = ahrs.roll_sensor;
1309  }

1312  } else {
1313  nav_pitch_cd = ahrs.pitch_sensor;
1314  }
1315  break;
1316  }
1317

1319  case FLY_BY_WIRE_A: {
1320  // set nav_roll and nav_pitch using sticks
1321  nav_roll_cd = channel_roll->norm_input() * roll_limit_cd;
1322  nav_roll_cd = constrain_int32(nav_roll_cd, -roll_limit_cd, roll_limit_cd);
1323  update_load_factor();

1325  if (pitch_input > 0) {
1326  nav_pitch_cd = pitch_input * aparm.pitch_limit_max_cd;
1327  } else {
1328  nav_pitch_cd = -(pitch_input * pitch_limit_min_cd);
1329  }

1332  if (fly_inverted()) {
1333  nav_pitch_cd = -nav_pitch_cd;
1334  }

1337  nav_roll_cd = 0;
1338  nav_pitch_cd = 0;
1339  channel_throttle->servo_out = 0;
1340  }

1346  auto_state.fbwa_tdrag_takeoff_mode = true;

1348  }
1349  }
1350  }
1351  break;
1352  }
1353 
1354  case FLY_BY_WIRE_B:

1356  nav_roll_cd = channel_roll->norm_input() * roll_limit_cd;
1357  nav_roll_cd = constrain_int32(nav_roll_cd, -roll_limit_cd, roll_limit_cd);
1358  update_load_factor();
1359  update_fbwb_speed_height();
1360  break;
1361 
1362  case CRUISE:

1364  in CRUISE mode we use the navigation code to control
1365  roll when heading is locked. Heading becomes unlocked on
1366  any aileron or rudder input
1367  */
1368  if ((channel_roll->control_in != 0 ||
1369  rudder_input != 0)) {
1370  cruise_state.locked_heading = false;
1371  cruise_state.lock_timer_ms = 0;

1378  } else {

1380  }
1381  update_fbwb_speed_height();
1382  break;
1383 
1384  case STABILIZE:

1386  nav_pitch_cd = 0;
1387  // throttle is passthrough
1388  break;
1389 
1390  case CIRCLE:

1392  // or we just want to fly around in a gentle circle w/o GPS,
1393  // holding altitude at the altitude we set when we
1394  // switched into the mode
1395  nav_roll_cd = roll_limit_cd / 3;

1403  // ---------------------------------
1404  channel_roll->servo_out = channel_roll->pwm_to_angle();
1405  channel_pitch->servo_out = channel_pitch->pwm_to_angle();
1406  steering_control.steering = steering_control.rudder = channel_rudder->pwm_to_angle();
1407  break;

1409 
1410  case INITIALISING:
1411  // handled elsewhere
