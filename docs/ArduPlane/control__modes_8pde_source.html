6  static bool switch_debouncer;
7  uint8_t switchPosition = readSwitch();
8 
9  // If switchPosition = 255 this indicates that the mode control channel input was out of range
10  // If we get this value we do not want to change modes.
11  if(switchPosition == 255) return;
12 
13  if (failsafe.ch3_failsafe || failsafe.ch3_counter > 0) {
14  // when we are in ch3_failsafe mode then RC input is not
15  // working, and we need to ignore the mode switch channel
16  return;
17  }
18 
19  if (hal.scheduler->millis() - failsafe.last_valid_rc_ms > 100) {
20  // only use signals that are less than 0.1s old.
21  return;
22  }
23 
24  // we look for changes in the switch position. If the
25  // RST_SWITCH_CH parameter is set, then it is a switch that can be
26  // used to force re-reading of the control switch. This is useful
27  // when returning to the previous mode after a failsafe or fence
28  // breach. This channel is best used on a momentary switch (such
29  // as a spring loaded trainer switch).
30  if (oldSwitchPosition != switchPosition ||
31  (g.reset_switch_chan != 0 &&
32  hal.rcin->read(g.reset_switch_chan-1) > RESET_SWITCH_CHAN_PWM)) {
33 
34  if (switch_debouncer == false) {
35  // this ensures that mode switches only happen if the
36  // switch changes for 2 reads. This prevents momentary
37  // spikes in the mode control channel from causing a mode
38  // switch
39  switch_debouncer = true;
40  return;
41  }
42 
43  set_mode((enum FlightMode)(flight_modes[switchPosition].get()));
44 
45  oldSwitchPosition = switchPosition;
46  prev_WP_loc = current_loc;
47  }
48 
49  if (g.reset_mission_chan != 0 &&
50  hal.rcin->read(g.reset_mission_chan-1) > RESET_SWITCH_CHAN_PWM) {
51  mission.start();
52  prev_WP_loc = current_loc;
53  }
54 
55  switch_debouncer = false;
56

58  // if the user has configured an inverted flight channel, then
59  // fly upside down when that channel goes above INVERTED_FLIGHT_PWM
60  inverted_flight = (control_mode != MANUAL && hal.rcin->read(g.inverted_flight_ch-1) > INVERTED_FLIGHT_PWM);
61  }

104 static uint8_t readSwitch(void)
105 {
106  uint16_t pulsewidth = hal.rcin->read(g.flight_mode_channel - 1);
107  if (pulsewidth <= 900 || pulsewidth >= 2200) return 255; // This is an error condition
108  if (pulsewidth > 1230 && pulsewidth <= 1360) return 1;
109  if (pulsewidth > 1360 && pulsewidth <= 1490) return 2;
110  if (pulsewidth > 1490 && pulsewidth <= 1620) return 3;
111  if (pulsewidth > 1620 && pulsewidth <= 1749) return 4; // Software Manual
112  if (pulsewidth >= 1750) return 5; // Hardware Manual
113  return 0;
114 }
