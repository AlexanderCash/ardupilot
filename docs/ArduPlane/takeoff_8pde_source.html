3 /* Check for automatic takeoff conditions being met using the following sequence:
4  * 1) Check for adequate GPS lock - if not return false
5  * 2) Check the gravity compensated longitudinal acceleration against the threshold and start the timer if true
6  * 3) Wait until the timer has reached the specified value (increments of 0.1 sec) and then check the GPS speed against the threshold
7  * 4) If the GPS speed is above the threshold and the attitude is within limits then return true and reset the timer
8  * 5) If the GPS speed and attitude within limits has not been achieved after 2.5 seconds, return false and reset the timer
9  * 6) If the time lapsed since the last timecheck is greater than 0.2 seconds, return false and reset the timer
10  * NOTE : This function relies on the TECS 50Hz processing for its acceleration measure.
11  */
12 static bool auto_takeoff_check(void)
13 {
14  // this is a more advanced check that relies on TECS
15  uint32_t now = hal.scheduler->millis();
16  static bool launchTimerStarted;
17  static uint32_t last_tkoff_arm_time;
18  static uint32_t last_check_ms;
19  uint16_t wait_time_ms = min(uint16_t(g.takeoff_throttle_delay)*100,12700);
20 
21  // Reset states if process has been interrupted
22  if (last_check_ms && (now - last_check_ms) > 200) {
23  gcs_send_text_fmt(PSTR("Timer Interrupted AUTO"));
24  launchTimerStarted = false;
25  last_tkoff_arm_time = 0;
26  last_check_ms = now;
27  return false;
28  }
29 
30  last_check_ms = now;
31 
32  // Check for bad GPS
33  if (gps.status() < AP_GPS::GPS_OK_FIX_3D) {
34  // no auto takeoff without GPS lock
35  return false;
36  }

39  if (!launchTimerStarted &&
40  !is_zero(g.takeoff_throttle_min_accel) &&

42  goto no_launch;
43  }
44 
45  // we've reached the acceleration threshold, so start the timer
46  if (!launchTimerStarted) {
47  launchTimerStarted = true;
48  last_tkoff_arm_time = now;
49  gcs_send_text_fmt(PSTR("Armed AUTO, xaccel = %.1f m/s/s, waiting %.1f sec"),
50  (double)SpdHgt_Controller->get_VXdot(), (double)(wait_time_ms*0.001f));
51  }
52

55  gcs_send_text_fmt(PSTR("Timeout AUTO"));
56  goto no_launch;
57  }
58

60  if (ahrs.pitch_sensor <= -3000 ||

62  abs(ahrs.roll_sensor) > 3000) {
63  gcs_send_text_fmt(PSTR("Bad Launch AUTO"));
64  goto no_launch;
65  }
66

69  ((now - last_tkoff_arm_time) >= wait_time_ms)) {
70  gcs_send_text_fmt(PSTR("Triggered AUTO, GPSspd = %.1f"), (double)gps.ground_speed());
71  launchTimerStarted = false;
72  last_tkoff_arm_time = 0;
73  return true;
74  }
75 
76  // we're not launching yet, but the timer is still going
77  return false;
78 
79 no_launch:
80  launchTimerStarted = false;
81  last_tkoff_arm_time = 0;
82  return false;

107  */
108 static void takeoff_calc_pitch(void)
109 {
110  if (auto_state.highest_airspeed < g.takeoff_rotate_speed) {
111  // we have not reached rotate speed, use a target pitch of 5
112  // degrees. This should be enough to get the tail off the
113  // ground, while making it unlikely that overshoot in the
114  // pitch controller will cause a prop strike
115  nav_pitch_cd = 500;
116  return;
117  }
118 
119  if (ahrs.airspeed_sensor_enabled()) {
120  calc_nav_pitch();
121  if (nav_pitch_cd < auto_state.takeoff_pitch_cd) {
122  nav_pitch_cd = auto_state.takeoff_pitch_cd;
123  }
124  } else {

126  nav_pitch_cd = constrain_int32(nav_pitch_cd, 500, auto_state.takeoff_pitch_cd);

140  (control_mode == FLY_BY_WIRE_A && auto_state.fbwa_tdrag_takeoff_mode));
141  if (!in_takeoff) {
142  // not in takeoff
143  return 0;
144  }
145  if (g.takeoff_tdrag_elevator == 0) {
146  // no takeoff elevator set
147  goto return_zero;
148  }
149  if (auto_state.highest_airspeed >= g.takeoff_tdrag_speed1) {
150  // we've passed speed1. We now raise the tail and aim for
151  // level pitch. Return 0 meaning no fixed elevator setting
152  goto return_zero;
153  }
154  if (ahrs.pitch_sensor > auto_state.initial_pitch_cd + 1000) {
155  // the pitch has gone up by more then 10 degrees over the
156  // initial pitch. This may mean the nose is coming up for an
157  // early liftoff, perhaps due to a bad setting of
158  // g.takeoff_tdrag_speed1. Go to level flight to prevent a
159  // stall
160  goto return_zero;
161  }
162  // we are holding the tail down
163  return g.takeoff_tdrag_elevator;
164 
165 return_zero:

167  gcs_send_text_P(SEVERITY_LOW, PSTR("FBWA tdrag off"));
168  auto_state.fbwa_tdrag_takeoff_mode = false;
169  }
